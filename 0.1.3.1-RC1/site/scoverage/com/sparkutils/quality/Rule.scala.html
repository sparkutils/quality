35 case class OutputExpression( rule: String ) extends OutputExprLogic with HasRuleText with Logging {
36   private[quality] override def expression() = {
37     val parsed = RuleLogicUtils.expr(rule)
38     // output expressions can be:
39     // 1. simple expressions for ruleEngine
40     // 2. single argument lambda's returning the same type as the arg for folder
41     // 3. as of 0.0.2 #8 set( attribute = valueExpression, attribute = valueExpression) converted to the form of 2 with an updateField call
42     parsed match {
43       case uf: UnresolvedFunction if toName(uf) == "set" =>
44         // case 3
45         val args = arguments(uf)
46         val paired =
47           args.flatMap {
48             case EqualTo(name: UnresolvedAttribute, right) =>
49               // updateField takes paired args of field names to expression
50               Some(Seq(Literal(name.name), right))
51             case a =>
52               logInfo(s"Attempt to convert set OutputExpression argument $a failed as types do not match expected EqualTo(attribute, expression), will default to full expression")
53               None
54           }
55 
56         if (paired.size != args.size)
57           // one of the args didn't match type
58           parsed
59         else
60           // need to keep first arg
61           UpdateFolderExpression.withArgsAndSubstitutedLambdaVariable(paired.flatten)
62       case _ =>
63         // for everything else (1+2) it's already good enough
64         parsed
65     }
66   }
