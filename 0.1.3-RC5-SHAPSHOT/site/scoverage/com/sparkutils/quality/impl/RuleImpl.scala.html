54   /**
55    * Same as functions.expr without the wrapping Column
56    * @param rule
57    * @return
58    */
59   def expr(rule: String) = {
60     val parser = SparkSession.getActiveSession.map(_.sessionState.sqlParser).getOrElse {
61       newParser()
62     }
63     /*
64     attempt for a simple expression, then try a plan with exactly one output row.
65     "In" will handle ListQuery, "Exists" similarly exists, everything else will likely fail as a parser error
66      */
67     val rawExpr =
68       try {
69         parser.parseExpression(rule)
70       } catch {
71         case ot: Throwable => // e.g. suitable for output expressions
72           try {
73             ScalarSubquery(parser.parsePlan(rule))
74           } catch {
75             case _: Throwable =>
76               // quite possibly a lambda using a subquery so try and force it via ( sub ) trick,
77               // but allow us to replace later after resolution.  DBR > 12 requires () on simple expressions as well
78               val r = rule.split("->")
79               val wrapped =
80                 if (r.size == 2)
81                   s"${r(0)} -> ( ${r(1)} )"
82                 else
83                   s"( ${r(0)} )"
84 
85               try {
86                 parser.parseExpression(wrapped)
87               } catch {
88                 case _: Throwable => throw ot // if this didn't work return the original error
89               }
90           }
91       }
92 
93     val res =
94       rawExpr match {
95         case l: SparkLambdaFunction if hasSubQuery(l) =>
96           // The lambda's will be parsed as UnresolvedAttributes and not the needed lambdas
97           val names = l.arguments.map(a => a.name -> a).toMap
98           l.transform {
99             case s: SubqueryExpression => s.withNewPlan( s.plan.transform{
100               case snippet =>
101                 snippet.transformAllExpressions {
102                   case a: UnresolvedAttribute =>
103                     names.get(a.name).map(lamVar => lamVar).getOrElse(a)
104                 }
105             })
106           }
107         case _ => rawExpr
108       }
109 
110     res
111   }

126     any match {
