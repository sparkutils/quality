31   /**
32    * Calculate optimal size according to the number of distinct values and false positive probability.
33    *
34    * @param n : The number of distinct values.
35    * @param p : The false positive probability.
36    * @return optimal number of bits of given n and p.
37    */
38   def optimalNumOfBits(n: Long, p: Double): Int = {
39     require(p > 0.0 && p < 1.0, "FPP should be less than 1.0 and great than 0.0")
40     val m = -8 * n / Math.log(1 - Math.pow(p, 1.0 / 8))
41     var numBits = m.toInt // 2147483647
42     // Handle overflow.
43     if (numBits > (UPPER_BOUND_BYTES << 3) || m < 0) {
44       numBits = UPPER_BOUND_BYTES << 3 // 1073741824
45     }
46 
47     // Round numBits up to (k * BITS_PER_BLOCK)
48     numBits = (numBits + BITS_PER_BLOCK - 1) & ~BITS_PER_BLOCK // 1073742079
49 
50     if (numBits < (LOWER_BOUND_BYTES << 3)) {
51       numBits = LOWER_BOUND_BYTES << 3
52     }
53     numBits
54   }

61       m.toLong

65     var numBytes = optimalNumOfBits(n, p) / 8 // 134217759

70     if (numBytes > origBytes)
71       (numBytes.toDouble / origBytes.toDouble).ceil.toLong
72     else
73       (origBytes.toDouble / numBytes.toDouble).ceil.toLong

78 /*
79  * This Bloom filter is implemented using block-based Bloom filter algorithm from Putze et al.'s
80  * "Cache-, Hash- and Space-Efficient Bloom filters". The basic idea is to hash the item to a tiny
81  * Bloom filter which size fit a single cache line or smaller. This implementation sets 8 bits in
82  * each tiny Bloom filter. Each tiny Bloom filter is 32 bytes to take advantage of 32-byte SIMD
83  * instruction.
84  */
85 object BlockSplitBloomFilterImpl { // Bytes in a tiny Bloom filter block.
86   private[parquet] val BYTES_PER_BLOCK = 32
87   // Bits in a tiny Bloom filter block.
88   private[parquet] val BITS_PER_BLOCK = 256
89   // The lower bound of bloom filter size, set to the size of a tiny Bloom filter block.
90   val LOWER_BOUND_BYTES = 32
91   // The upper bound of bloom filter size, set to default row group size.
92   val UPPER_BOUND_BYTES: Int = 128 * 1024 * 1024
93   // The number of bits to set in a tiny Bloom filter
94   private[parquet] val BITS_SET_PER_BLOCK = 8
95   // The metadata in the header of a serialized Bloom filter is four four-byte values: the number of bytes,
96   // the filter algorithm, the hash algorithm, and the compression.

245   def insertHash(hash: Long): Unit = {
246     val numBlocks = getBitsetSize / BlockSplitBloomFilterImpl.BYTES_PER_BLOCK
247     val lowHash = hash >>> 32
248     val blockIndex = ((lowHash * numBlocks) >> 32).toInt
249     val key = hash.toInt
250     // Calculate mask for bucket.
251     val mask = setMask(key)
252     for (i <- 0 until BlockSplitBloomFilterImpl.BITS_SET_PER_BLOCK) {
253       var value = intBuffer.get(blockIndex * (BlockSplitBloomFilterImpl.BYTES_PER_BLOCK / 4) + i)
254       value |= mask(i)
255       intBuffer.put(blockIndex * (BlockSplitBloomFilterImpl.BYTES_PER_BLOCK / 4) + i, value)
256     }
257   }
