39   /**
40    * Are there any stateful expressions in interpreted mode (or fallback) require fresh copies
41    * @param exprs These expressions must be fully resolved, bound and optimised
42    * @param compile when false any stateful expression returns true, by default compilation for stateful CodegenFallback is true
43    * @return
44    */
45   def isCopyNeeded(exprs: Seq[Expression], compile: Boolean = true): Boolean =
46     exprs.collect {
47       case e: CodegenFallback if ShimUtils.isStateful(e)  => e
48     }.nonEmpty || ( !compile &&
49       exprs.collect {
50         case e: Expression if ShimUtils.isStateful(e)  => e
51       }.nonEmpty
52     )

54   /**
55    * Arbitrarily high number assuming 2 java fields per input fields and a couple of functions per field we'll
56    * hit the 65k class namespace limit.
57    */
58   val maxVarCompilationInputFields = 12000

60   /**
61    * Creates a ProcessFactory[I,O], when forceMutable || !compile is true it uses chained MutableProjections,
62    * otherwise, and by default, GenerateDecoderOpEncoderProjection.
63    *
64    * GenerateDecoderOpEncoderProjection is likely far faster on larger rules or higher number of threads for throughput.
65    * Unlike Spark compilation caching, which evaluates codegen for an expression tree every time, the codegen takes place
66    * once, each instance call simply returns a new class instance with fresh state for a thread.  If using forceVarCompilation,
67    * once the number of input fields goes beyond [[maxVarCompilationInputFields]] the code will fallback to INPUT_ROW based processing.
68    *
69    * The chained MutableProjections generate new class code for each instance.  They will also recreate the entire
70    * expression tree if a stateful expression is identified when compile = false
71    *
72    * @param dataFrameFunction
73    * @param compile when false reverts to interpreted mode, when true and forceMutable is true it is recommended to cache the instances
74    * @param forceMutable when true it forces MutableProjection's to be used, compiled or otherwise, the default of false is likely far faster
75    * @param forceVarCompilation defaulting to false it will, when compile is true and forceMutable is false, use variables rather than INPUT_ROW to generate code.
76    *                            Although using true is faster past 10 fields, the compilation approach (as of 0.1.3.1) is experimental
77    *                            and only supported on version OSS 3.2.1 and above
78    * @param toSize specifies the number of fields required to deserialize and create the [[O]]
79    * @tparam I
80    * @tparam O
81    * @return
82    */
83   def processFactory[I: Encoder, O: Encoder](dataFrameFunction: DataFrame => DataFrame, toSize: Int, compile: Boolean = true,
84       forceMutable: Boolean = false, forceVarCompilation: Boolean = false, extraProjection: DataFrame => DataFrame = identity,
85       enableQualityOptimisations: Boolean = true): ProcessorFactory[I, O] = {
86     if (forceMutable || !compile)
87       MutableProjectionProcessor.processFactory[I, O](dataFrameFunction, toSize, compile, extraProjection,
88         enableQualityOptimisations = enableQualityOptimisations)
89     else {
90       if (enableQualityOptimisations) {
91         enableOptimizations(Seq(FunNRewrite, ConstantFolding))
92       }
93 
94       val iEnc = implicitly[Encoder[I]]
95       val exprs = QualitySparkUtils.resolveExpressions[I](iEnc, df => {
96         dataFrameFunction(extraProjection(df))
97       })
98 
99       // the code references to the other fields is already present inside of expressions, works for input_row based,
100       // but not wholestage approach, this is performed by all the CodegenSupport execs via attribute lookups
101       val exprsToUse = exprs.drop( exprs.length - toSize)
102 
103       val allOrdinals =
104         exprsToUse.flatMap{
105           e => e.collect {
106             case b: BoundReference => b.ordinal
107           }
108         }.distinct.toSet
109 
110       val projector =
111         if (forceVarCompilation && allOrdinals.size < maxVarCompilationInputFields)
112           GenerateDecoderOpEncoderVarProjection.create[I, O](exprs, toSize, allOrdinals)
113         else
114           GenerateDecoderOpEncoderProjection.generate[I, O](exprs, useSubexprElimination = true, toSize)
115       new ProcessorFactory[I, O] {
116         override def instance: Processor[I, O] = new Processor[I, O] {
117           private val theInstance = projector.newInstance
118           override def apply(i: I): O = theInstance(i)
119           override def setPartition(partition: Int): Unit = theInstance.initialize(partition)
120           override def close(): Unit = {}
121         }
122       }
123     }
124   }
