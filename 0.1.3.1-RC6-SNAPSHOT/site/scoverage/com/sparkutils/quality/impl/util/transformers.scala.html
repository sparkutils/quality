10   /**
11    * Leverages the foldRunner to replace fields, the input fields are used to create a structure that the rules fold over.
12    * The fields are then dropped from the original Dataframe and added back from the resulting structure.
13    *
14    * NOTE: The field order and types of the original DF will be maintained only when maintainOrder is true.  As it requires access to the schema it may incur extra work.
15    *
16    * @param rules
17    * @param debugMode when true the last results are taken for the replaced fields
18    * @param maintainOrder when true the schema is used to replace fields in the correct location, when false they are simply appended
19    * @param useType In the case you must use select and can't use withColumn you may provide a type directly to stop the NPE
20    * @return
21    */
22   def ifoldAndReplaceFields[P[R] >: DatasetBase[R]](rules: RuleSuite, fields: Seq[String], foldFieldName: String = "foldedFields",
23                            debugMode: Boolean = false, tempFoldDebugName: String = "tempFOLDDEBUG",
24                            maintainOrder: Boolean = true, useType: Option[StructType] = None,
25                            compileEvals: Boolean = false, forceRunnerEval: Boolean = false,
26                            forceTriggerEval: Boolean = false): P[SRow] => P[SRow] = rdf => {
27     val df = rdf.asInstanceOf[DataFrame]
28     import org.apache.spark.sql.functions._
29 
30     val theStruct = struct(fields.head, fields.tail :_*)
31     val withFolder = {
32       // select NPEs needs projection to work
33       //df.select(expr("*"), ruleFolderRunner(rules, theStruct).as(foldFieldName))
34       df.withColumn(foldFieldName, ruleFolderRunner(rules, theStruct, debugMode = debugMode, useType = useType,
35         compileEvals = compileEvals, forceRunnerEval = forceRunnerEval, forceTriggerEval = forceTriggerEval) )
36     }
37 
38     // create now as the schema will have the folder, which we may want to keep
39     val namesInOrder =
40       if (maintainOrder)
41         withFolder.schema.map(_.name)
42       else
43         Seq()
44     // lift the results
45     val result =
46       if (debugMode)
47         withFolder.drop(fields : _*).selectExpr("*",
48           s"if(size($foldFieldName.result) == 0 or $foldFieldName.result is null, null, element_at($foldFieldName.result, -1)).result as $tempFoldDebugName"
49         ).selectExpr("*", s"$tempFoldDebugName.*").drop(tempFoldDebugName)
50       else
51         withFolder.drop(fields : _*).selectExpr("*", s"$foldFieldName.result.*" )
52 
53     // bring back to top level in the correct order
54     if (maintainOrder)
55       result.select(namesInOrder.head, namesInOrder.tail :_*)
56     else
57       result
