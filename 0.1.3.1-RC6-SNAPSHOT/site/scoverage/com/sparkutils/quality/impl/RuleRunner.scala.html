44   /**
45    * Creates a column that runs the RuleSuite.  This also forces registering the lambda functions used by that RuleSuite
46    *
47    * @param ruleSuite The Qualty RuleSuite to evaluate
48    * @param compileEvals Should the rules be compiled out to interim objects - by default false, allowing optimisations
49    * @param resolveWith This experimental parameter can take the DataFrame these rules will be added to and pre-resolve and optimise the sql expressions, see the documentation for details on when to and not to use this. RuleRunner does not currently do wholestagecodegen when resolveWith is used.
50    * @param variablesPerFunc Defaulting to 40, it allows, in combination with variableFuncGroup customisation of handling the 64k jvm method size limitation when performing WholeStageCodeGen.  You _shouldn't_ need it but it's there just in case.
51    * @param variableFuncGroup Defaulting to 20
52    * @param forceRunnerEval Defaulting to false, passing true forces a simplified partially interpreted evaluation (compileEvals must be false to get fully interpreted)
53    * @return A Column representing the Quality DQ expression built from this ruleSuite
54    */
55   def ruleRunnerImpl(ruleSuite: RuleSuite, compileEvals: Boolean = false, resolveWith: Option[DataFrame] = None,
56                      variablesPerFunc: Int = 40, variableFuncGroup: Int = 20, forceRunnerEval: Boolean = false): Column = {
57     com.sparkutils.quality.registerLambdaFunctions( ruleSuite.lambdaFunctions )
58     val flattened = flattenExpressions(ruleSuite)
59     val cleaned = RuleLogicUtils.cleanExprs(ruleSuite)
60     val input =
61       // ExpressionProxy and SubExprEvaluationRuntime cannot be used with compileEvals
62       if (compileEvals)
63         PassThroughCompileEvals( flattened )
64       else
65         PassThroughEvalOnly( flattened )
66 
67     val runner =
68       if (forceRunnerEval || resolveWith.isDefined)
69         new RuleRunnerEval(cleaned, input,
70           compileEvals, variablesPerFunc, variableFuncGroup)
71       else
72         new RuleRunner(cleaned, input,
73           compileEvals, variablesPerFunc, variableFuncGroup)
74 
75     column(
76       QualitySparkUtils.resolveWithOverride(resolveWith).map { df =>
77         val resolved = QualitySparkUtils.resolveExpression(df, runner)
78 
79         resolved.withNewChildren(Seq(resolved.children.head match {
80           // replace the expr
81           case PassThroughCompileEvals(children) => NonPassThrough(children)
82           case PassThroughEvalOnly(children) => NonPassThrough(children)
83         }))
84       } getOrElse runner
85     )
86   }
