44   /**
45    * Creates a column that runs the RuleSuite.  This also forces registering the lambda functions used by that RuleSuite
46    *
47    * @param ruleSuite The Qualty RuleSuite to evaluate
48    * @param compileEvals Should the rules be compiled out to interim objects - by default true for eval usage, wholeStageCodeGen will evaluate in place
49    * @param resolveWith This experimental parameter can take the DataFrame these rules will be added to and pre-resolve and optimise the sql expressions, see the documentation for details on when to and not to use this. RuleRunner does not currently do wholestagecodegen when resolveWith is used.
50    * @param variablesPerFunc Defaulting to 40, it allows, in combination with variableFuncGroup customisation of handling the 64k jvm method size limitation when performing WholeStageCodeGen.  You _shouldn't_ need it but it's there just in case.
51    * @param variableFuncGroup Defaulting to 20
52    * @param forceRunnerEval Defaulting to false, passing true forces a simplified partially interpreted evaluation (compileEvals must be false to get fully interpreted)
53    * @return A Column representing the Quality DQ expression built from this ruleSuite
54    */
55   def ruleRunnerImpl(ruleSuite: RuleSuite, compileEvals: Boolean = true, resolveWith: Option[DataFrame] = None, variablesPerFunc: Int = 40, variableFuncGroup: Int = 20, forceRunnerEval: Boolean = false): Column = {
56     com.sparkutils.quality.registerLambdaFunctions( ruleSuite.lambdaFunctions )
57     val flattened = flattenExpressions(ruleSuite)
58     val cleaned = RuleLogicUtils.cleanExprs(ruleSuite)
59     val input =
60       // ExpressionProxy and SubExprEvaluationRuntime cannot be used with compileEvals
61       if (compileEvals)
62         PassThroughCompileEvals( flattened )
63       else
64         PassThroughEvalOnly( flattened )
65 
66     val runner =
67       if (forceRunnerEval || resolveWith.isDefined)
68         new RuleRunnerEval(cleaned, input,
69           compileEvals, variablesPerFunc, variableFuncGroup)
70       else
71         new RuleRunner(cleaned, input,
72           compileEvals, variablesPerFunc, variableFuncGroup)
73 
74     column(
75       QualitySparkUtils.resolveWithOverride(resolveWith).map { df =>
76         val resolved = QualitySparkUtils.resolveExpression(df, runner)
77 
78         resolved.withNewChildren(Seq(resolved.children.head match {
79           // replace the expr
80           case PassThroughCompileEvals(children) => NonPassThrough(children)
81           case PassThroughEvalOnly(children) => NonPassThrough(children)
82         }))
83       } getOrElse runner
84     )
85   }
