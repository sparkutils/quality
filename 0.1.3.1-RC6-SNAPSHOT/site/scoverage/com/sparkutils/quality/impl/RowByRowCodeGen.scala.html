13 /**
14  * CODE is based on MutableProjection and generates a transformation between two encoders over a middle operation.
15  * The generated class itself can create new instances directly, unlike Spark projections that need to go through the
16  * codegen source generation cycle, only the compilation is cached.
17  *
18  * If the expression tree contains stateful expressions with codegenfallback the code must be regenerated against a
19  * fresh tree.
20  */

36   def projections(ctx: CodegenContext, expressions: Seq[Expression], mutableRow: String, useSubexprElimination: Boolean = false) = {
37     val validExpr = expressions.zipWithIndex.filter {
38       case (NoOp, _) => false
39       case _ => true
40     }
41     val exprVals = ctx.generateExpressions(validExpr.map(_._1), useSubexprElimination)
42 
43     // 4-tuples: (code for projection, isNull variable name, value variable name, column index)
44     val projectionCodes: Seq[(String, String)] = validExpr.zip(exprVals).map {
45       case ((e, i), ev) =>
46         val value = JavaCode.global(
47           ctx.addMutableState(CodeGenerator.javaType(e.dataType), "value"),
48           e.dataType)
49         val (code, isNull) = if (e.nullable) {
50           val isNull = ctx.addMutableState(CodeGenerator.JAVA_BOOLEAN, "isNull")
51           (s"""
52               |${ev.code}
53               |$isNull = ${ev.isNull};
54               |$value = ${ev.value};
55             """.stripMargin, JavaCode.isNullGlobal(isNull))
56         } else {
57           (s"""
58               |${ev.code}
59               |$value = ${ev.value};
60             """.stripMargin, FalseLiteral)
61         }
62         val update = CodeGenerator.updateColumn(
63           mutableRow,
64           e.dataType,
65           i,
66           ExprCode(isNull, value),
67           e.nullable)
68         (code, update)
69     }
70     projectionCodes
71   }

73   private def create[I: Encoder, O: Encoder](
74                       expressions: Seq[Expression],
75                       useSubexprElimination: Boolean): DecoderOpEncoderProjection[I,O] = {
76 
77     val iEnc = implicitly[Encoder[I]]
78     val exprFrom = ShimUtils.expressionEncoder(iEnc).resolveAndBind().serializer
79     val exprTo = ShimUtils.expressionEncoder(implicitly[Encoder[O]]).resolveAndBind().deserializer
80 
81     val hasSubQuery =
82       expressions.map(_.collectFirst {
83         case s: PlanExpression[_] => true
84       }.getOrElse(false))
85 
86     if (hasSubQuery.contains(true)) {

90     val toSize =
91       exprTo.dataType match {
92         case s: StructType => s.length

94       }
95 
96     val resTypeIsStruct = expressions(exprFrom.length).dataType.isInstanceOf[StructType]

98       if (resTypeIsStruct)

100       else
101         null
102     val offset = exprFrom.length
103 
104     val ctx = newCodeGenContext()
105 
106     val projectionCodes = projections(ctx, expressions, "mutableRow", useSubexprElimination)
107 
108     // Evaluate all the subexpressions.
109     val evalSubexpr = ctx.subexprFunctionsCode
110 
111     val allProjections = ctx.splitExpressionsWithCurrentInputs(projectionCodes.map(_._1))
112     val allUpdates = ctx.splitExpressionsWithCurrentInputs(projectionCodes.map(_._2))
113 
114     ctx.INPUT_ROW = "enc"
115     val encProjectionCodes = projections(ctx, exprFrom, "encRow")
116 
117     val encProjections = ctx.splitExpressionsWithCurrentInputs(encProjectionCodes.map(_._1))
118     val encUpdates = ctx.splitExpressionsWithCurrentInputs(encProjectionCodes.map(_._2))

120     ctx.INPUT_ROW = "dec"
121     val decProjectionCodes = projections(ctx, Seq(exprTo), "decRow")
122 
123     val decProjections = ctx.splitExpressionsWithCurrentInputs(decProjectionCodes.map(_._1))
124     val decUpdates = ctx.splitExpressionsWithCurrentInputs(decProjectionCodes.map(_._2))
125 
126     val codeBody = s"""
127       public java.lang.Object generate(Object[] references) {
128         return new SpecificMutableProjection(references);
129       }
130 
131       class SpecificMutableProjection extends ${classOf[DecoderOpEncoderProjection[I,O]].getName}<${iEnc.clsTag.runtimeClass.getName}, ${implicitly[Encoder[O]].clsTag.runtimeClass.getName}> {
132 
133         private Object[] references;
134         private InternalRow inRow;
135         private InternalRow mutableRow;
136         private InternalRow encRow;
137         private InternalRow decRow;
138         private InternalRow interim;
139         ${ctx.declareMutableStates()}
140 
141         public SpecificMutableProjection(Object[] references) {
142           this.references = references;
143           mutableRow = new $genericMutableRowType(${expressions.size});
144           inRow = new $genericMutableRowType(1);
145           encRow = new $genericMutableRowType(${exprFrom.length});
146           decRow = new $genericMutableRowType(${toSize});
147           ${
148             if (expressions.length - offset == 1 && resTypeIsStruct)
149               ""
150             else
151               s"interim = new $genericMutableRowType(${expressions.length - offset});"
152           }
153           ${ctx.initMutableStates()}
154         }
155 
156         public ${classOf[DecoderOpEncoderProjection[I,O]].getName} newInstance() {
157           return new SpecificMutableProjection(references);
158         }
159 
160         public void initialize(int partitionIndex) {
161           ${ctx.initPartition()}
162         }
163 
164         //public ${implicitly[Encoder[O]].clsTag.runtimeClass.getName} apply(${implicitly[Encoder[I]].clsTag.runtimeClass.getName} _i) {
165         public java.lang.Object apply(java.lang.Object _i) {
166           inRow.update(0, _i);
167           InternalRow enc = (InternalRow) inRow;
168           $encProjections
169           $encUpdates
170 
171           InternalRow i = (InternalRow) encRow;
172           $evalSubexpr
173           $allProjections
174           // copy all the results into MutableRow
175           $allUpdates

177           ${
178             if (expressions.length - offset == 1 && resTypeIsStruct)
179               s"interim = mutableRow.getStruct(${exprFrom.length}, ${resType.length});"
180             else
181               (for(i <- 0 until (expressions.length - offset)) yield
182                // s"interim.update($i, mutableRow.get(${offset + i}, org.apache.spark.sql.types.${exprsToUse(offset + i).dataType.typeName}));"
183                 s"interim.update($i, ${CodeGenerator.getValue("mutableRow", expressions(offset + i).dataType, (offset + i).toString)});"
184                 ).mkString("\n")
185           }
186 
187           InternalRow dec = (InternalRow) interim;
188           $decProjections
189           $decUpdates
190 
191           return (${implicitly[Encoder[O]].clsTag.runtimeClass.getName}) decRow.get(0, new org.apache.spark.sql.types.ObjectType(Object.class));
192         }
193

196       }
197     """
198 
199     val code = CodeFormatter.stripOverlappingComments(
200       new CodeAndComment(codeBody, ctx.getPlaceHolderToComments()))
201     logDebug(s"code for ${expressions.mkString(",")}:\n${CodeFormatter.format(code)}")
202 
203     val (clazz, _) = CodeGenerator.compile(code)
204     val initial = clazz.generate(ctx.references.toArray).asInstanceOf[DecoderOpEncoderProjection[I,O]]
205     if (isCopyNeeded(expressions))
206       new DecoderOpEncoderProjection[I,O] {
207 
208         override def apply(input: I): O = initial(input)
209 
210         // needs a fresh tree copy for each newInstance, so we need to proxy it
211         override def newInstance: DecoderOpEncoderProjection[I, O] =
212           create[I, O] (
213             expressions.map(e => e.transformUp { case t => t.withNewChildren(t.children) }),
214             useSubexprElimination)
215 
216         override def initialize(partitionIndex: Int): Unit = initial.initialize(partitionIndex)
217       }
218     else
219       // with no stateful CodegenFallback's the generated newInstance function is correct
220       initial
221   }
