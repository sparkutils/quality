13 case class HashFunctionProxy(hasher: Hasher) extends Digest {
14 
15   override def hashInt(i: Int): Unit =
16     hasher.putInt(i)
17 
18   override def hashLong(l: Long): Unit =
19     hasher.putLong(l)

21   override def hashBytes(base: Array[Byte], offset: Int, length: Int): Unit =
22     hasher.putBytes(base, offset, length)
23 
24   override def digest: Array[Long] = {
25     val bytes = hasher.hash.asBytes
26     val res = Array.ofDim[Long]((bytes.length + 7) / 8) // next whole long size up
27 
28     // to make sure the array is large enough pad as needed, so a 32bit digest will always take 64
29     val use =
30       if (bytes.length / 8 != res.length) {
31         val nbytes = Array.ofDim[Byte](res.length * 8)
32         Array.copy(bytes, 0, nbytes, 0, bytes.length)
33         nbytes
34       } else
35         bytes
36 
37     for{ i <- 0 until res.length} {
38       res(i) = BytePackingUtils.unencodeLong(i * 8, use)
39     }
40 
41     res
42   }
43 }
