53   /**
54    * Creates a ProcessFactory[I,O], when forceMutable || !compile is true it uses chained MutableProjections,
55    * otherwise, and by default, GenerateDecoderOpEncoderProjection.
56    *
57    * GenerateDecoderOpEncoderProjection is likely far faster on larger rules or higher number of threads for throughput.
58    * Unlike Spark compilation caching, which evaluates codegen for an expression tree every time, the codegen takes place
59    * once, each instance call simply returns a new class instance with fresh state for a thread.
60    *
61    * The chained MutableProjections generate new class code for each instance.  They will also recreate the entire
62    * expression tree if a stateful expression is identified when compile = false
63    *
64    * @param dataFrameFunction
65    * @param compile when false reverts to interpreted mode, when true and forceMutable is true it is recommended to cache the instances
66    * @param forceMutable when true it forces MutableProjection's to be used, compiled or otherwise, the default of false is likely far faster
67    * @param toSize specifies the number of fields required to deserialize and create the O
68    * @tparam I
69    * @tparam O
70    * @return
71    */
72   def processFactory[I: Encoder, O: Encoder](dataFrameFunction: DataFrame => DataFrame, toSize: Int, compile: Boolean = true,
73       forceMutable: Boolean = false, extraProjection: DataFrame => DataFrame = identity): ProcessorFactory[I, O] = {
74     if (forceMutable || !compile)
75       MutableProjectionProcessor.processFactory[I, O](dataFrameFunction, toSize, compile, extraProjection)
76     else {
77       enableOptimizations(Seq(FunNRewrite, ConstantFolding))
78 
79       val iEnc = implicitly[Encoder[I]]
80       val exprs = QualitySparkUtils.resolveExpressions[I](iEnc, df => {
81         dataFrameFunction(extraProjection(df))
82       })
83 
84       val projector = GenerateDecoderOpEncoderProjection.generate[I, O](exprs, useSubexprElimination = true, toSize)
85       new ProcessorFactory[I, O] {
86         override def instance: Processor[I, O] = new Processor[I, O] {
87           private val theInstance = projector.newInstance
88           override def apply(i: I): O = theInstance(i)
89           override def setPartition(partition: Int): Unit = theInstance.initialize(partition)
90           override def close(): Unit = {}
91         }
92       }
93     }
94   }
