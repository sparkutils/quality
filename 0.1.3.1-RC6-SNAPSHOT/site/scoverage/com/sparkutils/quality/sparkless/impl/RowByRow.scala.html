51   /**
52    * Creates a ProcessFactory[I,O], when forceMutable || !compile is true it uses chained MutableProjections,
53    * otherwise, and by default, GenerateDecoderOpEncoderProjection.
54    *
55    * GenerateDecoderOpEncoderProjection is likely far faster on larger rules or higher number of threads for throughput.
56    * Unlike Spark compilation caching, which evaluates codegen for an expression tree every time, the codegen takes place
57    * once, each instance call simply returns a new class instance with fresh state for a thread.
58    *
59    * The chained MutableProjections generate new class code for each instance.  They will also recreate the entire
60    * expression tree if a stateful expression is identified when compile = false
61    *
62    * @param dataFrameFunction
63    * @param compile when false reverts to interpreted mode, when true and forceMutable is true it is recommended to cache the instances
64    * @param forceMutable when true it forces MutableProjection's to be used, compiled or otherwise, the default of false is likely far faster
65    * @tparam I
66    * @tparam O
67    * @return
68    */
69   def processFactory[I: Encoder, O: Encoder](dataFrameFunction: DataFrame => DataFrame, compile: Boolean = true, forceMutable: Boolean = false): ProcessorFactory[I, O] = {
70     if (forceMutable || !compile)
71       MutableProjectionProcessor.processFactory[I, O](dataFrameFunction, compile)
72     else {
73       enableOptimizations(Seq(FunNRewrite, ConstantFolding))
74 
75       val iEnc = implicitly[Encoder[I]]
76       val exprs = QualitySparkUtils.resolveExpressions[I](iEnc, dataFrameFunction)
77 
78       val projector = GenerateDecoderOpEncoderProjection.generate[I, O](exprs, true)
79       new ProcessorFactory[I, O] {
80         override def instance: Processor[I, O] = new Processor[I, O] {
81           private val theInstance = projector.newInstance
82           override def apply(i: I): O = theInstance(i)
83           override def setPartition(partition: Int): Unit = theInstance.initialize(partition)
84           override def close(): Unit = {}
85         }
86       }
87     }
88   }
