70 /**
71  * Functionality related to LambdaCompilation.  Seemingly all HigherOrderFunctions use a lazy val match to extract the
72  * NamedLambdaVariable's from the spark LambdaFunction after bind has been called.
73  * When doGenCode is called eval _could_ have been called and the lazy val evaluated, as such simply rewriting the tree
74  * may not fully work.  Additionally the type for NamedLambdaVariable is bound in the lazy val's which means _ANY_
75  * HigherOrderFunction may not tolerate swapping out NamedLambdaVariables for another NamedExpression.
76  *
77  * To add to the fun OpenSource Spark HoF's all use CodegenFallback, as does NamedLambdaVariable, so it's possible to
78  * swap out some of these implementations if an array_transform is nested in a Fun1 or Fun2.  Similarly Fun1's can call Fun2
79  * so the assumptions are for each Fun1/FunN doCodeGen:
80  *
81  * 1. Use the processLambda function to evaluate the function
82  * 2. compilationHandlers uses the quality.lambdaHandlers environment variable to load a comma separated list of fqn=handler pairs
83  * 3. each fully qualified class name pair (e.g. org.apache.spark.sql.catalyst.expressions.ZipWith=handler.fqn) handler is loaded
84  * 4. processLambda then evaluates the expression tree, for each matching HoF classname it will call the handler
85  * 5. handlers are used to perform the custom doGenCode for that expression rather than the default OSS CodegenFallback
86  * 6. handlers return the ExprCode AND a list of NamedLambdaVariables who must have .value.set called upon them (e.g. we can't optimise them)
87  *
88  * NB The fqn will also be used to check for named' lambdas used through registerLambdaFunctions.
89  *
90  * https://github.com/apache/spark/pull/21954 introduced the lambdavariable with AtomicReference, it's inherent performance hit and,
91  * due to the difficulty of threading the holder through the expression chain did not have a compilation approach.  After
92  * it's threaded and bind has been called the variable id is stable as is the AtomicReference, as such it can be swapped out
93  * for a simple variable in the same object.
94  *
95  * quality.lambdaHandlers will override the default for a given platform on an fqn basis, so you only need to "add" or "replace" the
96  * HoFs that cause issue not the entire list of OSS HigherOrderFunctions for example TransformValues.  Note that some versions
97  * of Databricks provide compilation of their HoF's that may not be compatible in approach.
98  *
99  * Disable this approach by using the quality.lambdaHandlers to disable FunN with the default DoCodegenFallbackHandler:
100  *   quality.lambdaHandlers=org.apache.spark.sql.qualityFunctions.FunN=org.apache.spark.sql.qualityFunctions.DoCodegenFallbackHandler
101  */

184   /**
185    * Replaces all NamedLambdaVariables for expressions which are not in compilationHandlers with a simple offset generator
186    *
187    * If a handler exists (i.e. something should be skipped) it's NamedLambdaVariables are returned as these must be kept.
188    *
189    * @param expr
190    * @param ctx
191    * @param ev
192    * @return
193    */
194   def processLambda(expr: Expression, ctx: CodegenContext, ev: ExprCode): Expression = {
195     // collection is expensive, do both the nlv and exprs together
196     val (exprMaps, allNLVs) =
197       expr.collect {
198         case e: HigherOrderFunction if compilationHandlers.contains(e.getClass.getName) =>
199           Left(e -> compilationHandlers(e.getClass.getName).shouldTransform(e))
200         case e @ FunN(_, _, Some(name), _, _, _) if compilationHandlers.contains(name) =>
201           Left(e -> compilationHandlers(name).shouldTransform(e))
202         case f: FunN =>
203           Left(f -> Seq())
204         case e: HigherOrderFunction =>
205           // default should be to collect as we don't know about it, default catch all for OSS and Databricks, if you have an impl add it
206           Left(e -> defaultGen.shouldTransform(e))
207         case e: NamedLambdaVariable =>
208           Right(e.exprId -> e)
209       }.partition(_.isLeft)
210 
211     val allToBeExcluded = exprMaps.flatMap(p => p.left.get._2.map( n => n.exprId -> n)).toMap
212 
213     val exprIdMaps = allNLVs.map(_.right.get).toMap.filterKeys(k => !allToBeExcluded.contains(k)).map(p => p._2.toCodeGenPair(ctx)).toMap
214 
215     def replaceNLVs(expr: Expression) =
216       expr.transform{
217         case n: NamedLambdaVariable =>
218           exprIdMaps.get(n.exprId).fold[Expression](n)(identity)
219       }
220 
221     def replaceWithHandler(expr: HigherOrderFunction, name: String) =
222       compilationHandlers(name).transform(expr, exprIdMaps)
223 
224     // correctness with map lookup, ref eq should be fine
225     val newTree =
226       expr.transformDown {
227         case e: HigherOrderFunction if compilationHandlers.contains(e.getClass.getName) =>
228           replaceWithHandler(e, e.getClass.getName)
229         case e @ FunN(_, _, Some(name), _, _, _) if compilationHandlers.contains(name) =>
230           replaceWithHandler(e, name)
231         case e: FunN =>
232           replaceNLVs(e).asInstanceOf[FunN].
233             copy(processed = true, attemptCodeGen = true)
234         case e: HigherOrderFunction =>
235           // default is don't transform at all
236           e
237       }
238 
239     newTree
240   }
