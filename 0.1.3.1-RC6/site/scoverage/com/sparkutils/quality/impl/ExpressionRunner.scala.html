21 object ExpressionRunner {
22   /**
23    * Runs the ruleSuite expressions saving results as a tuple of (ruleResult: yaml, resultType: String)
24    * Supplying a ddlType triggers the output type for the expression to be that ddl type, rather than using yaml conversion.
25    * @param renderOptions provides rendering options to the underlying snake yaml implementation
26    * @param ddlType optional DDL string, when present yaml output is disabled and the output expressions must all have the same type
27    * @param name the default column name "expressionResults"
28    */
29   def apply(ruleSuite: RuleSuite, name: String = "expressionResults", renderOptions: Map[String, String] = Map.empty, ddlType: String = "",
30             variablesPerFunc: Int = 40, variableFuncGroup: Int = 20, forceRunnerEval: Boolean = false, compileEvals: Boolean = false): Column = {
31     com.sparkutils.quality.registerLambdaFunctions( ruleSuite.lambdaFunctions )
32     val expressions = flattenExpressions(ruleSuite)
33     val collectExpressions =
34       if (ddlType.isEmpty)
35         expressions.map( i => YamlEncoderExpr(i, renderOptions))
36       else
37         expressions
38     val exprs =
39       // ExpressionProxy and SubExprEvaluationRuntime cannot be used with compileEvals
40       if (compileEvals)
41         PassThroughCompileEvals(collectExpressions)
42       else
43         PassThroughEvalOnly(collectExpressions)
44 
45     val ddl_type =
46       if (ddlType.isEmpty)
47         expressionResultTypeYaml
48       else
49         DataType.fromDDL(ddlType)
50 
51     val cleaned = RuleLogicUtils.cleanExprs(ruleSuite)
52 
53     ShimUtils.column(
54       if (forceRunnerEval)
55         new ExpressionRunnerEval(cleaned, exprs,
56           ddl_type, variablesPerFunc = variablesPerFunc, variableFuncGroup = variableFuncGroup,
57           compileEvals = compileEvals)
58       else
59         new ExpressionRunner(cleaned, exprs,
60           ddl_type, variablesPerFunc = variablesPerFunc, variableFuncGroup = variableFuncGroup,
61           compileEvals = compileEvals)
62     ).as(name)
63   }
64 }
