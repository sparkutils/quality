55   /**
56    * Same as functions)without the wrapping Column
57    * @param rule
58    * @return
59    */
60   def expr(rule: String) = {
61     val parser = SparkSession.getActiveSession.map(_.sessionState.sqlParser).getOrElse {
62       newParser()
63     }
64     /*
65     attempt for a simple expression, then try a plan with exactly one output row.
66     "In" will handle ListQuery, "Exists" similarly exists, everything else will likely fail as a parser error
67      */
68     val rawExpr =
69       try {
70         parser.parseExpression(rule)
71       } catch {
72         case ot: Throwable => // e.g. suitable for output expressions
73           try {
74             ScalarSubquery(parser.parsePlan(rule))
75           } catch {
76             case _: Throwable =>
77               // quite possibly a lambda using a subquery so try and force it via ( sub ) trick,
78               // but allow us to replace later after resolution.  DBR > 12 requires () on simple expressions as well
79               val r = rule.split("->")
80               val wrapped =
81                 if (r.size == 2)
82                   s"${r(0)} -> ( ${r(1)} )"
83                 else
84                   s"( ${r(0)} )"
85 
86               try {
87                 parser.parseExpression(wrapped)
88               } catch {
89                 case _: Throwable => throw ot // if this didn't work return the original error
90               }
91           }
92       }
93 
94     val res =
95       rawExpr match {
96         case l: SparkLambdaFunction if hasSubQuery(l) =>
97           // The lambda's will be parsed as UnresolvedAttributes and not the needed lambdas
98           val names = l.arguments.map(a => a.name -> a).toMap
99           l.transformUp {
100             case s: SubqueryExpression =>
101               s.withNewPlan( s.plan.transform{
102               case snippet =>
103                 snippet.transformAllExpressions {
104                   case a: UnresolvedAttribute =>
105                     names.get(a.name).map(lamVar => lamVar).getOrElse(a)
106                 }
107               }
108             )
109           }
110         case _ => rawExpr
111       }
112 
113     res
114   }
