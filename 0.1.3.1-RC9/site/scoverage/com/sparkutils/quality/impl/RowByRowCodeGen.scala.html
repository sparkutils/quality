14 /**
15  * CODE is based on MutableProjection and generates a transformation between two encoders over a middle operation.
16  * The generated class itself can create new instances directly, unlike Spark projections that need to go through the
17  * codegen source generation cycle, only the compilation is cached.
18  *
19  * If the expression tree contains stateful expressions with codegenfallback the code must be regenerated against a
20  * fresh tree.
21  */

39   def projections(ctx: CodegenContext, expressions: Seq[Expression], mutableRow: String, useSubexprElimination: Boolean = false) = {
40     val validExpr = expressions.zipWithIndex.filter {
41       case (NoOp, _) => false
42       case _ => true
43     }
44     val exprVals = ctx.generateExpressions(validExpr.map(_._1), useSubexprElimination).toIndexedSeq
45 
46     // 4-tuples: (code for projection, isNull variable name, value variable name, column index)
47     val projectionCodes: Seq[(String, String)] = validExpr.zip(exprVals).map {
48       case ((e, i), ev) =>
49         val value = JavaCode.global(
50           ctx.addMutableState(CodeGenerator.javaType(e.dataType), "value"),
51           e.dataType)
52         val (code, isNull) = if (e.nullable) {
53           val isNull = ctx.addMutableState(CodeGenerator.JAVA_BOOLEAN, "isNull")
54           (s"""
55               |${ev.code}
56               |$isNull = ${ev.isNull};
57               |$value = ${ev.value};
58             """.stripMargin, JavaCode.isNullGlobal(isNull))
59         } else {
60           (s"""
61               |${ev.code}
62               |$value = ${ev.value};
63             """.stripMargin, FalseLiteral)
64         }
65         val update = CodeGenerator.updateColumn(
66           mutableRow,
67           e.dataType,
68           i,
69           ExprCode(isNull, value),
70           e.nullable)
71         (code, update)
72     }
73     projectionCodes
74   }
