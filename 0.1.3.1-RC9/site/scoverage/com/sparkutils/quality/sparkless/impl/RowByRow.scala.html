39   /**
40    * Are there any stateful expressions in interpreted mode (or fallback) require fresh copies
41    * @param exprs These expressions must be fully resolved, bound and optimised
42    * @param compile when false any stateful expression returns true, by default compilation for stateful CodegenFallback is true
43    * @return
44    */
45   def isCopyNeeded(exprs: Seq[Expression], compile: Boolean = true): Boolean =
46     exprs.collect {
47       case e: CodegenFallback if ShimUtils.isStateful(e)  => e
48     }.nonEmpty || ( !compile &&
49       exprs.collect {
50         case e: Expression if ShimUtils.isStateful(e)  => e
51       }.nonEmpty
52     )

54   /**
55    * Creates a ProcessFactory[I,O], when forceMutable || !compile is true it uses chained MutableProjections,
56    * otherwise, and by default, GenerateDecoderOpEncoderProjection.
57    *
58    * GenerateDecoderOpEncoderProjection is likely far faster on larger rules or higher number of threads for throughput.
59    * Unlike Spark compilation caching, which evaluates codegen for an expression tree every time, the codegen takes place
60    * once, each instance call simply returns a new class instance with fresh state for a thread.  Once the number of
61    * input fields goes beyond spark.sql.codegen.maxFields (100) the code will fallback to INPUT_ROW based processing.
62    *
63    * The chained MutableProjections generate new class code for each instance.  They will also recreate the entire
64    * expression tree if a stateful expression is identified when compile = false
65    *
66    * @param dataFrameFunction
67    * @param compile when false reverts to interpreted mode, when true and forceMutable is true it is recommended to cache the instances
68    * @param forceMutable when true it forces MutableProjection's to be used, compiled or otherwise, the default of false is likely far faster
69    * @param forceVarCompilation defaulting to false it will, when compile is true and forceMutable is false, use variables rather than INPUT_ROW to generate code.  Using true is (at 0.1.3.1) not advised, there are performance and code size issues
70    * @param toSize specifies the number of fields required to deserialize and create the O
71    * @tparam I
72    * @tparam O
73    * @return
74    */
75   def processFactory[I: Encoder, O: Encoder](dataFrameFunction: DataFrame => DataFrame, toSize: Int, compile: Boolean = true,
76       forceMutable: Boolean = false, forceVarCompilation: Boolean = false, extraProjection: DataFrame => DataFrame = identity, enableQualityOptimisations: Boolean = true): ProcessorFactory[I, O] = {
77     if (forceMutable || !compile)
78       MutableProjectionProcessor.processFactory[I, O](dataFrameFunction, toSize, compile, extraProjection, enableQualityOptimisations = enableQualityOptimisations)
79     else {
80       if (enableQualityOptimisations) {
81         enableOptimizations(Seq(FunNRewrite, ConstantFolding))
82       }
83 
84       val iEnc = implicitly[Encoder[I]]
85       val exprs = QualitySparkUtils.resolveExpressions[I](iEnc, df => {
86         dataFrameFunction(extraProjection(df))
87       })
88 
89       def numOfNestedFields(dataType: DataType): Int = dataType match {
90         case dt: StructType => dt.fields.map(f => numOfNestedFields(f.dataType)).sum
91         case m: MapType => numOfNestedFields(m.keyType) + numOfNestedFields(m.valueType)
92         case a: ArrayType => numOfNestedFields(a.elementType)
93         //case u: UserDefinedType[_] => numOfNestedFields(u.sqlType)
94         case _ => 1
95       }
96 
97       def isTooManyFields(conf: SQLConf, dataType: DataType): Boolean = {
98         numOfNestedFields(dataType) > conf.wholeStageMaxNumFields
99       }
100 
101       val projector =
102         if (forceVarCompilation && !isTooManyFields(SQLConf.get, iEnc.schema))
103           GenerateDecoderOpEncoderVarProjection.create[I, O](exprs, toSize)
104         else
105           GenerateDecoderOpEncoderProjection.generate[I, O](exprs, useSubexprElimination = true, toSize)
106       new ProcessorFactory[I, O] {
107         override def instance: Processor[I, O] = new Processor[I, O] {
108           private val theInstance = projector.newInstance
109           override def apply(i: I): O = theInstance(i)
110           override def setPartition(partition: Int): Unit = theInstance.initialize(partition)
111           override def close(): Unit = {}
112         }
113       }
114     }
115   }
