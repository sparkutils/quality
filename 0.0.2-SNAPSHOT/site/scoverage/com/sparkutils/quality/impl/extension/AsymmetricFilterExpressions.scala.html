13 /**
14  * Replaces any attributes aliased by an expression which is then used in select filters / constraints wth
15  * constituent predicates that can be pushed down.  A simple example is:
16  * ```
17  *   select as_uuid(lower, higher) as context
18  *   where context = 'string'
19  * ```
20  * this would be replaced with:
21  * ```
22  *   select as_uuid(lower, higher) as context
23  *   where (lower = longPairFromUUID('string').lower and higher = longPairFromUUID('string').higher)
24  * ```
25  * in the case of the string being a constant, constant folding should replace both of these with simple lookups that are easy to push down.
26  *
27  * Only EqualTo is supported currently and expressions not used in select (i.e. as_uuid(lower ,higher) = 'uuid-string'
28  */
29 abstract class AsymmetricFilterExpressions extends Rule[LogicalPlan]
30   with PredicateHelperPlus with ConstraintHelper {
31 
32   /**
33    * Should the matchOnExpression be defined in a filter provide the rewrite.  It's possible for joins that both comparedTo AND generating are the
34    * same expression type, this should be explicitly checked for.
35    *
36    * @param comparedTo can be used transparently, irrespective of usage (NamedExpression, Expression etc.)
37    * @param generating the underlying expression which is generating either an Alias or the direct expression itself
38    * @return a rewritten expression tree, logically the opposite of the generating expression or None where generating does not match
39    */
40   def reWriteExpression(generating: Expression, comparedTo: Expression): Option[Expression]
41 
42   // attempt to find / replace any part of the filter tree
43   def transform(expr: Expression, topPlan: LogicalPlan): (Expression, Boolean) = {
44     var found = false
45     (expr.transform {
46       case e@EqualTo(elhs, erhs) =>
47         findRootExpression(elhs, topPlan).flatMap{ a =>
48           val rhs = findRootExpression(erhs, topPlan)
49           rhs.map( b => (a, b) )
50         }.map {
51           case (lhs, rhs) =>
52             // flip the sides as needed
53             reWriteExpression(lhs, rhs).orElse(
54               reWriteExpression(rhs, lhs)
55             ).map { e => found = true; e }.getOrElse(e)
56         }.getOrElse(e)
57     }, found)
58   }
59 
60   class FilterTransformer(topPlan: LogicalPlan) {
61     def unapply(plan: LogicalPlan): Option[(Expression, LogicalPlan)] =
62       plan match {
63         case Filter(expr, child) =>
64           val (r, found) = transform(expr, topPlan)
65           if (found)
66             Some((r, child))
67           else
68             None
69         case _ => None
70       }
71   }
72 
73   class JoinTransformer(topPlan: LogicalPlan) {
74     def unapply(plan: LogicalPlan): Option[Join] =
75       plan match {
76         case j: Join =>
77           // attempt re-write for join
78           j.condition.flatMap { condition =>
79 
80             val (r, found) = transform(condition, topPlan)
81 
82             if (found)
83               Some(j.copy(condition = Some(r)))
84             else
85               None
86           }
87         case _ => None
88       }
89   }
90 
91   def apply(plan: LogicalPlan): LogicalPlan = {
92     val FilterT = new FilterTransformer(plan)
93     val JoinT = new JoinTransformer(plan)
94     plan transform {
95       case FilterT(resulting, child) =>
96         Filter(resulting, child)
97       case JoinT(join) =>
98         join
99     }
100   }
