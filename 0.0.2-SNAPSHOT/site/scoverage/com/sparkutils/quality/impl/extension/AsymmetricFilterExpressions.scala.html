11 /**
12  * Replaces any attributes aliased by an expression which is then used in select filters / constraints wth
13  * constituent predicates that can be pushed down.  A simple example is:
14  * {{{
15  *     select as_uuid(lower, higher) as context
16  *     where context = 'string'
17  * }}}
18  * this would be replaced with:
19  * {{{
20  *     select as_uuid(lower, higher) as context
21  *     where (lower = longPairFromUUID('string').lower and higher = longPairFromUUID('string').higher)
22  * }}}
23  * in the case of the string being a constant, constant folding should replace both of these with simple lookups that are easy to push down.
24  *
25  * *NB.* Pushdown predicates based on constant folding only work via the SparkExtension approach, the use of experimental.extraOptimizations does provide this as it happens too late in the plans
26  *
27  * Only In and BinaryComparison's are supported currently
28  */
29 abstract class AsymmetricFilterExpressions extends Rule[LogicalPlan]
30   with PredicateHelperPlus with ConstraintHelper {
31 
32   /**
33    * Should the matchOnExpression be defined in a filter provide the rewrite.  It's possible for joins that both comparedTo AND generating are the
34    * same expression type, this should be explicitly checked for.
35    *
36    * @param generating the underlying expression which is generating either an Alias or the direct expression itself
37    * @param comparedTo either an Expression for BinaryComparison where it can be used transparently, irrespective of usage (NamedExpression, Expression etc.), or Seq[Expression] for In's.
38    *                   Where In's are used and the re-write involves structures you should aim to repeat the constituent parts as separate In's in order to get predicate push down
39    * @param filter the Expression used in the filter e.g. EqualTo, In, GreaterThan etc.
40    * @return a rewritten expression tree, logically the opposite of the generating expression or None where generating does not match
41    */
42   def reWriteExpression(generating: Expression, comparedTo: Object, filter: Expression): Option[Expression]
43 
44   // attempt to find / replace any part of the filter tree
45   def transform(expr: Expression, topPlan: LogicalPlan): (Expression, Boolean) = {
46     var found = false
47     (expr.transform {
48       case i@In(expr, list) =>
49         findRootExpression(expr, topPlan).map { lhs =>
50           reWriteExpression(lhs, list, i).map { e => found = true; e }.getOrElse(i)
51         }.getOrElse(i)
52       case e@BinaryComparison(elhs, erhs) =>
53         findRootExpression(elhs, topPlan).flatMap{ a =>
54           val rhs = findRootExpression(erhs, topPlan)
55           rhs.map( b => (a, b) )
56         }.map {
57           case (lhs, rhs) =>
58             // flip the sides as needed
59             reWriteExpression(lhs, rhs, e).orElse(
60               reWriteExpression(rhs, lhs, e)
61             ).map { e => found = true; e }.getOrElse(e)
62         }.getOrElse(e)
63     }, found)
64   }

66   class FilterTransformer(topPlan: LogicalPlan) {
67     def unapply(plan: LogicalPlan): Option[(Expression, LogicalPlan)] =
68       plan match {
69         case Filter(expr, child) =>
70           val (r, found) = transform(expr, topPlan)
71           if (found)
72             Some((r, child))
73           else
74             None
75         case _ => None
76       }
77   }
78 
79   class JoinTransformer(topPlan: LogicalPlan) {
80     def unapply(plan: LogicalPlan): Option[Join] =
81       plan match {
82         case j: Join =>
83           // attempt re-write for join
84           j.condition.flatMap { condition =>
85 
86             val (r, found) = transform(condition, topPlan)
87 
88             if (found)
89               Some(j.copy(condition = Some(r)))
90             else
91               None
92           }
93         case _ => None
94       }
95   }
96 
97   def apply(plan: LogicalPlan): LogicalPlan = {
98     val FilterT = new FilterTransformer(plan)
99     val JoinT = new JoinTransformer(plan)
100     plan transform {
101       case FilterT(resulting, child) =>
102         Filter(resulting, child)
103       case JoinT(join) =>
104         join
105     }
106   }

129   override def reWriteExpression(generating: Expression, comparedTo: Object,
130                                  filter: Expression): Option[Expression] = {
131     val create = AsymmetricFilterExpressions.create(filter,_: Expression,_: Expression)
132     (generating, comparedTo, filter) match {
133       // join case (possibly a filter)
134       case (AsUUID(alower, ahigher), AsUUID(blower, bhigher), Equality(_,_) ) =>
135         Some(
136           And( create(alower, blower), create(ahigher, bhigher))
137         )
138       // filter case's
139       case (AsUUID(lower, higher), c: Expression, Equality(_,_)) if c.dataType == StringType =>
140         Some(
141           And( create( GetStructField( UUIDToLongsExpression(c), 0) , lower), create( GetStructField( UUIDToLongsExpression(c), 1 ), higher) )
142         )
143       // In verifies the rest of the seq are the same type
144       case (a@AsUUID(lower, higher), l: Seq[Expression], i: In) if l.headOption.exists(_.dataType == StringType) =>
145         def struct(lower: Expression, higher: Expression): Expression =
146           CreateNamedStruct(Seq(Literal("lower"), lower,
147             Literal("higher"), higher))
148 
149         def structg(expr: Expression): Expression =
150           struct( GetStructField(UUIDToLongsExpression(expr), 0), GetStructField(UUIDToLongsExpression(expr), 1))
151 
152         Some(
153           And( In(struct(lower, higher), l.map(structg _)),
154             And( In(lower, l.map(t => GetStructField(UUIDToLongsExpression(t), 0))),
155               In(higher, l.map(t => GetStructField(UUIDToLongsExpression(t), 1)))
156             ))
157         )
158       // join + filter
159       case (AsUUID(alower, ahigher), AsUUID(blower, bhigher), _: LessThan | _: LessThanOrEqual | _: GreaterThan | _: GreaterThanOrEqual) =>
160         Some(
161           Or(And(EqualTo(ahigher, bhigher), create(alower, blower)), create(ahigher, bhigher))
162         )
163       // filter case's
164       case (AsUUID(lower, higher), c: Expression, _: LessThan | _: LessThanOrEqual | _: GreaterThan | _: GreaterThanOrEqual) if c.dataType == StringType =>
165         val blower = GetStructField(UUIDToLongsExpression(c), 0)
166         val bhigher = GetStructField(UUIDToLongsExpression(c), 1)
167 
168         Some(
169           Or(And(EqualTo(higher, bhigher), create(lower, blower)), create(higher, bhigher))
170         )
171       case _ => None
172     }
173   }
