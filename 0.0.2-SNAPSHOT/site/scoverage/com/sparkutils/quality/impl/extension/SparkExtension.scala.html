17 /**
18  * Registers Quality sql functions using the defaults for registerQualityFunctions, these can be overridden without having to subclass DriverPlugin.
19  *
20  * It also registers plan optimiser rule's such as AsUUIDFilter, which rewrites filters with variables backed by as_uuid.
21  *
22  * Optimiser rules can be disabled via the quality.disable.optimiser.rules system environment variable. "*" disables all rules, otherwise a comma separated list of fqn class names may be used.
23  */
24 class QualitySparkExtension extends ((SparkSessionExtensions) => Unit) with Logging {
25 
26   def parseTypes: String => Option[DataType] = com.sparkutils.quality.defaultParseTypes _
27   def zero: DataType => Option[Any] = com.sparkutils.quality.defaultZero _
28   def add: DataType => Option[(Expression, Expression) => Expression] = (dataType: DataType) => com.sparkutils.quality.defaultAdd(dataType)
29   def mapCompare: DataType => Option[(Any, Any) => Int] = (dataType: DataType) => utils.defaultMapCompare(dataType)
30   def writer: String => Unit = println(_)
31 
32   /**
33    * Adds AsymmetricFilterExpressions for AsUUID
34    * Derived implementations should also call super.
35    * These are registered after resolution is done.
36    * The first String should be the fqn classname / rule name for the rule
37    * @param sparkSession
38    * @return
39    */
40   def optimiserRules: Seq[(String, SparkSession => Rule[LogicalPlan])] = Seq((AsUUIDFilter.getClass.getName, _ => AsUUIDFilter))
41 
42   override def apply(extensions: SparkSessionExtensions): Unit = {
43     com.sparkutils.quality.registerQualityFunctions(parseTypes, zero, add, mapCompare, writer,
44       register = QualitySparkUtils.registerFunctionViaExtension(extensions) _
45     )
46 
47     if (Testing.testing) {
48       ExtensionTesting.disableRuleResult = ""
49     }
50     val disableConf = com.sparkutils.quality.getConfig(disableRulesConf)
51     if (disableConf != "*") {
52       val disabledRules = disableConf.split(",").map(_.trim).toSet
53       val filteredRules = optimiserRules.filterNot(p => disabledRules.contains(p._1.trim))
54       val str = s"$disableRulesConf = $disabledRules leaving ${filteredRules.map(_._1)} remaining"
55       logInfo(str)
56       println(s"Quality SparkExtensions: $str")
57       if (Testing.testing) {
58         ExtensionTesting.disableRuleResult = str
59       }
60       filteredRules.map(_._2).foreach(extensions.injectOptimizerRule _)
61     } else {
62       val str = s"All optimiser rules are disabled via $disableRulesConf"
63       logInfo(str)
64       println(s"Quality SparkExtensions: $str")
65     }
66   }
67 
68 }
