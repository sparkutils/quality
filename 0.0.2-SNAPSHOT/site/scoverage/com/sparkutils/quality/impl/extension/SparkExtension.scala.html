20 /**
21  * Registers Quality sql functions using the defaults for registerQualityFunctions, these can be overridden without having to subclass DriverPlugin.
22  *
23  * It also registers plan optimiser rule's such as AsUUIDFilter, which rewrites filters with variables backed by as_uuid.
24  *
25  * Optimiser rules can be disabled via the quality_disable_optimiser_rules system environment variable. "*" disables all rules, otherwise a comma separated list of fqn class names may be used.
26  */
27 class QualitySparkExtension extends ((SparkSessionExtensions) => Unit) with Logging {
28 
29   def parseTypes: String => Option[DataType] = com.sparkutils.quality.defaultParseTypes _
30   def zero: DataType => Option[Any] = com.sparkutils.quality.defaultZero _
31   def add: DataType => Option[(Expression, Expression) => Expression] = (dataType: DataType) => com.sparkutils.quality.defaultAdd(dataType)
32   def mapCompare: DataType => Option[(Any, Any) => Int] = (dataType: DataType) => utils.defaultMapCompare(dataType)
33   def writer: String => Unit = println(_)
34 
35   /**
36    * Adds AsymmetricFilterExpressions for AsUUID
37    * Derived implementations should also call super.
38    * These are registered after resolution is done.
39    * The first String should be the fqn classname / rule name for the rule
40    * @param sparkSession
41    * @return
42    */
43   def optimiserRules: Seq[(String, SparkSession => Rule[LogicalPlan])] = Seq((AsUUIDFilter.getClass.getName, _ => AsUUIDFilter))
44 
45   override def apply(extensions: SparkSessionExtensions): Unit = {
46     com.sparkutils.quality.registerQualityFunctions(parseTypes, zero, add, mapCompare, writer,
47       register = QualitySparkUtils.registerFunctionViaExtension(extensions) _
48     )
49 
50     if (Testing.testing) {
51       ExtensionTesting.disableRuleResult = ""
52     }
53     val disableConf = com.sparkutils.quality.getConfig(disableRulesConf)
54     if (disableConf != "*") {
55       val disabledRules = disableConf.split(",").map(_.trim).toSet
56       val filteredRules = optimiserRules.filterNot(p => disabledRules.contains(p._1.trim))
57       val str = s"$disableRulesConf = $disabledRules leaving ${filteredRules.map(_._1)} remaining"
58       logInfo(str)
59       println(s"Quality SparkExtensions: $str")
60       if (Testing.testing) {
61         ExtensionTesting.disableRuleResult = str
62       }
63       filteredRules.map(_._2).foreach(extensions.injectOptimizerRule _)
64     } else {
65       val str = s"All optimiser rules are disabled via $disableRulesConf"
66       logInfo(str)
67       println(s"Quality SparkExtensions: $str")
68     }
69   }
70 
71 }
