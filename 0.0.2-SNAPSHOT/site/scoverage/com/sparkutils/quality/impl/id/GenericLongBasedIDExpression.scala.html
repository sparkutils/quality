66 /**
67  * Delegates ID creation to some other expression which must provide an array of longs result.
68  *
69  * @param id type of the GenericLong compatible ID to be generated
70  * @param child an expression generating an Array Of Longs
71  * @param prefix how to name the field
72  */
73 case class GenericLongBasedIDExpression(id: IDType, child: Expression, prefix: String) extends UnaryExpression with CodegenFallback {
74   private lazy val (converter, length) = GenericLongBasedIDExpression.longArrayConverter(child)
75   private lazy val headerHolder = GenericLongBasedID(id, Array.ofDim[Long](length))
76   private lazy val base = headerHolder.base
77 
78   override protected def nullSafeEval(input: Any): Any = {
79     // an array of longs
80     val ar = converter(input)
81     // fill the data back out
82     InternalRow.fromSeq(base +: ar)
83   }
84 
85 //  override protected def doGenCode(ctx: CodegenContext, ev: ExprCode): ExprCode = ???
86 
87   override def dataType: DataType = headerHolder.dataType(prefix)
88 
89   protected def withNewChildInternal(newChild: Expression): Expression = copy(child = newChild)
90 }

92 object GenericLongBasedIDExpression {
93 
94   def longArrayConverter( expr: Expression): (Any => Array[Long], Int) = {
95     val struct =
96       expr.dataType match{
97         case e: StructType => e
98         case _ => null
99       }
100 
101     if (expr.dataType == RandomLongs.structType || (
102       (struct ne null) &&
103       struct.fields.forall(_.dataType == LongType)
104     )) {
105       val count = struct.fields.length
106       ((input: Any) => {
107         val row = input.asInstanceOf[InternalRow]
108         (0 until count).map(row.getLong).toArray[Long]
109       }, count)
110     } else
111       qualityException("Only structures with arrays of longs are supported")
112   }
