172 /**
173  * Buckets smaller optimised blooms by a mod on bucket size with array backing by default
174  *
175  * @param numBytes
176  * @param iMinimumBytes
177  * @param iMaximumBytes
178  * @param hashImpl
179  * @param numBuckets max is 15 algo wise but 8 breaks defaults for  spark.driver.maxResultSize on local 1gb it's 4g on db
180  */
181 case class BucketedCreator[H, SerializedType](numBytes: Int, iMinimumBytes: Int, iMaximumBytes: Int, hashImpl: BloomHash, fpp: Double, numBuckets: Int,
182                                            toType: ToSerializedType[SerializedType, H], hint: H)(
183   val filters: Seq[BlockSplitBloomFilterImpl] = (1 to numBuckets).map(_ => BlockSplitBloomFilterImpl(numBytes, iMinimumBytes, iMaximumBytes, hashImpl)),
184   val bucketedFiles: Option[BucketedFiles] = None
185 ) extends Bloom[SerializedType] {
186   type BloomType = BucketedCreator[H, SerializedType]
187 
188   /** CTw added the or'ing */
189   def |= (that: BucketedCreator[H, SerializedType]): BucketedCreator[H, SerializedType] = {
190     checkCompatibility(that)
191     // Breeze copies but we'll do in place
192     for(i <- 0 until numBuckets) {
193       filters(i) |= that.filters(i)
194     }
195     this
196   }
197 
198   private def checkCompatibility(that: BucketedCreator[H, SerializedType]): Unit =
199     require(iMaximumBytes == that.iMaximumBytes
200       && iMinimumBytes == that.iMinimumBytes &&
201       hashImpl.hashStrategy == that.hashImpl.hashStrategy
202       && filters.zip( that.filters ).forall( p => p._1.intBuffer.limit() == p._2.intBuffer.limit() )
203       , "Must have the hash and bitset length to intersect")
204 
205   def += (value: Any) = {
206     val h = hashImpl.hash(value)
207     filters(Bucketed.whichBloom(h, numBuckets)).insertHash(h)
208     this.asInstanceOf[BloomType]
209   }
210 
211   /**
212    * Combines all the buckets together via the serialize type.  If this is already linked to memory mapped files
213    * do not re-write, just serialize the bucketedFiles reference
214    * @return combined serialized blooms
215    */
216   override def serialized: Array[Byte] = bucketedFiles.fold(toType.serializeBuckets(filters, fpp, numBuckets, hint))(_.serialize)
