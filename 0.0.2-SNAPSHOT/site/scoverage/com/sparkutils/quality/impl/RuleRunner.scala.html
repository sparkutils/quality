51   def ruleRunner(ruleSuite: RuleSuite, compileEvals: Boolean = true, resolveWith: Option[DataFrame] = None, variablesPerFunc: Int = 40, variableFuncGroup: Int = 20, forceRunnerEval: Boolean = false): Column = {
52     com.sparkutils.quality.registerLambdaFunctions( ruleSuite.lambdaFunctions )
53     val flattened = flattenExpressions(ruleSuite)
54     val runner = new RuleRunner(RuleLogicUtils.cleanExprs(ruleSuite), PassThrough(flattened), compileEvals, variablesPerFunc, variableFuncGroup, forceRunnerEval)
55     new Column(
56       QualitySparkUtils.resolveWithOverride(resolveWith).map { df =>
57         val resolved = QualitySparkUtils.resolveExpression(df, runner)
58 
59         resolved.asInstanceOf[RuleRunner].copy(child = resolved.children(0) match {
60           // replace the expr
61           case PassThrough(children) => NonPassThrough(children)
62         })
63       } getOrElse runner
64     )
65   }
