51   def ruleRunner(ruleSuite: RuleSuite, compileEvals: Boolean = true, resolveWith: Option[DataFrame] = None, variablesPerFunc: Int = 40, variableFuncGroup: Int = 20, forceRunnerEval: Boolean = false): Column = {
52     com.sparkutils.quality.registerLambdaFunctions( ruleSuite.lambdaFunctions )
53     val runner = new RuleRunner(ruleSuite, PassThrough(flattenExpressions(ruleSuite)), compileEvals, variablesPerFunc, variableFuncGroup, forceRunnerEval)
54     new Column(
55       QualitySparkUtils.resolveWithOverride(resolveWith).map { df =>
56         val resolved = QualitySparkUtils.resolveExpression(df, runner)
57 
58         resolved.asInstanceOf[RuleRunner].copy(child = resolved.children(0) match {
59           // replace the expr
60           case PassThrough(children) => NonPassThrough(children)
61         })
62       } getOrElse runner
63     )
64   }
