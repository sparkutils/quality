10 trait ProcessDisableIfMissingImports extends RuleRunnerImports {
11   /**
12    * Processes a given RuleSuite to replace any coalesceIfMissingAttributes.  This may be called before validate / docs but
13    * *must* be called *before* adding the expression to a dataframe.
14    *
15    * @param ruleSuite
16    * @param schema The names to validate against, if empty no attempt to process coalesceIfAttributeMissing will be made
17    * @return
18    */
19   def processIfAttributeMissing(ruleSuite: RuleSuite, schema: StructType = StructType(Seq())) = {
20     val names = namesFromSchema(schema)
21 
22     val lambdas = ruleSuite.lambdaFunctions.map(lf => processIfMissingLambdaCoalesce(lf.parsed, names))
23 
24     val wrs = ruleSuite.copy(lambdaFunctions = lambdas)
25 
26     mapRules( wrs ) {
27       processCoalesceIfAttributeMissing(_, names)
28     }
29   }
30 
31   /**
32    * For a given schema evaluates all calls to coalesce in a lambda to swap out missing attributes
33    *
34    * Must be called before any planning - otherwise Spark will already have rejected it.
35    *
36    * @param rule
37    * @return
38    */
39   protected def processIfMissingLambdaCoalesce(rule: LambdaFunctionParsed, names: Set[String]): LambdaFunction =
40     rule.expr match {
41       case lambdaFunc: SparkLambdaFunction =>
42         lambdaFunc.function match {
43           case func: UnresolvedFunction =>
44             val rexp = processCoalesceIfAttributeMissing(func, names)
45             rule.copy(expr = lambdaFunc.copy(function = rexp))
46 
47           case _ => rule
48         }
49     }
50 
51   protected def isReplaceCoalesceName(unresolvedFunction: UnresolvedFunction) = {
52     val name = toName(unresolvedFunction)
53     name == "coalesceIfAttributesMissing" || name == "coalesceIfAttributesMissingDisable"
54   }
55 
56   protected def isCoalesceDisabled(unresolvedFunction: UnresolvedFunction) = {
57     val name = toName(unresolvedFunction)
58     name == "coalesceIfAttributesMissingDisable"
59   }
60 
61   /**
62    * Converts expression removing any coalesceIfAttributeMissing placeholding expressions using names to find each
63    * first expression with no missing names (or replacing with null should all of them be).
64    * @param expression
65    * @param names
66    * @return
67    */
68   def processCoalesceIfAttributeMissing(expression: Expression, names: Set[String]): Expression =
69     expression match {
70       case funcExpr: UnresolvedFunction if isReplaceCoalesceName(funcExpr) =>
71         processCoalesceIfAttributeMissing(expression.children, names, isCoalesceDisabled(funcExpr))
72       case e: Expression =>
73         e.withNewChildren(e.children.map(processCoalesceIfAttributeMissing(_, names)))
74     }
75 
76   /**
77    * Returns the first expression that has no missing names or null
78    * @param expressions
79    * @param names
80    * @return
81    */
82   protected def processCoalesceIfAttributeMissing(expressions: Seq[Expression], names: Set[String], isDisabled: Boolean): Expression = {
83     val res =
84       expressions.flatMap{ e =>
85         // make sure no nested children are coalesce's
86         val r = processCoalesceIfAttributeMissing(e, names)
87         // get the names from this expression
88         val explookup = fieldsFromExpression(r)
89         // lookup each of them in the names and if any are not true we can't use this expression
90         if (!explookup.attributesUsed.map(names).forall(identity))
91           None
92         else
93           Some(r)
94       }
95 
96     if (res.isEmpty) {
97       if (isDisabled)
98         DisabledRuleExpr // special case as this is probably before the function registry gets called
99       else
100         Literal(null, NullType)
101     } else
102       res.head
103   }
104 
105   /**
106    * Converts both rule expressions and any output expressions removing any coalesceIfAttributeMissing placeholding expressions using names to find each
107    * first expression with no missing names (or replacing with null should all of them be).
108    * @param expression
109    * @param names
110    * @return
111    */
112   protected def processCoalesceIfAttributeMissing(rule: Rule, names: Set[String]): Rule =
113     rule match {
114       // rule and output
115       case Rule(id, ExpressionRule(rule: String),
116         iorule @ RunOnPassProcessorImpl(_, _, _, OutputExpression(oruleExpr: String))) if oruleExpr.nonEmpty =>
117         Rule(id, ExpressionRuleExpr(rule, processCoalesceIfAttributeMissing(RuleLogicUtils.expr(rule), names)),
118           iorule.copy(returnIfPassed = OutputExpressionExpr(oruleExpr,
119             processCoalesceIfAttributeMissing(RuleLogicUtils.expr(oruleExpr), names))))
120 
121       // just a rule
122       case orule @ Rule(_, ExpressionRule(rule: String), _) =>
123         orule.copy( expression = ExpressionRuleExpr(rule, processCoalesceIfAttributeMissing(RuleLogicUtils.expr(rule), names)))
124 
125       case _ => rule
126     }
127 }
