63 
64   def namesFromSchema(schema: StructType): Set[String] = {
65 
66     def withParent(name: String, parent: String) =
67       if (parent.isEmpty)
68         name
69       else
70         parent + "." + name
71 
72     def accumulate(set: Set[String], schema: StructType, parent: String): Set[String] =
73       schema.foldLeft(set) {
74         (s, field) =>
75           val name = withParent(field.name, parent)
76           field.dataType match {
77             case struct: StructType =>
78               accumulate(s + name, struct, name)
79             case _ => s + name
80           }
81       }
82 
83     accumulate(Set.empty, schema, "")
84   }
85 
86   /**
87    * Use this function to identify which maps / blooms etc. are used by a given rulesuite
88    * collects all rules that are using lookup functions but without constant expressions and the list of lookups that are constants.
89    *
90    */
91   def identifyLookups(ruleSuite: RuleSuite): LookupResults = {
92     val olambdaResults =
93       ruleSuite.lambdaFunctions.flatMap{r =>
94         val exp = RuleLogicUtils.expr(r.rule)
95         LookupIdFunctionImpl.identifyLookups(exp).map((_,r))
96       }.foldLeft(ExpressionLookupResults[Id](Map.empty, Set.empty)) {
97         (acc, res) =>
98           val r = acc.copy( lookupConstants = acc.lookupConstants + (res._2.id -> res._1.constants))
99           if (res._1.hasExpressionLookups)
100             r.copy(lookupExpressions = r.lookupExpressions + res._2.id)
101           else
102             r
103       }
104     val lambdaResults = olambdaResults.copy(lookupConstants = olambdaResults.lookupConstants.filter(_._2.nonEmpty))
105 
106     LookupResults(ruleSuite, ExpressionLookupResults(Map.empty, Set.empty), lambdaResults)
107   }
