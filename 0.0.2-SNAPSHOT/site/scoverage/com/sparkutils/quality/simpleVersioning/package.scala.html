58     val versionedRules = df.sparkSession.sql(
59       """
60        select ruleExpr, ruleId, ruleVersion, ruleEngineSalience, ruleEngineId, ruleEngineVersion,
61        ruleSetId, ruleSetVersion, -- don't need to bump versions as they can coexist
62         suiteversions.ruleSuiteId, suiteversions.ruleSuiteVersion -- force the versions to be bumped to latest ruleSuite Versions
63         from
64          (select distinct ruleSuiteId, ruleSuiteVersion from rules) suiteversions join
65          rules l0 on l0.ruleSuiteId = suiteversions.ruleSuiteId and l0.ruleSuiteVersion <= suiteversions.ruleSuiteVersion
66          where
67           not exists (
68             select 0 from rules l1
69             where l1.ruleSuiteId = l0.ruleSuiteId
70              and l1.ruleId = l0.ruleId
71              and l1.ruleSetId = l0.ruleSetId
72              and l1.ruleSuiteVersion <= suiteversions.ruleSuiteVersion
73              and l1.ruleVersion > l0.ruleVersion
74           )
75           and l0.ruleExpr != "DELETED"
76        """)

133   protected[quality] def lambdaOutputSQL(tableName: String, extra: String ="") =
134       s"""
135        select $extra ruleExpr, functionId, functionVersion,
136         versions.ruleSuiteId, versions.ruleSuiteVersion -- force the versions to be bumped to latest ruleSuiteVersions
137         from
138          (select distinct ruleSuiteId, ruleSuiteVersion from $tableName) versions join
139          $tableName l0 on l0.ruleSuiteId = versions.ruleSuiteId and l0.ruleSuiteVersion <= versions.ruleSuiteVersion
140          where
141           not exists (
142             select 0 from $tableName l1
143             where l1.ruleSuiteId = l0.ruleSuiteId and
144              l1.functionId = l0.functionId and
145              l1.ruleSuiteVersion <= versions.ruleSuiteVersion
146              and l1.functionVersion > l0.functionVersion
147           )
148           and l0.ruleExpr != "DELETED"

187   protected[quality] case class SameOrNextVersionLower[T](map: Map[Id, Seq[T]]) extends Function1[Id, Option[Seq[T]]] {
188     implicit val ordering = new Ordering[Id] {
189       def compare(x: Id, y: Id): Int = {
190         if (x == y)
191           0
192         else
193           if (x.id != y.id)
194             x.id - y.id
195           else
196             x.version - y.version
197       }
198     }
199 
200     val sorted = TreeMap[Id, Seq[T]]() ++ map
201 
202     override def apply(id: Id): Option[Seq[T]] =
203       sorted.to(id).lastOption.map(_._2)
204   }
