59    * Same as functions.expr without the wrapping Column
60    * @param rule
61    * @return

63   def expr(rule: String) = {
64     val parser = SparkSession.getActiveSession.map(_.sessionState.sqlParser).getOrElse {
65       QualitySparkUtils.newParser()
66     }
67     /*
68     attempt for a simple expression, then try a plan with exactly one output row.
69     "In" will handle ListQuery, "Exists" similarly exists, everything else will likely fail as a parser error
70      */
71     val rawExpr =
72       try {
73         parser.parseExpression(rule)
74       } catch {
75         case ot: Throwable => // e.g. suitable for output expressions
76           try {
77             ScalarSubquery(parser.parsePlan(rule))
78           } catch {
79             case _: Throwable =>
80               // quite possibly a lambda using a subquery so try and force it via ( sub ) trick,
81               // but allow us to replace later after resolution.  DBR > 12 requires () on simple expressions as well
82               val r = rule.split("->")
83               val wrapped =
84                 if (r.size == 2)
85                   s"${r(0)} -> ( ${r(1)} )"
86                 else
87                   s"( ${r(0)} )"
88 
89               try {
90                 parser.parseExpression(wrapped)
91               } catch {
92                 case _: Throwable => throw ot // if this didn't work return the original error
93               }
94           }
95       }
96 
97     val res =
98       rawExpr match {
99         case l: SparkLambdaFunction if hasSubQuery(l) =>
100           // The lambda's will be parsed as UnresolvedAttributes and not the needed lambdas
101           val names = l.arguments.map(a => a.name -> a).toMap
102           l.transform {
103             case s: SubqueryExpression => s.withNewPlan( s.plan.transform{
104               case snippet =>
105                 snippet.transformAllExpressions {
106                   case a: UnresolvedAttribute =>
107                     names.get(a.name).map(lamVar => lamVar).getOrElse(a)
108                 }
109             })
110           }
111         case _ => rawExpr
112       }
113 
114     res
115   }

347   private[quality] override def expression() = {
348     val parsed = RuleLogicUtils.expr(rule)
349     // output expressions can be:
350     // 1. simple expressions for ruleEngine
351     // 2. single argument lambda's returning the same type as the arg for folder
352     // 3. as of 0.0.2 #8 set( attribute = valueExpression, attribute = valueExpression) converted to the form of 2 with an updateField call
353     parsed match {
354       case uf: UnresolvedFunction if VariablesLookup.toName(uf) == "set" =>
355         // case 3
356         val args = QualitySparkUtils.arguments(uf)
357         val paired =
358           args.flatMap {
359             case EqualTo(name: UnresolvedAttribute, right) =>
360               // updateField takes paired args of field names to expression
361               Some(Seq(Literal(name.name), right))
362             case a =>
363               logInfo(s"Attempt to convert set OutputExpression argument $a failed as types do not match expected EqualTo(attribute, expression), will default to full expression")
364               None
365           }
366 
367         if (paired.size != args.size)
368           // one of the args didn't match type
369           parsed
370         else
371           // need to keep first arg
372           UpdateFolderExpression.withArgsAndSubstitutedLambdaVariable(paired.flatten)
373       case _ =>
374         // for everything else (1+2) it's already good enough
375         parsed
376     }
