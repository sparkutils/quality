63   def expr(rule: String) = {
64     val parser = SparkSession.getActiveSession.map(_.sessionState.sqlParser).getOrElse {
65       QualitySparkUtils.newParser()
66     }
67     /*
68     attempt for a simple expression, then try a plan with exactly one output row.
69     "In" will handle ListQuery, "Exists" similarly exists, everything else will likely fail as a parser error
70      */
71     val rawExpr =
72       try {
73         parser.parseExpression(rule)
74       } catch {
75         case ot: Throwable => // e.g. suitable for output expressions
76           try {
77             ScalarSubquery(parser.parsePlan(rule))
78           } catch {
79             case t: Throwable => // quite possibly a lambda using a subquery so try and force it via the struct(( sub )).col1 trick
80               val r = rule.split("->")
81               if (r.size == 2) {
82                 val wrapped = s"${r(0)} -> struct(( ${r(1)} )).col1"
83                 try {
84                   parser.parseExpression(wrapped)
85                 } catch {
86                   case _: Throwable => throw ot // if this didn't work return the original error
87                 }
88               } else
89                 throw ot // if this didn't work return the original error
90           }
91       }
92 
93     val res =
94     rawExpr match {
95       case l: SparkLambdaFunction if hasSubQuery(l) =>
96         // The lambda's will be parsed as UnresolvedAttributes and not the needed lambdas
97         val names = l.arguments.map(a => a.name -> a).toMap
98         l.transform {
99           case s: SubqueryExpression => s.withNewPlan( s.plan.transform{
100             case snippet =>
101               snippet.transformAllExpressions {
102                 case a: UnresolvedAttribute =>
103                   names.get(a.name).map(lamVar => lamVar).getOrElse(a)
104               }
105           })
106         }
107       case _ => rawExpr
108     }
109 
110     res

343   private[quality] override def expression() = {
344     val parsed = RuleLogicUtils.expr(rule)
345     // output expressions can be:
346     // 1. simple expressions for ruleEngine
347     // 2. single argument lambda's returning the same type as the arg for folder
348     // 3. as of 0.0.2 #8 set( attribute = valueExpression, attribute = valueExpression) converted to the form of 2 with an updateField call
349     parsed match {
350       case uf: UnresolvedFunction if VariablesLookup.toName(uf) == "set" =>
351         // case 3
352         val args = QualitySparkUtils.arguments(uf)
353         val paired =
354           args.flatMap {
355             case EqualTo(name: UnresolvedAttribute, right) =>
356               // updateField takes paired args of field names to expression
357               Some(Seq(Literal(name.name), right))
358             case a =>

360               None
361           }
362 
363         if (paired.size != args.size)
364           // one of the args didn't match type
365           parsed
366         else
367           // need to keep first arg
368           UpdateFolderExpression.withArgsAndSubstitutedLambdaVariable(paired.flatten)
369       case _ =>
370         // for everything else (1+2) it's already good enough
371         parsed
372     }
