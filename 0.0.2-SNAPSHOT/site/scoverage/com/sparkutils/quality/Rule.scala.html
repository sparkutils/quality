59    * Same as functions.expr without the wrapping Column

63   def expr(rule: String) = {
64     val parser = SparkSession.getActiveSession.map(_.sessionState.sqlParser).getOrElse {
65       QualitySparkUtils.newParser()
66     }
67     /*
68     attempt for a simple expression, then try a plan with exactly one output row.
69     "In" will handle ListQuery, "Exists" similarly exists, everything else will likely fail as a parser error
70      */
71     val rawExpr =
72       try {
73         parser.parseExpression(rule)
74       } catch {
75         case ot: Throwable => // e.g. suitable for output expressions
76           try {
77             ScalarSubquery(parser.parsePlan(rule))
78           } catch {
79             case _: Throwable =>
80               // quite possibly a lambda using a subquery so try and force it via ( sub ) trick,
81               // but allow us to replace later after resolution
82               val r = rule.split("->")
83               if (r.size == 2) {
84                 val wrapped = s"${r(0)} -> ( ${r(1)} )"
85                 try {
86                   parser.parseExpression(wrapped)
87                 } catch {
88                   case _: Throwable => throw ot // if this didn't work return the original error
89                 }
90               } else
91                 throw ot // if this didn't work return the original error
92           }
93       }
94 
95     val res =
96       rawExpr match {
97         case l: SparkLambdaFunction if hasSubQuery(l) =>
98           // The lambda's will be parsed as UnresolvedAttributes and not the needed lambdas
99           val names = l.arguments.map(a => a.name -> a).toMap
100           l.transform {
101             case s: SubqueryExpression => s.withNewPlan( s.plan.transform{
102               case snippet =>
103                 snippet.transformAllExpressions {
104                   case a: UnresolvedAttribute =>
105                     names.get(a.name).map(lamVar => lamVar).getOrElse(a)
106                 }
107             })
108           }
109         case _ => rawExpr
110       }
111 
112     res
113   }

345   private[quality] override def expression() = {
346     val parsed = RuleLogicUtils.expr(rule)
347     // output expressions can be:
348     // 1. simple expressions for ruleEngine
349     // 2. single argument lambda's returning the same type as the arg for folder
350     // 3. as of 0.0.2 #8 set( attribute = valueExpression, attribute = valueExpression) converted to the form of 2 with an updateField call
351     parsed match {
352       case uf: UnresolvedFunction if VariablesLookup.toName(uf) == "set" =>
353         // case 3
354         val args = QualitySparkUtils.arguments(uf)
355         val paired =
356           args.flatMap {
357             case EqualTo(name: UnresolvedAttribute, right) =>
358               // updateField takes paired args of field names to expression
359               Some(Seq(Literal(name.name), right))
360             case a =>
361               logInfo(s"Attempt to convert set OutputExpression argument $a failed as types do not match expected EqualTo(attribute, expression), will default to full expression")
362               None
363           }
364 
365         if (paired.size != args.size)
366           // one of the args didn't match type
367           parsed
368         else
369           // need to keep first arg
370           UpdateFolderExpression.withArgsAndSubstitutedLambdaVariable(paired.flatten)
371       case _ =>
372         // for everything else (1+2) it's already good enough
373         parsed
374     }
