98   def execute(logicalPlan: LogicalPlan, batch: Batch) = {
99     var iteration = 1
100     var curPlan = logicalPlan
101     var lastPlan = logicalPlan
102 
103     var start = System.currentTimeMillis
104 
105 
106     var continue = true
107     val analyzer = SparkSession.getActiveSession.get.sessionState.analyzer
108 
109     // Run until fix point (or the max number of iterations as specified in the strategy.
110     while (continue) {
111       curPlan = batch.rules.foldLeft(curPlan) {
112         case (plan, rule) =>
113           val startTime = System.nanoTime()
114           val result = rule(plan)
115 
116           result
117       }
118       iteration += 1
119       if (iteration > batch.strategy.maxIterations) {
120         // Only log if this is a rule that is supposed to run more than once.
121         if (iteration != 2) {
122           val endingMsg = if (batch.strategy.maxIterationsSetting == null) {
123             "."
124           } else {
125             s", please set '${batch.strategy.maxIterationsSetting}' to a larger value."
126           }
127           val message = s"Max iterations (${iteration - 1}) reached for batch ${batch.name}" +
128             s"$endingMsg"
129           if (Utils.isTesting || batch.strategy.errorOnExceed) {
130             throw new Exception(message)
131           } else {
132           }
133         }
134         continue = false
135       }
136 
137       if (curPlan.fastEquals(lastPlan)) {
138         continue = false
139       }
140       lastPlan = curPlan
141     }
142     var stop = System.currentTimeMillis
143     //println(s"did $iteration iterations to execute the plan in ${stop-start}ms")
144     curPlan

211   protected def tryResolveReferences(
212                                       sparkSession: SparkSession)(
213                                       expr: Expression,
214                                       child: LogicalPlan): Expression = {
215     val analyzer = sparkSession.sessionState.analyzer
216 
217     def forExpr(expr: Expression) = {
218       val newPlan = FakePlan(expr, child)
219       //analyzer.execute(newPlan)
220       execute(newPlan, resolution(analyzer, sparkSession, newPlan))
221       match {
222         case FakePlan(resolvedExpr, _) =>
223           // Return even if it did not successfully resolve
224           resolvedExpr
225         case _ =>
226           // This is unexpected
227           throw new Exception(
228             s"Could not resolve expression $expr with child $child}")
229       }
230     }
231     // special case as it's faster to do individual items it seems, 36816ms vs 48974ms
232     expr match {
233       case r @ RuleEngineRunner(ruleSuite, PassThrough( expressions ), realType, compileEvals, debugMode, func, group, forceRunnerEval, expressionOffsets, forceTriggerEval) =>
234         val nexprs = expressions.map(forExpr)
235         RuleEngineRunner(ruleSuite, PassThrough( nexprs ), realType, compileEvals, debugMode, func, group, forceRunnerEval, expressionOffsets, forceTriggerEval)
236       case r @ RuleFolderRunner(ruleSuite, left, PassThrough( expressions ), resultDataType, compileEvals, debugMode, variablesPerFunc,
237         variableFuncGroup, forceRunnerEval, expressionOffsets, dataRef, forceTriggerEval) =>
238         val nexprs = expressions.map(forExpr)
239         RuleFolderRunner(ruleSuite, left, PassThrough( nexprs ), resultDataType, compileEvals, debugMode, variablesPerFunc,
240           variableFuncGroup, forceRunnerEval, expressionOffsets, dataRef, forceTriggerEval)
241       case r @ RuleRunner(ruleSuite, PassThrough( expressions ), compileEvals, func, group, forceRunnerEval) =>
242         val nexprs = expressions.map(forExpr)
243         RuleRunner(ruleSuite, PassThrough( nexprs ), compileEvals, func, group, forceRunnerEval)
244       case _ => forExpr(expr)
245     }

248   case class FakePlan(expr: Expression, child: LogicalPlan)
249     extends UnaryNode {

255     protected def mygetAllValidConstraints(projectList: Seq[Expression]): Set[Expression] = {
256       var allConstraints = Set.empty[Expression]
257       projectList.foreach {
258         case a@Alias(l: Literal, _) =>
259           allConstraints += EqualNullSafe(a.toAttribute, l)
260         case a@Alias(e, _) =>
261           // For every alias in `projectList`, replace the reference in constraints by its attribute.
262           allConstraints ++= allConstraints.map(_ transform {
263             case expr: Expression if expr.semanticEquals(e) =>
264               a.toAttribute
265           })
266           allConstraints += EqualNullSafe(e, a.toAttribute)
267         case _ => // Don't change.
268       }
269 
270       allConstraints
271     }
272 
273     override lazy val validConstraints: ExpressionSet = ExpressionSet(mygetAllValidConstraints(Seq(expr)))
