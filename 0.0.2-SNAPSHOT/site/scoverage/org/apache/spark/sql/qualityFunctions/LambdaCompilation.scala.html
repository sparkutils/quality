73 /**
74  * Functionality related to LambdaCompilation.  Seemingly all HigherOrderFunctions use a lazy val match to extract the
75  * NamedLambdaVariable's from the spark LambdaFunction after bind has been called.
76  * When doGenCode is called eval _could_ have been called and the lazy val evaluated, as such simply rewriting the tree
77  * may not fully work.  Additionally the type for NamedLambdaVariable is bound in the lazy val's which means _ANY_
78  * HigherOrderFunction may not tolerate swapping out NamedLambdaVariables for another NamedExpression.
79  *
80  * To add to the fun OpenSource Spark HoF's all use CodegenFallback, as does NamedLambdaVariable, so it's possible to
81  * swap out some of these implementations if an array_transform is nested in a Fun1 or Fun2.  Similarly Fun1's can call Fun2
82  * so the assumptions are for each Fun1/FunN doCodeGen:
83  *
84  * 1. Use the processLambda function to evaluate the function
85  * 2. compilationHandlers uses the quality.lambdaHandlers environment variable to load a comma separated list of fqn=handler pairs
86  * 3. each fully qualified class name pair (e.g. org.apache.spark.sql.catalyst.expressions.ZipWith=handler.fqn) handler is loaded
87  * 4. processLambda then evaluates the expression tree, for each matching HoF classname it will call the handler
88  * 5. handlers are used to perform the custom doGenCode for that expression rather than the default OSS CodegenFallback
89  * 6. handlers return the ExprCode AND a list of NamedLambdaVariables who must have .value.set called upon them (e.g. we can't optimise them)
90  *
91  * NB The fqn will also be used to check for named' lambdas used through registerLambdaFunctions.
92  *
93  * https://github.com/apache/spark/pull/21954 introduced the lambdavariable with AtomicReference, it's inherent performance hit and,
94  * due to the difficulty of threading the holder through the expression chain did not have a compilation approach.  After
95  * it's threaded and bind has been called the variable id is stable as is the AtomicReference, as such it can be swapped out
96  * for a simple variable in the same object.
97  *
98  * quality.lambdaHandlers will override the default for a given platform on an fqn basis, so you only need to "add" or "replace" the
99  * HoFs that cause issue not the entire list of OSS HigherOrderFunctions for example TransformValues.  Note that some versions
100  * of Databricks provide compilation of their HoF's that may not be compatible in approach.
101  *
102  * Disable this approach by using the quality.lambdaHandlers to disable FunN with the default DoCodegenFallbackHandler:
103  *   quality.lambdaHandlers=org.apache.spark.sql.qualityFunctions.FunN=org.apache.spark.sql.qualityFunctions.DoCodegenFallbackHandler
104  */

198   /**
199    * Replaces all NamedLambdaVariables for expressions which are not in compilationHandlers with a simple offset generator
200    *
201    * If a handler exists (i.e. something should be skipped) it's NamedLambdaVariables are returned as these must be kept.
202    *
203    * @param expr
204    * @param ctx
205    * @param ev
206    * @return

208   def processLambda(expr: Expression, ctx: CodegenContext, ev: ExprCode): Expression = {
209     // collection is expensive, do both the nlv and exprs together
210     val (exprMaps, allNLVs) =
211       expr.collect {
212         case e: HigherOrderFunction if compilationHandlers.contains(e.getClass.getName) =>
213           Left(e -> compilationHandlers(e.getClass.getName).shouldTransform(e))
214         case e @ FunN(_, _, Some(name), _, _) if compilationHandlers.contains(name) =>
215           Left(e -> compilationHandlers(name).shouldTransform(e))
216         case f: FunN =>
217           Left(f -> Seq())
218         case e: HigherOrderFunction =>
219           // default should be to collect as we don't know about it, default catch all for OSS and Databricks, if you have an impl add it
220           Left(e -> defaultGen.shouldTransform(e))
221         case e: NamedLambdaVariable =>
222           Right(e.exprId -> e)
223       }.partition(_.isLeft)
224 
225     val allToBeExcluded = exprMaps.flatMap(p => p.left.get._2.map( n => n.exprId -> n)).toMap
226 
227     val exprIdMaps = allNLVs.map(_.right.get).toMap.filterKeys(k => !allToBeExcluded.contains(k)).map(p => p._2.toCodeGenPair(ctx))
228 
229     def replaceNLVs(expr: Expression) =
230       expr.transform{
231         case n: NamedLambdaVariable =>
232           exprIdMaps.get(n.exprId).fold[Expression](n)(identity)
233       }
234 
235     def replaceWithHandler(expr: HigherOrderFunction, name: String) =
236       compilationHandlers(name).transform(expr, exprIdMaps)
237 
238     // correctness with map lookup, ref eq should be fine
239     val newTree =
240       expr.transformDown {
241         case e: HigherOrderFunction if compilationHandlers.contains(e.getClass.getName) =>
242           replaceWithHandler(e, e.getClass.getName)
243         case e @ FunN(_, _, Some(name), _, _) if compilationHandlers.contains(name) =>
244           replaceWithHandler(e, name)
245         case e: FunN =>
246           replaceNLVs(e).asInstanceOf[FunN].
247             copy(processed = true, attemptCodeGen = true)
248         case e: HigherOrderFunction =>
249           // default is don't transform at all
250           e
251       }
252 
253     newTree
254   }
