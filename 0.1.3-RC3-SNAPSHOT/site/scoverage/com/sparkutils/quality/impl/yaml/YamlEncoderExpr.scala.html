142   private def createMapNode(mt: MapType, map: MapData, keyType: Any => Node, valueType: Any => Node)(implicit renderOptions: Map[String, String]) =
143     if (map == null)

157   private def createStructNode(st: StructType)(implicit renderOptions: Map[String, String]) = {
158     val converters = st.fields.map(f =>

232   override def eval(inputRow: InternalRow): Any = {
233     val input = child.eval(inputRow)
234 
235     val writer = new StringWriter()
236     val options = new DumperOptions()
237     import org.yaml.snakeyaml.DumperOptions
238     options.setDefaultFlowStyle(DumperOptions.FlowStyle.FLOW)
239 
240     val resolver = new Resolver
241     val node = valueConverter(input)
242 
243     // 1.24 and 1.33 aren't compatible interface for serialize so Yaml.serialize(node, output) isn't an option
244     val serializer = new Serializer(new Emitter(writer, options), resolver, options, null)
245     try {
246       serializer.open()
247       serializer.serialize(node)
248       serializer.close()
249     } catch {
250       case e: IOException =>
251         throw new YAMLException(e)
252     }
253 
254     val str = writer.getBuffer.toString
255     writer.close()
256     UTF8String.fromString(str)
257   }
