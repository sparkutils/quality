17   /**
18    * Adds fields, in order, for each field path it's paired transformation is applied to the update column
19    *
20    * @param update
21    * @param transformations
22    * @return a new copy of update with the changes applied
23    */
24   def update_field(update: Column, transformations: (String, Column)*): Column =
25     new Column(
26       transformFields{

28           case ((path, col), origin) =>

81  * Add or replace a field by name.
82  *
83  * We extend [[Unevaluable]] here to ensure that [[UpdateFields]] can include it as part of its
84  * children, and thereby enable the analyzer to resolve and transform valExpr as necessary.
85  */
86 case class WithField(name: String, valExpr: Expression)
87   extends UnaryExpression with StructFieldsOperation {
88 
89   override def apply(values: Seq[(StructField, Expression)]): Seq[(StructField, Expression)] = {
90     val newFieldExpr = (StructField(name, valExpr.dataType, valExpr.nullable), valExpr)
91     val result = ArrayBuffer.empty[(StructField, Expression)]
92     var hasMatch = false
93     for (existingFieldExpr @ (existingField, _) <- values) {
94       if (resolver(existingField.name, name)) {
95         hasMatch = true
96         result += newFieldExpr
97       } else {
98         result += existingFieldExpr
99       }
100     }
101     if (!hasMatch) result += newFieldExpr
102     result.toSeq
103   }
104 
105   override def child: Expression = valExpr
106 
107   override def prettyName: String = "WithField"
108 
109   protected def withNewChildInternal(newChild: Expression): WithField =
110     copy(valExpr = newChild)
111 }

154   private lazy val newFieldExprs: Seq[(StructField, Expression)] = {
155     def getFieldExpr(i: Int): Expression = structExpr match {
156       case c: CreateNamedStruct => c.valExprs(i)
157       case _ => GetStructField(structExpr, i)
158     }
159     val fieldsWithIndex = structExpr.dataType.asInstanceOf[StructType].fields.zipWithIndex
160     val existingFieldExprs: Seq[(StructField, Expression)] =
161       fieldsWithIndex.map { case (field, i) => (field, getFieldExpr(i)) }
162     fieldOps.foldLeft(existingFieldExprs)((exprs, op) => op(exprs))
163   }

193 object UpdateFields {
194   private def nameParts(fieldName: String): Seq[String] = {
195     require(fieldName != null, "fieldName cannot be null")
196 
197     if (fieldName.isEmpty) {
198       fieldName :: Nil
199     } else {
200       UnresolvedAttribute.parseAttributeName(fieldName)
201       //CatalystSqlParser.parseMultipartIdentifier(fieldName)
202     }
203   }
204 
205   /**
206    * Adds/replaces field of `StructType` into `col` expression by name.
207    */
208   def apply(col: Expression, fieldName: String, expr: Expression): UpdateFields =
209     updateFieldsHelper(col, nameParts(fieldName), name => WithField(name, expr))
210 
211   /**
212    * Drops fields of `StructType` in `col` expression by name.
213    */
214   def apply(col: Expression, fieldName: String): UpdateFields =
215     updateFieldsHelper(col, nameParts(fieldName), name => DropField(name))
216 
217   private def updateFieldsHelper(
218                                   structExpr: Expression,
219                                   namePartsRemaining: Seq[String],
220                                   valueFunc: String => StructFieldsOperation) : UpdateFields = {
221     val fieldName = namePartsRemaining.head
222     if (namePartsRemaining.length == 1) {
223       UpdateFields(structExpr, Seq(valueFunc(fieldName)))
224     } else {
225       val newStruct = if (structExpr.resolved) {
226         val resolver = SQLConf.get.resolver
227         ExtractValue(structExpr, Literal(fieldName), resolver)
228       } else {
229         UnresolvedExtractValue(structExpr, Literal(fieldName))
230       }
231 
232       val newValue = updateFieldsHelper(
233         structExpr = newStruct,
234         namePartsRemaining = namePartsRemaining.tail,
235         valueFunc = valueFunc)
236       UpdateFields(structExpr, Seq(WithField(fieldName, newValue) ))
237     }
238   }
