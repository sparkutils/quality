16   /**
17    * Creates a column that runs the folding RuleSuite.  This also forces registering the lambda functions used by that RuleSuite.
18    *
19    * FolderRunner runs all output expressions for matching rules in order of salience, the startingStruct is passed ot the first
20    * matching, the result passed to the second etc.  In contrast to ruleEngineRunner OutputExpressions should be lambdas with one parameter, that of the structure
21    *
22    * @param ruleSuite The ruleSuite with runOnPassProcessors
23    * @param startingStruct This struct is passed to the first matching rule, ideally you would use the spark dsl struct function to refer to existing columns
24    * @param compileEvals Should the rules be compiled out to interim objects - by default true
25    * @param debugMode When debugMode is enabled the resultDataType is wrapped in Array of (salience, result) pairs to ease debugging
26    * @param resolveWith This experimental parameter can take the DataFrame these rules will be added to and pre-resolve and optimise the sql expressions, see the documentation for details on when to and not to use this.
27    * @param variablesPerFunc Defaulting to 40 allows, in combination with variableFuncGroup allows customisation of handling the 64k jvm method size limitation when performing WholeStageCodeGen
28    * @param variableFuncGroup Defaulting to 20
29    * @param forceRunnerEval Defaulting to false, passing true forces a simplified partially interpreted evaluation (compileEvals must be false to get fully interpreted)
30    * @param forceTriggerEval Defaulting to true, passing true forces each trigger expression to be compiled (compileEvals) and used in place, false instead expands the trigger in-line giving possible performance boosts based on JIT.  Most testing has however shown this not to be the case hence the default, ymmv.
31    * @param useType In the case you must use select and can't use withColumn you may provide a type directly to stop the NPE
32    * @return A Column representing the QualityRules expression built from this ruleSuite
33    */
34   def ruleFolderRunner(ruleSuite: RuleSuite, startingStruct: Column, compileEvals: Boolean = true,
35                        debugMode: Boolean = false, resolveWith: Option[DataFrame] = None, variablesPerFunc: Int = 40,
36                        variableFuncGroup: Int = 20, forceRunnerEval: Boolean = false, useType: Option[StructType] = None,
37                        forceTriggerEval: Boolean = true): Column = {
38     com.sparkutils.quality.registerLambdaFunctions( ruleSuite.lambdaFunctions )
39 
40     // needed to resolve variables
41     val dataRef = new AtomicReference[DataType]()
42 
43     val realType = () => {
44       val starter =
45         useType.getOrElse(dataRef.get())
46       if (debugMode)
47       // wrap it in an array with the priority result
48         ArrayType(StructType(Seq(StructField("salience", IntegerType), StructField("result", starter))))
49       else
50         starter
51     }
52 
53     val liftLambda = (e: Expression) => FunN(Seq(RefExpressionLazyType(() => dataRef.get(), true)), e)
54 
55     val (expressions, indexes) = flattenExpressions(ruleSuite, liftLambda)
56 
57     val runner = new RuleFolderRunner(RuleLogicUtils.cleanExprs(ruleSuite), startingStruct.expr, PassThrough( expressions ), realType, compileEvals = compileEvals,
58       debugMode = debugMode, variablesPerFunc, variableFuncGroup, forceRunnerEval = forceRunnerEval, expressionOffsets = indexes, dataRef, forceTriggerEval)
59 
60     new Column(
61       QualitySparkUtils.resolveWithOverride(resolveWith).map { df =>
62         val resolved = QualitySparkUtils.resolveExpression(df, runner)
63 
64         resolved.asInstanceOf[RuleFolderRunner].copy(right = resolved.children(1) match {
65           // replace the expr
66           case PassThrough(children) => NonPassThrough(children)
67         })
68       } getOrElse runner
69     )
70   }
