52   private def mapFromDF(id: String, df: DataFrame, key: Column, value: Column) = {
53     val translated = df.select(key.as("key"), value.as("value"))
54     val map = translated.toLocalIterator().asScala.map {
55       mapPair =>
56         toCatalyst(mapPair.get(0)) ->
57           toCatalyst(mapPair.get(1))
58     }.toMap
59 
60     val mapData: MapData = ArrayBasedMapData(map)
61     id -> (df.sparkSession.sparkContext.broadcast(mapData), translated.schema.last.dataType)
62   }
