106   protected def ifoldAndReplaceFields(rules: RuleSuite, fields: Seq[String], foldFieldName: String = "foldedFields", debugMode: Boolean = false,

108                            maintainOrder: Boolean = true, useType: Option[StructType] = None): Dataset[Row] => Dataset[Row] = df => {
109     import org.apache.spark.sql.functions._
110 
111     val theStruct = struct(fields.head, fields.tail :_*)
112     val withFolder = {
113       // select NPEs needs projection to work
114       //df.select(expr("*"), ruleFolderRunner(rules, theStruct).as(foldFieldName))
115       df.withColumn(foldFieldName, ruleFolderRunner(rules, theStruct, debugMode = debugMode, useType = useType) )
116     }
117 
118     // create now as the schema will have the folder, which we may want to keep
119     val namesInOrder =
120       if (maintainOrder)
121         withFolder.schema.map(_.name)
122       else
123         Seq()
124     // lift the results
125     val result =
126       if (debugMode)
127         withFolder.drop(fields : _*).selectExpr("*",
128           s"element_at($foldFieldName.result, -1).result as $tempFoldDebugName"
129         ).selectExpr("*", s"$tempFoldDebugName.*").drop(tempFoldDebugName)
130       else
131         withFolder.drop(fields : _*).selectExpr("*", s"$foldFieldName.result.*" )
132 
133     // bring back to top level in the correct order
134     if (maintainOrder)
135       result.select(namesInOrder.head, namesInOrder.tail :_*)
136     else
137       result
