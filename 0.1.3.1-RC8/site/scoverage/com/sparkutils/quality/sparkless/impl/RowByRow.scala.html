53   /**
54    * Creates a ProcessFactory[I,O], when forceMutable || !compile is true it uses chained MutableProjections,
55    * otherwise, and by default, GenerateDecoderOpEncoderProjection.
56    *
57    * GenerateDecoderOpEncoderProjection is likely far faster on larger rules or higher number of threads for throughput.
58    * Unlike Spark compilation caching, which evaluates codegen for an expression tree every time, the codegen takes place
59    * once, each instance call simply returns a new class instance with fresh state for a thread.
60    *
61    * The chained MutableProjections generate new class code for each instance.  They will also recreate the entire
62    * expression tree if a stateful expression is identified when compile = false
63    *
64    * @param dataFrameFunction
65    * @param compile when false reverts to interpreted mode, when true and forceMutable is true it is recommended to cache the instances
66    * @param forceMutable when true it forces MutableProjection's to be used, compiled or otherwise, the default of false is likely far faster
67    * @param forceVarCompilation defaulting to true it will, when compile is true and forceMutable is false, use variables rather than INPUT_ROW to generate code
68    * @param toSize specifies the number of fields required to deserialize and create the O
69    * @tparam I
70    * @tparam O
71    * @return
72    */
73   def processFactory[I: Encoder, O: Encoder](dataFrameFunction: DataFrame => DataFrame, toSize: Int, compile: Boolean = true,
74       forceMutable: Boolean = false, forceVarCompilation: Boolean = true, extraProjection: DataFrame => DataFrame = identity, enableQualityOptimisations: Boolean = true): ProcessorFactory[I, O] = {
75     if (forceMutable || !compile)
76       MutableProjectionProcessor.processFactory[I, O](dataFrameFunction, toSize, compile, extraProjection, enableQualityOptimisations = enableQualityOptimisations)
77     else {
78       if (enableQualityOptimisations) {
79         enableOptimizations(Seq(FunNRewrite, ConstantFolding))
80       }
81 
82       val iEnc = implicitly[Encoder[I]]
83       val exprs = QualitySparkUtils.resolveExpressions[I](iEnc, df => {
84         dataFrameFunction(extraProjection(df))
85       })
86 
87       val projector =
88         if (forceVarCompilation)
89           GenerateDecoderOpEncoderVarProjection.create[I, O](exprs, toSize)
90         else
91           GenerateDecoderOpEncoderProjection.generate[I, O](exprs, useSubexprElimination = true, toSize)
92       new ProcessorFactory[I, O] {
93         override def instance: Processor[I, O] = new Processor[I, O] {
94           private val theInstance = projector.newInstance
95           override def apply(i: I): O = theInstance(i)
96           override def setPartition(partition: Int): Unit = theInstance.initialize(partition)
97           override def close(): Unit = {}
98         }
99       }
100     }
101   }
