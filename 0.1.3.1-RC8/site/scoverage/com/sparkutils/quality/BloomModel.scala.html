28 /**
29  * Represents the shared file location of a bucked bloom filter.  There should be files with names 0..numBuckets containing
30  * the same number of bytes representing each bucket.
31  *
32  * @param rootDir  The directory which contains each bucket
33  * @param fpp  The fpp for this bloom - note it is informational only and will not be used in further processing
34  * @param numBuckets The number of buckets within this bloom
35  */
36 case class BloomModel(rootDir: String, fpp: Double, numBuckets: Int) extends Serializable {
37   /**
38    * Provides memory mapped buffers from the underlying files
39    * @return
40    */
41   def maps: Seq[IntBuffer] =
42     (0 until numBuckets).map { i =>
43       val file = new File(rootDir, i.toString)
44       val mapped = new FileInputStream(file).getChannel.map(FileChannel.MapMode.READ_ONLY, 0, file.length())
45       mapped.order(ByteOrder.LITTLE_ENDIAN).asIntBuffer()
46     }
47 
48   /**
49    * Serializes the definition of this bucketfiles, not the underlying bytes of the bloom
50    * @return
51    */
52   def serialize: Array[Byte] = {
53     val bos = new ByteArrayOutputStream()
54     val oos = new ObjectOutputStream(bos)
55     oos.writeObject(this)
56     oos.flush()
57     val bytes = bos.toByteArray
58     bos.close()
59     bytes
60   }
61 
62   /**
63    * Only performed lazily
64    * @return
65    */
66   def read: Array[Array[Byte]] = {
67     val size = new File(rootDir, "0").length().toInt // safe as it's never more than an array
68     val ar = Array.ofDim[Array[Byte]](numBuckets)
69     for { i <- 0 until numBuckets} {
70       val f = new File(rootDir, i.toString)
71       val fis = new FileInputStream(f)
72       ar(i) = Array.ofDim[Byte](size)
73       fis.read(ar(i))
74       fis.close()
75     }
76     ar
77   }

79   /**
80    * Removes other directories from the parent root id
81    */
82   def cleanupOthers(): Unit ={

88   }

93   def removeThisBloom(): Unit = {

97     siblings.foreach(_.delete())
98   }
