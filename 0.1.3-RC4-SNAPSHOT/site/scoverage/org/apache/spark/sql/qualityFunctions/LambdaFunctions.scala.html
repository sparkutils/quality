28   def applyFunN(ff: FunForward, transformFunction: (LambdaFunction, NamedExpression, Expression) => Expression =
29     (fun, nvl, exp) =>
30       fun.function.transform {
31         case unvl: NamedLambdaVariable if
32           nvl.exprId == unvl.exprId =>
33           exp
34       } ): FunN = {
35       val placeHolders = ff.params.collect{ case r: RefExpression => r.index }.toSet
36       val nonPlaceHolders = ff.function.arguments.zipWithIndex.foldLeft(ff.function.function.asInstanceOf[LambdaFunction]){ // always a LambdaFunction given you can't partially apply a 0 arg expression
37         case (fun, (exp, index) ) =>
38           // if it's an nvl leave alone
39           if (placeHolders(index))
40             fun
41           else {
42             val nvl = fun.arguments(index)
43             fun.copy(function = transformFunction(fun, nvl, exp))
44           }
45       }
46       ff.function.copy(function = nonPlaceHolders.copy( arguments =
47         nonPlaceHolders.arguments.zipWithIndex.filter{case (_, index) => placeHolders(index)}.map(_._1) ) )
48     }
