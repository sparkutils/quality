82   def namesFromSchema(schema: StructType): Set[String] = {
83 
84     def withParent(name: String, parent: String) =
85       if (parent.isEmpty)
86         name
87       else
88         parent + "." + name
89 
90     def accumulate(set: Set[String], schema: StructType, parent: String): Set[String] =
91       schema.foldLeft(set) {
92         (s, field) =>
93           val name = withParent(field.name, parent)
94           field.dataType match {
95             case struct: StructType =>
96               accumulate(s + name, struct, name)
97             case _ => s + name
98           }
99       }
100 
101     accumulate(Set.empty, schema, "")
102   }
103 
104   /**
105    * Use this function to identify which maps / blooms etc. are used by a given rulesuite

107    *
108    */
109   def identifyLookups(ruleSuite: RuleSuite): LookupResults = {
110     val olambdaResults =
111       ruleSuite.lambdaFunctions.flatMap{r =>
112         val exp = RuleLogicUtils.expr(r.rule)
113         LookupIdFunctionImpl.identifyLookups(exp).map((_,r))
114       }.foldLeft(ExpressionLookupResults[Id](Map.empty, Set.empty)) {
115         (acc, res) =>
116           val r = acc.copy( lookupConstants = acc.lookupConstants + (res._2.id -> res._1.constants))
117           if (res._1.hasExpressionLookups)
118             r.copy(lookupExpressions = r.lookupExpressions + res._2.id)
119           else
120             r
121       }
122     val lambdaResults = olambdaResults.copy(lookupConstants = olambdaResults.lookupConstants.filter(_._2.nonEmpty))
123 
124     LookupResults(ruleSuite, ExpressionLookupResults(Map.empty, Set.empty), lambdaResults)

229   /**
230    * UnsafeArrayData doesn't allow calling .array, foreach when needed and for others use array
231    *
232    * @param array
233    * @param dataType
234    * @param f
235    * @return
236    */
237   def mapArray[T: ClassTag](array: ArrayData, dataType: DataType, f: Any => T): Array[T] =
238     array match {
239       case _: UnsafeArrayData =>
240         val res = Array.ofDim[T](array.numElements())
241         array.foreach(dataType, (i, v) => res.update(i, f(v)))
242         res
243       case _ => array.array.map(f)
244     }
245

248    * @param array
249    * @param dataType
250    * @return
251    */
252   def toArray(array: ArrayData, dataType: DataType): Array[Any] =
253     array match {
254       case _: UnsafeArrayData =>
