32   def genArrayCompare(elementType: DataType, elComp: (Any, Any) => Int): (Any, Any) => Int = (oleft, oright) => {
33     val left = oleft.asInstanceOf[ArrayData]
34     val right = oright.asInstanceOf[ArrayData]
35     if (left.numElements() > right.numElements())
36       1
37     else if (left.numElements() < right.numElements())
38       -1
39     else {
40       // same
41       var i = 0
42       var res = 0
43       val al = (idx: Int) => left.get(idx, elementType)// Arrays.toArray(left, elementType)
44       val ar = (idx: Int) => right.get(idx, elementType)// Arrays.toArray(right, elementType)
45       while (i < left.numElements() && res == 0) {
46         res = elComp(al(i), ar(i))
47         i += 1
48       }
49       res
