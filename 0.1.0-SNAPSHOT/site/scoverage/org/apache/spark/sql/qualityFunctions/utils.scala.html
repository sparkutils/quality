37     val left = oleft.asInstanceOf[ArrayData]
38     val right = oright.asInstanceOf[ArrayData]
39     if (left.numElements() > right.numElements())
40       1
41     else if (left.numElements() < right.numElements())
42       -1
43     else {

49       while (i < left.numElements() && res == 0) {

52       }

57   def defaultMapCompare(dataType: DataType, extension: DataType => Option[(Any, Any) => Int] = noopCompare): Option[(Any, Any) => Int] =
58     dataType match {
59       case mt: MapType => {

61           getOrElse(sys.error(s"Could not find compare function for map key type ${mt.keyType}"))
62 
63         val kr = BoundReference(0, mt.keyType, nullable = true)
64 
65         // actual field is in a key val struct

67           genArrayCompare(keyValueType(mt.keyType, mt.valueType),
68             (left, right) => {
69               val lkv = kr.eval(left.asInstanceOf[InternalRow])
70               val rkv = kr.eval(right.asInstanceOf[InternalRow])
71 
72               val r = kt(lkv, rkv)
73               if (r != 0)
74                 r
75               else {
76                 /* shouldn't happen unless MapData is created directly and it's not
77                    actually a map checking on key uniqueness, here for completeness */
78                 val vt = defaultMapCompare(mt.valueType, extension).

83                 val rvv = vr.eval(right.asInstanceOf[InternalRow])
84 
85                 vt(lvv, rvv)
86               }
87             })
88         )
89       }
90       case dt: AtomicType => Some(compareToOrdering(QualitySparkUtils.sparkOrdering(dt)))
91       case arrayType: ArrayType =>
92         val oelcomp = defaultMapCompare(arrayType.elementType, extension)
93         oelcomp.map{
94           elcomp =>
95             genArrayCompare(arrayType.elementType, elcomp)

101               getOrElse(sys.error(s"Could not find compare function for ${f.dataType} in struct $structType"))
