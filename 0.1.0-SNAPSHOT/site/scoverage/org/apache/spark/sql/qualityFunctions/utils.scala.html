33   def genArrayCompare(elementType: DataType, elComp: (Any, Any) => Int): (Any, Any) => Int = (oleft, oright) => {
34     val left = oleft.asInstanceOf[ArrayData]
35     val right = oright.asInstanceOf[ArrayData]
36     if (left.numElements() > right.numElements())
37       1
38     else if (left.numElements() < right.numElements())
39       -1
40     else {
41       // same
42       var i = 0
43       var res = 0
44       val al = Arrays.toArray(left, elementType)
45       val ar = Arrays.toArray(right, elementType)
46       while (i < left.numElements() && res == 0) {
47         res = elComp(al(i), ar(i))
48         i += 1
49       }
50       res
51     }
52   }
