63   def expr(rule: String) = {
64     val parser = SparkSession.getActiveSession.map(_.sessionState.sqlParser).getOrElse {
65       QualitySparkUtils.newParser()
66     }
67     /*
68     attempt for a simple expression, then try a plan with exactly one output row.
69     "In" will handle ListQuery, "Exists" similarly exists, everything else will likely fail as a parser error
70      */
71     val rawExpr =
72       try {
73         parser.parseExpression(rule)
74       } catch {
75         case ot: Throwable => // e.g. suitable for output expressions
76           try {
77             ScalarSubquery(parser.parsePlan(rule))
78           } catch {
79             case _: Throwable =>
80               // quite possibly a lambda using a subquery so try and force it via ( sub ) trick,
81               // but allow us to replace later after resolution.  DBR > 12 requires () on simple expressions as well
82               val r = rule.split("->")
83               val wrapped =
84                 if (r.size == 2)
85                   s"${r(0)} -> ( ${r(1)} )"
86                 else
87                   s"( ${r(0)} )"
88 
89               try {
90                 parser.parseExpression(wrapped)
91               } catch {
92                 case _: Throwable => throw ot // if this didn't work return the original error
93               }
94           }
95       }
96 
97     val res =
98       rawExpr match {
99         case l: SparkLambdaFunction if hasSubQuery(l) =>
100           // The lambda's will be parsed as UnresolvedAttributes and not the needed lambdas
101           val names = l.arguments.map(a => a.name -> a).toMap
102           l.transform {
103             case s: SubqueryExpression => s.withNewPlan( s.plan.transform{
104               case snippet =>
105                 snippet.transformAllExpressions {
106                   case a: UnresolvedAttribute =>
107                     names.get(a.name).map(lamVar => lamVar).getOrElse(a)
108                 }
109             })
110           }
111         case _ => rawExpr
112       }
113 
114     res

333   private[quality] override def expression() = {
334     val parsed = RuleLogicUtils.expr(rule)
335     // output expressions can be:
336     // 1. simple expressions for ruleEngine
337     // 2. single argument lambda's returning the same type as the arg for folder
338     // 3. as of 0.0.2 #8 set( attribute = valueExpression, attribute = valueExpression) converted to the form of 2 with an updateField call
339     parsed match {
340       case uf: UnresolvedFunction if VariablesLookup.toName(uf) == "set" =>
341         // case 3
342         val args = QualitySparkUtils.arguments(uf)
343         val paired =
344           args.flatMap {
345             case EqualTo(name: UnresolvedAttribute, right) =>
346               // updateField takes paired args of field names to expression
347               Some(Seq(Literal(name.name), right))
348             case a =>
349               logInfo(s"Attempt to convert set OutputExpression argument $a failed as types do not match expected EqualTo(attribute, expression), will default to full expression")
350               None
351           }
352 
353         if (paired.size != args.size)
354           // one of the args didn't match type
355           parsed
356         else
357           // need to keep first arg
358           UpdateFolderExpression.withArgsAndSubstitutedLambdaVariable(paired.flatten)
359       case _ =>
360         // for everything else (1+2) it's already good enough
361         parsed
362     }
363   }
