66

68       if (parent.isEmpty)

74       schema.foldLeft(set) {
75         (s, field) =>
76           val name = withParent(field.name, parent)
77           field.dataType match {
78             case struct: StructType =>
79               accumulate(s + name, struct, name)

82       }
