25   lazy val notAToken =
26     new Parser[String] {
27       def apply(in: Input) = {
28         val source = in.source
29         val offset = in.offset
30         val start = handleWhiteSpace(source, offset)
31         var j = start
32         var break = 0
33         var fromStart = false
34         while (j < source.length && break == 0) {
35           val breakOption =
36             breakingTokens.find { token =>
37               val len = j - token.length
38               if (len > start) {
39                 val comp = source.subSequence(j - token.length, j)
40 
41                 fromStart = false
42                 token == comp
43               } else
44                 if (j + token.length < source.length) {
45                   val comp = source.subSequence(j, j + token.length)
46 
47                   fromStart = true
48                   token == comp
49                 } else
50                   false
51 
52             }
53           break = breakOption.fold(0)(_.length)
54 
55           if (break == 0) {
56             j += 1
57           }
58         }
59 
60         if (break > 0 && fromStart)
61           Failure("breakingToken found at at start of token parsing", in.drop(start - offset))
62         else
63           if (break > 0)
64             Success(source.subSequence(start, j - break).toString, in.drop((j - break) - offset))
65           else
66             Success(source.subSequence(start, j).toString, in.drop(j - offset))
67       }
68     }
69 }
