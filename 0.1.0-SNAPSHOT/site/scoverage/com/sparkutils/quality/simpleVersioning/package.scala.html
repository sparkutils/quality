59     val versionedRules = df.sparkSession.sql(
60       """
61        select ruleExpr, ruleId, ruleVersion, ruleEngineSalience, ruleEngineId, ruleEngineVersion,
62        ruleSetId, ruleSetVersion, -- don't need to bump versions as they can coexist
63         suiteversions.ruleSuiteId, suiteversions.ruleSuiteVersion -- force the versions to be bumped to latest ruleSuite Versions
64         from
65          (select distinct ruleSuiteId, ruleSuiteVersion from rules) suiteversions join
66          rules l0 on l0.ruleSuiteId = suiteversions.ruleSuiteId and l0.ruleSuiteVersion <= suiteversions.ruleSuiteVersion
67          where
68           not exists (
69             select 0 from rules l1
70             where l1.ruleSuiteId = l0.ruleSuiteId
71              and l1.ruleId = l0.ruleId
72              and l1.ruleSetId = l0.ruleSetId
73              and l1.ruleSuiteVersion <= suiteversions.ruleSuiteVersion
74              and l1.ruleVersion > l0.ruleVersion
75           )
76           and l0.ruleExpr != "DELETED"
77        """)

134   protected[quality] def lambdaOutputSQL(tableName: String, extra: String ="") =
135       s"""
136        select $extra ruleExpr, functionId, functionVersion,
137         versions.ruleSuiteId, versions.ruleSuiteVersion -- force the versions to be bumped to latest ruleSuiteVersions
138         from
139          (select distinct ruleSuiteId, ruleSuiteVersion from $tableName) versions join
140          $tableName l0 on l0.ruleSuiteId = versions.ruleSuiteId and l0.ruleSuiteVersion <= versions.ruleSuiteVersion
141          where
142           not exists (
143             select 0 from $tableName l1
144             where l1.ruleSuiteId = l0.ruleSuiteId and
145              l1.functionId = l0.functionId and
146              l1.ruleSuiteVersion <= versions.ruleSuiteVersion
147              and l1.functionVersion > l0.functionVersion
148           )
149           and l0.ruleExpr != "DELETED"
150        """

188   protected[quality] case class SameOrNextVersionLower[T](map: Map[Id, Seq[T]]) extends Function1[Id, Option[Seq[T]]] {
189     implicit val ordering = new Ordering[Id] {
190       def compare(x: Id, y: Id): Int = {
191         if (x == y)
192           0
193         else
194           if (x.id != y.id)
195             x.id - y.id
196           else
197             x.version - y.version
198       }
199     }
200 
201     val sorted = TreeMap[Id, Seq[T]]() ++ map
202 
203     override def apply(id: Id): Option[Seq[T]] =
204       sorted.to(id).lastOption.map(_._2)
205   }
