7 /**
8  * Represents the shared file location of a bucked bloom filter.  There should be files with names 0..numBuckets containing
9  * the same number of bytes representing each bucket.
10  *
11  * @param rootDir  The directory which contains each bucket
12  * @param fpp  The fpp for this bloom - note it is informational only and will not be used in further processing
13  * @param numBuckets The number of buckets within this bloom
14  */
15 case class BucketedFiles(rootDir: String, fpp: Double, numBuckets: Int) extends Serializable {
16   /**
17    * Provides memory mapped buffers from the underlying files
18    * @return
19    */
20   def maps: Seq[IntBuffer] =
21     (0 until numBuckets).map { i =>
22       val file = new File(rootDir, i.toString)
23       val mapped = new FileInputStream(file).getChannel.map(FileChannel.MapMode.READ_ONLY, 0, file.length())
24       mapped.order(ByteOrder.LITTLE_ENDIAN).asIntBuffer()
25     }
26 
27   /**
28    * Serializes the definition of this bucketfiles, not the underlying bytes of the bloom
29    * @return
30    */
31   def serialize: Array[Byte] = {
32     val bos = new ByteArrayOutputStream()
33     val oos = new ObjectOutputStream(bos)
34     oos.writeObject(this)
35     oos.flush()
36     val bytes = bos.toByteArray
37     bos.close()
38     bytes
39   }
40 
41   /**
42    * Only performed lazily
43    * @return
44    */
45   def read: Array[Array[Byte]] = {
46     val size = new File(rootDir, "0").length().toInt // safe as it's never more than an array
47     val ar = Array.ofDim[Array[Byte]](numBuckets)
48     for { i <- 0 until numBuckets} {
49       val f = new File(rootDir, i.toString)
50       val fis = new FileInputStream(f)
51       ar(i) = Array.ofDim[Byte](size)
52       fis.read(ar(i))
53       fis.close()
54     }
55     ar
56   }
57 
58   /**
59    * Removes other directories from the parent root id
60    */
61   def cleanupOthers(): Unit ={
62     val file = new File(rootDir)
63     val name = file.getName
64     val siblings = file.getParentFile.listFiles().filterNot(_.getName == name)
65     siblings.foreach(_.listFiles().foreach(_.delete()))
66     siblings.foreach(_.delete())
67   }
68 
69   /**
70    * Removes this bloom's files, this is advised only after you have processed or otherwise saved it's results.  It will remove everything under this bloomid
71    */
72   def removeThisBloom(): Unit = {
73     val file = new File(rootDir)
74     val siblings = file.getParentFile.listFiles()
75     siblings.foreach(_.listFiles().foreach(_.delete()))
76     siblings.foreach(_.delete())
77   }
78 }
