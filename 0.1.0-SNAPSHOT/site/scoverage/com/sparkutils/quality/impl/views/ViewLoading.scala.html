94   /**
95    * Attempts to load all the views present in the config.  If a view is already registered in that name it will be replaced.
96    * @param viewConfigs
97    * @return the names of views which have been replaced
98    */
99   def loadViews(viewConfigs: Seq[ViewConfig]): ViewLoadResults = {
100 
101     // assume some will not load, attempt count is used to stop cycles, 2x should be enough
102     var attemptCount = 0
103     var done = false
104     var leftToProcess = viewConfigs.map(t => t.name -> t).toMap
105 
106     val replaced = mutable.Set.empty[String]
107     // if we are missing a relation and the name is declared try to load that first
108     var mapOf = leftToProcess
109     var processed = mutable.Set.empty[String]
110 
111     def processView(viewPair: ViewConfig): Unit =
112       if (attemptCount < (viewConfigs.size * 2)) {
113         try {
114           val (name, config) = (viewPair.name, viewPair)
115           if (Validation.defaultViewLookup(name)) {
116             replaced += name
117           }
118 
119           config.source.fold(t => t
120             , sql =>
121               SparkSession.active.sql(sql)
122           ).createOrReplaceTempView(name)
123 
124           // it worked, remove it
125           leftToProcess = leftToProcess - name
126           processed = processed + name
127         } catch {
128           case ae: AnalysisException =>
129             val res = ViewLoader.tableOrViewNotFound(ae)
130             val sql = viewPair.source.right.getOrElse("")
131 
132             res.fold(a => throw ViewLoaderAnalysisException(a, s"AnalysisException for view ${viewPair.name}: $sql", viewPair.name, sql),
133               views => {
134                 val missingNames =
135                   views.flatMap { name =>
136                     if (mapOf.contains(name)) { // quoted must also be in the view name if it's got minus' etc.
137                       attemptCount += 1
138                       processView(mapOf(name))
139                       None
140                     } else // not one we can actually do anything about
141                       Some(name)
142                   }
143                 if (!missingNames.isEmpty) {
144                   throw MissingViewAnalysisException(ae, s"Missing relations for view ${viewPair.name}: $missingNames used in sql $sql", viewPair.name, sql, missingNames)
145                 }
146               }
147             )
148         }
149       }
150 
151     while((attemptCount < (viewConfigs.size * 2)) && !done){
152       attemptCount += 1
153       leftToProcess.headOption.fold {
154         done = true;
155       } { p =>
156         processView(p._2)
157       }
158     }
159     ViewLoadResults(replaced.toSet, !done, leftToProcess.keySet)
160   }

177     ae.plan.fold[Either[AnalysisException, Set[String]]]{
178       // spark 2.4 just has exception: Table or view not found: names
179       if (ae.message.contains("Table or view not found"))
180         Right(Set(ae.message.split(":")(1).trim))
181       else
182         Left(ae)
183     } {
184       plan =>
185         val c =
186           plan.collect {
187             case ur: UnresolvedRelation =>
188               ur.tableName
189           }
190 
191         if (c.isEmpty)
192           Left(ae) // not what we expected
193         else
194           Right(c.toSet)
195     }
