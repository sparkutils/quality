16 case class HashFunctionProxy(hasher: Hasher) extends Digest {
17 
18   override def hashInt(i: Int): Unit =
19     hasher.putInt(i)
20 
21   override def hashLong(l: Long): Unit =
22     hasher.putLong(l)

24   override def hashBytes(base: Array[Byte], offset: Int, length: Int): Unit =
25     hasher.putBytes(base, offset, length)

27   override def digest: Array[Long] = {
28     val bytes = hasher.hash.asBytes
29     val res = Array.ofDim[Long]((bytes.length + 7) / 8) // next whole long size up
30 
31     // to make sure the array is large enough pad as needed, so a 32bit digest will always take 64
32     val use =
33       if (bytes.length / 8 != res.length) {
34         val nbytes = Array.ofDim[Byte](res.length * 8)
35         Array.copy(bytes, 0, nbytes, 0, bytes.length)
36         nbytes
37       } else
38         bytes
39 
40     for{ i <- 0 until res.length} {
41       res(i) = BytePackingUtils.unencodeLong(i * 8, use)
42     }
43 
44     res
45   }
46 }
