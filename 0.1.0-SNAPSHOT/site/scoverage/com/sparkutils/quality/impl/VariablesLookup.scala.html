25  * Provides a variable lookup function, after using the sql parser it will return all the fields used in an expression,
26  * allowing sanity checks on rules to use only expected fields but also to attribute how much a rule does - does it check
27  * just one field or use 20 of them.
28  * It is also used to identify fields which are note provided by a lambda i.e. the ones bound at use.
29  * Note: this cannot process nested lambdas in a simple expression unless the lambdas are also passed in, so process lambdas first.
30  */

85       def faccumulate(identifiers: Identifiers, expression: Expression): Identifiers =
86         expression match {
87           case a : UnresolvedNamedLambdaVariable =>
88             val full = toName(a)
89             if (!args.contains(full)) // don't accept args, so we should only be left with outer scopes, which may be from a nested..
90               identifiers + full
91             else
92               identifiers
93 
94           case f @ UnresolvedFunction(functionIdentifier, argumentExpressions, _, _, _) => // nested....
95             val name = toName(functionIdentifier)
96             val nids =
97               if (evaluatedLambdas.contains(name))
98                 identifiers
99               else {
100                 if (lambdaExpressions.contains(name)) {
101                   // we haven't yet evaluated it, pass back up to the top and recurse down
102                   if ((parent ne null) && toName(functionIdentifier) == toName(parent.nameParts)) {
103                     // special case for recursion on the same identifier - are we calling the same id?
104                     // get the exact arity matching
105                     lambdaExpressions(name).find(_._2.children.size == argumentExpressions.size).fold{
106                       overflowIds += id
107                       logger.warn(s"Function ${name} calls itself, this may StackOverflowError on evaluation")
108                     }{
109                       i =>
110                         val r = children(Map.empty, Seq(i), f)
111                         evaluatedLambdas(name) = r
112                     }
113                   } else {
114                     val r = children(Map.empty, lambdaExpressions(name).toSeq, f)
115                     evaluatedLambdas(name) = r
116                   }
117                   identifiers
118                 } else {
119                   // it's not a lambda function we know, is it inbuilt?
120                   // NB you would have to register UDFs etc. before calling validate etc.
121                   val name = toName(functionIdentifier)
122                   val exists =
123                     SparkSession.active.catalog.functionExists(name)
124 
125                   if (!exists) {
126                     // add it in to the unknowns list
127                     val map = unknownSparkFunctionIds.getOrElse(id, Set.empty)
128                     unknownSparkFunctionIds(id) = map + name
129                   }
130                   identifiers
131                 }
132               }
133             // params may be including nested children
134             fieldChildren(nids, argumentExpressions)
135           case p : Expression => fieldChildren(identifiers, p.children)
136           case _  => identifiers
137         }
138 
139       faccumulate(ids, expr)
140     }

142     def accumulate(res: Map[Id, Identifiers], exp: (Id, Expression), parent: UnresolvedFunction): Map[Id, Identifiers] =
143       exp match {
144         // unresolved case where we cannot see more unresolved functions
145         case (id, SparkLambdaFunction(functionExpr, arguments, _)) =>
146           // remove the arguments from any unresolved bound variables
147           val names = arguments.map(v => toName(v.asInstanceOf[UnresolvedNamedLambdaVariable])).toSet
148           // parse the functionExpr with the names
149           Map( id -> processFields(names, functionExpr, id, parent)) //TODO is this parent?
150         case (id, a : UnresolvedAttribute) => // not as part of a lambda
151           val s = res.getOrElse(id, Set.empty)
152           res + ( id -> (s + toName(a.nameParts)) )
153         case (id, _ : LeafExpression) => res
154         case (id, parent: UnresolvedFunction) => res |+| children(res, parent.children.map((id,_)), parent) // override
155         case (id, newparent: Expression) => res |+| children(res, newparent.children.map((id,_)), parent)
156       }
157 
158     val ids = children(Map.empty, exprMap.toSeq, null)
159 
160     (( m + (name -> ids) ) ++ evaluatedLambdas, Set() ++ overflowIds, Map() ++ unknownSparkFunctionIds)

176     def accumulate(res: ExpressionLookup, exp: Expression): ExpressionLookup =
177       exp match {
178         // unresolved case where we cannot see more unresolved functions
179         case UnresolvedFunction(nameParts, arguments, _, _, _) =>
180           val name = toName(nameParts)
181           val r =
182             if (knownLambdaLookups.contains(name)) {
183               val lambdas = knownLambdaLookups(name)
184               res.copy(attributesUsed = res.attributesUsed ++ lambdas.flatMap(_._2), lambdas = res.lambdas ++ lambdas.keySet)  // merge the identifier set and lambdas
185             } else {
186               // it's not a lambda function we know, is it inbuilt?
187               // NB you would have to register UDFs etc. before calling validate etc.
188               val exists =
189                 SparkSession.active.catalog.functionExists(name)
190 
191               if (!exists)
192                 // add it in to the unknowns list
193                 res.copy(unknownSparkFunctions = res.unknownSparkFunctions + name)
194               else
195                 res.copy(sparkFunctions = res.sparkFunctions + name)
196             }
197 
198           // we still need to do the args
199           children(r, arguments)
200         case a : UnresolvedAttribute =>
201           res.copy(attributesUsed = res.attributesUsed + a.name)
202         // typically handled by the lambda functions above, but for coalesce this doesn't work, we need sub expression handling
203         case a : UnresolvedNamedLambdaVariable =>
204           res.copy(attributesUsed = res.attributesUsed + a.name)
205         case _ : LeafExpression => res
206         case parent: Expression => children(res, parent.children)
207       }
208 
209     val ids = accumulate(ExpressionLookup(), expr)
210 
211     ids
212   }
