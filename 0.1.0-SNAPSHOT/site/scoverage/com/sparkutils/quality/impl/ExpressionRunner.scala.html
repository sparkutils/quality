24    * @param ruleSuite

28   def apply(ruleSuite: RuleSuite, name: String = "expressionResults"): Column = {
29     com.sparkutils.quality.registerLambdaFunctions( ruleSuite.lambdaFunctions )
30     val expressions = flattenExpressions(ruleSuite)
31     val collectExpressions = expressions.map( i => cast(i, StringType))
32     new Column(ExpressionRunner(ruleSuite, collectExpressions)).as(name)
33   }
