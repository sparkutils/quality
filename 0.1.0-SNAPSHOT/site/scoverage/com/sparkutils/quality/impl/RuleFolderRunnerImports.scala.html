15   /**
16    * Creates a column that runs the folding RuleSuite.  This also forces registering the lambda functions used by that RuleSuite.
17    *
18    * FolderRunner runs all output expressions for matching rules in order of salience, the startingStruct is passed ot the first
19    * matching, the result passed to the second etc.  In contrast to ruleEngineRunner OutputExpressions should be lambdas with one parameter, that of the structure
20    *
21    * @param ruleSuite The ruleSuite with runOnPassProcessors
22    * @param startingStruct This struct is passed to the first matching rule, ideally you would use the spark dsl struct function to refer to existing columns
23    * @param compileEvals Should the rules be compiled out to interim objects - by default true
24    * @param debugMode When debugMode is enabled the resultDataType is wrapped in Array of (salience, result) pairs to ease debugging
25    * @param resolveWith This experimental parameter can take the DataFrame these rules will be added to and pre-resolve and optimise the sql expressions, see the documentation for details on when to and not to use this.
26    * @param variablesPerFunc Defaulting to 40 allows, in combination with variableFuncGroup allows customisation of handling the 64k jvm method size limitation when performing WholeStageCodeGen
27    * @param variableFuncGroup Defaulting to 20
28    * @param forceRunnerEval Defaulting to false, passing true forces a simplified partially interpreted evaluation (compileEvals must be false to get fully interpreted)
29    * @param forceTriggerEval Defaulting to true, passing true forces each trigger expression to be compiled (compileEvals) and used in place, false instead expands the trigger in-line giving possible performance boosts based on JIT.  Most testing has however shown this not to be the case hence the default, ymmv.
30    * @param useType In the case you must use select and can't use withColumn you may provide a type directly to stop the NPE
31    * @return A Column representing the QualityRules expression built from this ruleSuite
32    */
33   def ruleFolderRunner(ruleSuite: RuleSuite, startingStruct: Column, compileEvals: Boolean = true,
34                        debugMode: Boolean = false, resolveWith: Option[DataFrame] = None, variablesPerFunc: Int = 40,
35                        variableFuncGroup: Int = 20, forceRunnerEval: Boolean = false, useType: Option[StructType] = None,
36                        forceTriggerEval: Boolean = true): Column = {

38 
39     // needed to resolve variables

41 
42     val realType = () => {
43       val starter =
44         useType.getOrElse(dataRef.get())
45       if (debugMode)
46       // wrap it in an array with the priority result
47         ArrayType(StructType(Seq(StructField("salience", IntegerType), StructField("result", starter))))
48       else
49         starter
50     }
51 
52     val liftLambda = (e: Expression) => FunN(Seq(RefExpressionLazyType(() => dataRef.get(), true)), e)
53 
54     val (expressions, indexes) = flattenExpressions(ruleSuite, liftLambda)
55 
56     val runner = new RuleFolderRunner(ruleSuite, startingStruct.expr, PassThrough( expressions ), realType, compileEvals = compileEvals,
57       debugMode = debugMode, variablesPerFunc, variableFuncGroup, forceRunnerEval = forceRunnerEval, expressionOffsets = indexes, dataRef, forceTriggerEval)
58 
59     new Column(
60       QualitySparkUtils.resolveWithOverride(resolveWith).map { df =>
61         val resolved = QualitySparkUtils.resolveExpression(df, runner)
62 
63         resolved.asInstanceOf[RuleFolderRunner].copy(right = resolved.children(1) match {
64           // replace the expr
65           case PassThrough(children) => NonPassThrough(children)
66         })
67       } getOrElse runner
68     )
