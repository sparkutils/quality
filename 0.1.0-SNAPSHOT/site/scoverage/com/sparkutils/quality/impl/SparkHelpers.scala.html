119   /**
120    * Assuming MapData entries are stable, as they are array backed, over a dataset.  When the position of 'what' is not in the cachedPositions a full search is performed with the new location being returned in addition.
121    *
122    * @param mapData the source mapdata to search through.
123    * @param cachedPositions already known positions to check for what
124    * @param what compared against the keyArray to obtain the position
125    * @param getValue obtain the value from the mapData, with any necessary conversions, against the index parameter
126    * @tparam T
127    * @return (result from getValue or null when not found, either cachedPositions or cachedPositions and a newly found position)
128    */
129   def getMapEntry[T <: AnyRef](mapData: MapData, cachedPositions: Seq[Int], what: Any)(getValue: Int => T): (T, Seq[Int]) = {
130     val keys = mapData.keyArray()
131     val n = keys.numElements()
132     val acc = (idx: Int) => keys.getLong(idx)
133 
134     def search() = {
135       var i = 0
136       var found = false
137       while (i < n && !found) {
138         if (acc(i) == what) {
139           found = true
140         } else {
141           i += 1
142         }
143       }
144       if (!found)
145         -1
146       else
147         i
148     }
149 
150     def withCached() =
151       cachedPositions.find { i =>
152         if (i < n)
153           acc(i) == what
154         else
155           false
156       }
157 
158     withCached().map(i => (getValue(i), cachedPositions))
159       .getOrElse {
160         val r = search()
161         if (r == -1)
162           (null.asInstanceOf[T], cachedPositions)
163         else
164           (getValue(r), cachedPositions :+ r)
165       }
