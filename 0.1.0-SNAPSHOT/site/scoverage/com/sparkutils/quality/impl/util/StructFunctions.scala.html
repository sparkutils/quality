190 object UpdateFields {
191   private def nameParts(fieldName: String): Seq[String] = {
192     require(fieldName != null, "fieldName cannot be null")
193 
194     if (fieldName.isEmpty) {
195       fieldName :: Nil
196     } else {
197       UnresolvedAttribute.parseAttributeName(fieldName)
198       //CatalystSqlParser.parseMultipartIdentifier(fieldName)
199     }
200   }
201 
202   /**
203    * Adds/replaces field of `StructType` into `col` expression by name.
204    */
205   def apply(col: Expression, fieldName: String, expr: Expression): UpdateFields =
206     updateFieldsHelper(col, nameParts(fieldName), name => WithField(name, expr))
207 
208   /**
209    * Drops fields of `StructType` in `col` expression by name.
210    */
211   def apply(col: Expression, fieldName: String): UpdateFields =
212     updateFieldsHelper(col, nameParts(fieldName), name => DropField(name))
213 
214   private def updateFieldsHelper(
215                                   structExpr: Expression,
216                                   namePartsRemaining: Seq[String],
217                                   valueFunc: String => StructFieldsOperation) : UpdateFields = {
218     val fieldName = namePartsRemaining.head
219     if (namePartsRemaining.length == 1) {
220       UpdateFields(Seq(structExpr, valueFunc(fieldName)))
221     } else {
222       val newStruct = if (structExpr.resolved) {
223         val resolver = SQLConf.get.resolver
224         ExtractValue(structExpr, Literal(fieldName), resolver)
225       } else {
226         UnresolvedExtractValue(structExpr, Literal(fieldName))
227       }
228 
229       val newValue = updateFieldsHelper(
230         structExpr = newStruct,
231         namePartsRemaining = namePartsRemaining.tail,
232         valueFunc = valueFunc)
233       UpdateFields(Seq(structExpr, WithField(fieldName, newValue) ))
234     }
235   }
