80   def namesFromSchema(schema: StructType): Set[String] = {
81 
82     def withParent(name: String, parent: String) =
83       if (parent.isEmpty)
84         name
85       else
86         parent + "." + name
87 
88     def accumulate(set: Set[String], schema: StructType, parent: String): Set[String] =
89       schema.foldLeft(set) {
90         (s, field) =>
91           val name = withParent(field.name, parent)
92           field.dataType match {
93             case struct: StructType =>
94               accumulate(s + name, struct, name)
95             case _ => s + name
96           }
97       }
98 
99     accumulate(Set.empty, schema, "")
100   }

106    */
107   def identifyLookups(ruleSuite: RuleSuite): LookupResults = {

110         val exp = RuleLogicUtils.expr(r.rule)

112       }.foldLeft(ExpressionLookupResults[Id](Map.empty, Set.empty)) {
113         (acc, res) =>
114           val r = acc.copy( lookupConstants = acc.lookupConstants + (res._2.id -> res._1.constants))

116             r.copy(lookupExpressions = r.lookupExpressions + res._2.id)
117           else

119       }

122     LookupResults(ruleSuite, ExpressionLookupResults(Map.empty, Set.empty), lambdaResults)
