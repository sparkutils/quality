32   /**
33    * Calculate optimal size according to the number of distinct values and false positive probability.
34    *
35    * @param n : The number of distinct values.
36    * @param p : The false positive probability.
37    * @return optimal number of bits of given n and p.
38    */
39   def optimalNumOfBits(n: Long, p: Double): Int = {
40     require(p > 0.0 && p < 1.0, "FPP should be less than 1.0 and great than 0.0")
41     val m = -8 * n / Math.log(1 - Math.pow(p, 1.0 / 8))
42     var numBits = m.toInt // 2147483647
43     // Handle overflow.
44     if (numBits > (UPPER_BOUND_BYTES << 3) || m < 0) {
45       numBits = UPPER_BOUND_BYTES << 3 // 1073741824
46     }
47 
48     // Round numBits up to (k * BITS_PER_BLOCK)
49     numBits = (numBits + BITS_PER_BLOCK - 1) & ~BITS_PER_BLOCK // 1073742079
50 
51     if (numBits < (LOWER_BOUND_BYTES << 3)) {
52       numBits = LOWER_BOUND_BYTES << 3
53     }
54     numBits
55   }

66     var numBytes = optimalNumOfBits(n, p) / 8 // 134217759

72       (numBytes.toDouble / origBytes.toDouble).ceil.toLong
73     else
74       (origBytes.toDouble / numBytes.toDouble).ceil.toLong

79 /*
80  * This Bloom filter is implemented using block-based Bloom filter algorithm from Putze et al.'s
81  * "Cache-, Hash- and Space-Efficient Bloom filters". The basic idea is to hash the item to a tiny
82  * Bloom filter which size fit a single cache line or smaller. This implementation sets 8 bits in
83  * each tiny Bloom filter. Each tiny Bloom filter is 32 bytes to take advantage of 32-byte SIMD
84  * instruction.
85  */

87   private[parquet] val BYTES_PER_BLOCK = 32
88   // Bits in a tiny Bloom filter block.
89   private[parquet] val BITS_PER_BLOCK = 256
90   // The lower bound of bloom filter size, set to the size of a tiny Bloom filter block.
91   val LOWER_BOUND_BYTES = 32
92   // The upper bound of bloom filter size, set to default row group size.
93   val UPPER_BOUND_BYTES: Int = 128 * 1024 * 1024
94   // The number of bits to set in a tiny Bloom filter
95   private[parquet] val BITS_SET_PER_BLOCK = 8
96   // The metadata in the header of a serialized Bloom filter is four four-byte values: the number of bytes,
97   // the filter algorithm, the hash algorithm, and the compression.

246   def insertHash(hash: Long): Unit = {
247     val numBlocks = getBitsetSize / BlockSplitBloomFilterImpl.BYTES_PER_BLOCK
248     val lowHash = hash >>> 32
249     val blockIndex = ((lowHash * numBlocks) >> 32).toInt
250     val key = hash.toInt
251     // Calculate mask for bucket.
252     val mask = setMask(key)
253     for (i <- 0 until BlockSplitBloomFilterImpl.BITS_SET_PER_BLOCK) {
254       var value = intBuffer.get(blockIndex * (BlockSplitBloomFilterImpl.BYTES_PER_BLOCK / 4) + i)
255       value |= mask(i)
256       intBuffer.put(blockIndex * (BlockSplitBloomFilterImpl.BYTES_PER_BLOCK / 4) + i, value)
257     }
258   }
