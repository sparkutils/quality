180 case class BucketedCreator[H, SerializedType](numBytes: Int, iMinimumBytes: Int, iMaximumBytes: Int, hashImpl: BloomHash, fpp: Double, numBuckets: Int,
181                                            toType: ToSerializedType[SerializedType, H], hint: H)(
182   val filters: Seq[BlockSplitBloomFilterImpl] = (1 to numBuckets).map(_ => BlockSplitBloomFilterImpl(numBytes, iMinimumBytes, iMaximumBytes, hashImpl)),
183   val bucketedFiles: Option[BloomModel] = None
184 ) extends Bloom[SerializedType] {
185   type BloomType = BucketedCreator[H, SerializedType]
186 
187   /** CTw added the or'ing */
188   def |= (that: BucketedCreator[H, SerializedType]): BucketedCreator[H, SerializedType] = {
189     checkCompatibility(that)
190     // Breeze copies but we'll do in place
191     for(i <- 0 until numBuckets) {
192       filters(i) |= that.filters(i)
193     }
194     this
195   }
196 
197   private def checkCompatibility(that: BucketedCreator[H, SerializedType]): Unit =
198     require(iMaximumBytes == that.iMaximumBytes
199       && iMinimumBytes == that.iMinimumBytes &&
200       hashImpl.hashStrategy == that.hashImpl.hashStrategy
201       && filters.zip( that.filters ).forall( p => p._1.intBuffer.limit() == p._2.intBuffer.limit() )
202       , "Must have the hash and bitset length to intersect")
203 
204   def += (value: Any) = {
205     val h = hashImpl.hash(value)
206     filters(Bucketed.whichBloom(h, numBuckets)).insertHash(h)
207     this.asInstanceOf[BloomType]
208   }
209 
210   /**
211    * Combines all the buckets together via the serialize type.  If this is already linked to memory mapped files
212    * do not re-write, just serialize the bucketedFiles reference
213    * @return combined serialized blooms
214    */
215   override def serialized: Array[Byte] = bucketedFiles.fold(toType.serializeBuckets(filters, fpp, numBuckets, hint))(_.serialize)
