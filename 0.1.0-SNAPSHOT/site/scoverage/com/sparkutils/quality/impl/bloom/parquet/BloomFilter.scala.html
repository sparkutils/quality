197 private[bloom] trait BloomHash extends Serializable {
198   def hashStrategy: BloomFilter.HashStrategy
199 
200   val hashFunction =
201     hashStrategy match {
202       case BloomFilter.XXH64 =>
203         new XxHash()
204       case _ =>
205         throw new RuntimeException("Unsupported hash strategy")
206     }
207 
208   def cacheBuffer: ByteBuffer
209   def mask: Array[Int]
210 
211   def setMask(key: Int) = { // The following three loops are written separately so that they could be
212     // optimized for vectorization.
213     for (i <- 0 until BlockSplitBloomFilterImpl.BITS_SET_PER_BLOCK) {
214       mask(i) = key * BlockSplitBloomFilterImpl.SALT(i)
215     }
216     for (i <- 0 until BlockSplitBloomFilterImpl.BITS_SET_PER_BLOCK) {
217       mask(i) = mask(i) >>> 27
218     }
219     for (i <- 0 until BlockSplitBloomFilterImpl.BITS_SET_PER_BLOCK) {
220       mask(i) = 0x1 << mask(i)
221     }
222     mask
223   }
224 
225   def hash(value: Any): Long = {
226     value match {
227       case b: BloomFilter.Binary => return hashFunction.hashBytes(b)
228       case wa: mutable.WrappedArray[_] =>
229         val ar = wa.asInstanceOf[mutable.WrappedArray[Long]]
230         return hashFunction.hashLongs(ar.toArray)
231       case la: Array[Long] => return hashFunction.hashLongs(la)
232       case i: Integer => cacheBuffer.putInt(i)
233       case l: Long => cacheBuffer.putLong(l)
234       case f: Float => cacheBuffer.putFloat(f)
235       case d: Double => cacheBuffer.putDouble(d)
236       case s: String => return hashFunction.hashChars(s)
237       case _ => throw new RuntimeException("Parquet Bloom filter: Not supported type")
238     }
239     doHash
240   }
241 
242   def doHash = {
243     cacheBuffer.flip
244     val hashResult = hashFunction.hashByteBuffer(cacheBuffer)
245     cacheBuffer.clear
246     hashResult
247   }
248 
249 }

262 private[bloom] trait BloomLookupImpl extends Serializable with com.sparkutils.quality.BloomLookup with BloomHash {
263   def intBuffer: IntBuffer
264 
265   def findHash(hash: Long): Boolean = {
266     val numBlocks = (intBuffer.limit() * 4) / BlockSplitBloomFilterImpl.BYTES_PER_BLOCK
267     val lowHash = hash >>> 32
268     val blockIndex = ((lowHash * numBlocks) >> 32).toInt
269     val key = hash.toInt
270     // Calculate mask for the tiny Bloom filter.
271     val mask = setMask(key)
272     for (i <- 0 until BlockSplitBloomFilterImpl.BITS_SET_PER_BLOCK) {
273       if (0 == (intBuffer.get(blockIndex * (BlockSplitBloomFilterImpl.BYTES_PER_BLOCK / 4) + i) & mask(i))) return false
274     }
275     true
276   }
277 
278   def mightContain( value : Any) = {
279     val h = hash(value)
280     findHash(h)
281   }
