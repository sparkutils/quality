70     dataType match {
71       case _: MapType => Some((left, right) => MapMerge(Seq(left, right), (dataType: DataType) => defaultAdd(dataType, extension)))
72       case _: IntegerType | LongType | DoubleType =>
73         Some((left, right) => add(left, right, null))
74       case a: DecimalType =>
75         Some((left, right) => add(left, right, a))
76       case _ => extension(dataType)
77     }
