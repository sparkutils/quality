77     dataType match {
78       case _: MapType => Some((left, right) => MapMerge(Seq(left, right), (dataType: DataType) => defaultAdd(dataType, extension)))
79       case _: IntegerType | LongType | DoubleType =>
80         Some((left, right) => add(left, right, null))
81       case a: DecimalType =>
82         Some((left, right) => add(left, right, a))
83       case _ => extension(dataType)
84     }

121     val create =
122       if (paramNumbers.isEmpty && minimum == -1 )

124       else
125         (exps: Seq[Expression]) => {
126           if ((paramNumbers.nonEmpty && !paramNumbers.contains(exps.size)) || (minimum > exps.size)) {
127             val sizeerr =
128               if (paramNumbers.nonEmpty)
129                 s"Valid parameter counts are ${paramNumbers.mkString(", ")}"
130               else
131                 s"A minimum of $minimum parameters is required."
132             throw QualityException(s"Wrong number of arguments provided to Quality function $name. $sizeerr")
133           }
134           argsf(exps)
135         }
136     registerFunction(name, create)
137     if (!mustKeepNames(name)) {
138       registerFunction(name.replaceAll("_",""), create)
139     }
