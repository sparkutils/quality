71     dataType match {
72       case _: MapType => Some((left, right) => MapMerge(Seq(left, right), (dataType: DataType) => defaultAdd(dataType, extension)))
73       case _: IntegerType | LongType | DoubleType =>
74         Some((left, right) => add(left, right, null))
75       case a: DecimalType =>
76         Some((left, right) => add(left, right, a))
77       case _ => extension(dataType)
78     }
