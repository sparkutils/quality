53    * Same as functions.expr without the wrapping Column
54    * @param rule
55    * @return

57   def expr(rule: String) = {
58     val parser = SparkSession.getActiveSession.map(_.sessionState.sqlParser).getOrElse {
59       QualitySparkUtils.newParser()
60     }
61     /*
62     attempt for a simple expression, then try a plan with exactly one output row.
63     "In" will handle ListQuery, "Exists" similarly exists, everything else will likely fail as a parser error
64      */
65     val rawExpr =
66       try {
67         parser.parseExpression(rule)
68       } catch {
69         case ot: Throwable => // e.g. suitable for output expressions
70           try {
71             ScalarSubquery(parser.parsePlan(rule))
72           } catch {
73             case _: Throwable =>
74               // quite possibly a lambda using a subquery so try and force it via ( sub ) trick,
75               // but allow us to replace later after resolution.  DBR > 12 requires () on simple expressions as well
76               val r = rule.split("->")
77               val wrapped =
78                 if (r.size == 2)
79                   s"${r(0)} -> ( ${r(1)} )"
80                 else
81                   s"( ${r(0)} )"
82 
83               try {
84                 parser.parseExpression(wrapped)
85               } catch {
86                 case _: Throwable => throw ot // if this didn't work return the original error
87               }
88           }
89       }
90 
91     val res =
92       rawExpr match {
93         case l: SparkLambdaFunction if hasSubQuery(l) =>
94           // The lambda's will be parsed as UnresolvedAttributes and not the needed lambdas
95           val names = l.arguments.map(a => a.name -> a).toMap
96           l.transform {
97             case s: SubqueryExpression => s.withNewPlan( s.plan.transform{
98               case snippet =>
99                 snippet.transformAllExpressions {
100                   case a: UnresolvedAttribute =>
101                     names.get(a.name).map(lamVar => lamVar).getOrElse(a)
102                 }
103             })
104           }
105         case _ => rawExpr
106       }
107 
108     res
109   }

115   def anyToRuleResult(any: Any): RuleResult =
116     any match {
117       case b: Boolean => if (b) Passed else Failed
118       case 0 | 0.0 => Failed
119       case 1 | 1.0 => Passed

122       case d: Double => Probability(d) // only spark 2 unless configured to behave like spark 2
123       case d: Decimal => Probability(d.toDouble)
124       case "Passed" | "true" | "True" | "passed" | "Pass"| "pass" | "Yes" | "yes" | "1" | "1.0" => Passed
125       case "Failed" | "false" | "False" | "failed" | "Fail"| "fail" | "No" | "no" | "0" | "0.0" => Failed
126       case _ => Failed // anything else is a fail
127     }
