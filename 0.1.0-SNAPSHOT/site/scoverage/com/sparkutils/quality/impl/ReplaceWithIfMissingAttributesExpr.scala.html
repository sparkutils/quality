16   /**
17    * Processes a given RuleSuite to replace any coalesceIfMissingAttributes.  This may be called before validate / docs but
18    * *must* be called *before* adding the expression to a dataframe.
19    *
20    * @param ruleSuite
21    * @param schema The names to validate against, if empty no attempt to process coalesceIfAttributeMissing will be made
22    * @return
23    */
24   def processIfAttributeMissing(ruleSuite: RuleSuite, schema: StructType = StructType(Seq())) = {
25     val names = LookupIdFunctions.namesFromSchema(schema)
26 
27     val lambdas = ruleSuite.lambdaFunctions.map(lf => ProcessDisableIfMissing.processIfMissingLambdaCoalesce(lf.parsed, names))
28 
29     val wrs = ruleSuite.copy(lambdaFunctions = lambdas)
30 
31     mapRules(wrs) {
32       ProcessDisableIfMissing.processCoalesceIfAttributeMissing(_, names)
33     }
34   }

37   /**
38    * For a given schema evaluates all calls to coalesce in a lambda to swap out missing attributes
39    *
40    * Must be called before any planning - otherwise Spark will already have rejected it.
41    *
42    * @param rule
43    * @return
44    */
45   def processIfMissingLambdaCoalesce(rule: LambdaFunctionParsed, names: Set[String]): LambdaFunction =
46     rule.expr match {
47       case lambdaFunc: SparkLambdaFunction =>
48         lambdaFunc.function match {
49           case func: UnresolvedFunction =>
50             val rexp = processCoalesceIfAttributeMissing(func, names)
51             rule.copy(expr = lambdaFunc.copy(function = rexp))
52 
53           case _ => rule
54         }
55     }

67   /**
68    * Converts expression removing any coalesceIfAttributeMissing placeholding expressions using names to find each
69    * first expression with no missing names (or replacing with null should all of them be).
70    * @param expression
71    * @param names
72    * @return
73    */
74   def processCoalesceIfAttributeMissing(expression: Expression, names: Set[String]): Expression =
75     expression match {
76       case funcExpr: UnresolvedFunction if isReplaceCoalesceName(funcExpr) =>
77         processCoalesceIfAttributeMissing(expression.children, names, isCoalesceDisabled(funcExpr))
78       case e: Expression =>
79         e.withNewChildren(e.children.map(processCoalesceIfAttributeMissing(_, names)))
80     }
81 
82   /**
83    * Returns the first expression that has no missing names or null
84    * @param expressions
85    * @param names
86    * @return
87    */
88   protected def processCoalesceIfAttributeMissing(expressions: Seq[Expression], names: Set[String], isDisabled: Boolean): Expression = {
89     val res =
90       expressions.flatMap{ e =>
91         // make sure no nested children are coalesce's
92         val r = processCoalesceIfAttributeMissing(e, names)
93         // get the names from this expression
94         val explookup = fieldsFromExpression(r)
95         // lookup each of them in the names and if any are not true we can't use this expression
96         if (!explookup.attributesUsed.map(names).forall(identity))
97           None
98         else
99           Some(r)
100       }
101 
102     if (res.isEmpty) {
103       if (isDisabled) DisabledRuleExpr // special case as this is probably before the function registry gets called
104       else
105         Literal(null, NullType)
106     } else
107       res.head
108   }
109 
110   /**
111    * Converts both rule expressions and any output expressions removing any coalesceIfAttributeMissing placeholding expressions using names to find each
112    * first expression with no missing names (or replacing with null should all of them be).
113    * @param names
114    * @return
115    */
116   protected[quality] def processCoalesceIfAttributeMissing(rule: Rule, names: Set[String]): Rule =
117     rule match {
118       // rule and output
119       case Rule(id, ExpressionRule(rule: String),
120         iorule @ RunOnPassProcessorImpl(_, _, _, OutputExpression(oruleExpr: String))) if oruleExpr.nonEmpty =>
121         Rule(id, ExpressionRuleExpr(rule, processCoalesceIfAttributeMissing(RuleLogicUtils.expr(rule), names)),
122           iorule.copy(returnIfPassed = OutputExpressionExpr(oruleExpr,
123             processCoalesceIfAttributeMissing(RuleLogicUtils.expr(oruleExpr), names))))
124 
125       // just a rule
126       case orule @ Rule(_, ExpressionRule(rule: String), _) =>
127         orule.copy( expression = ExpressionRuleExpr(rule, processCoalesceIfAttributeMissing(RuleLogicUtils.expr(rule), names)))
128 
129       case _ => rule
130     }
