14   /**
15    * Processes a given RuleSuite to replace any coalesceIfMissingAttributes.  This may be called before validate / docs but
16    * *must* be called *before* adding the expression to a dataframe.
17    *
18    * @param ruleSuite
19    * @param schema The names to validate against, if empty no attempt to process coalesceIfAttributeMissing will be made
20    * @return
21    */
22   def processIfAttributeMissing(ruleSuite: RuleSuite, schema: StructType = StructType(Seq())) = {
23     val names = LookupIdFunctions.namesFromSchema(schema)
24 
25     val lambdas = ruleSuite.lambdaFunctions.map(lf => ProcessDisableIfMissing.processIfMissingLambdaCoalesce(lf.parsed, names))
26 
27     val wrs = ruleSuite.copy(lambdaFunctions = lambdas)
28 
29     mapRules(wrs) {
30       ProcessDisableIfMissing.processCoalesceIfAttributeMissing(_, names)
31     }
32   }

35   /**
36    * For a given schema evaluates all calls to coalesce in a lambda to swap out missing attributes
37    *
38    * Must be called before any planning - otherwise Spark will already have rejected it.
39    *
40    * @param rule
41    * @return
42    */
43   def processIfMissingLambdaCoalesce(rule: LambdaFunctionParsed, names: Set[String]): LambdaFunction =
44     rule.expr match {
45       case lambdaFunc: SparkLambdaFunction =>
46         lambdaFunc.function match {
47           case func: UnresolvedFunction =>
48             val rexp = processCoalesceIfAttributeMissing(func, names)
49             rule.copy(expr = lambdaFunc.copy(function = rexp))
50 
51           case _ => rule
52         }
53     }

65   /**
66    * Converts expression removing any coalesceIfAttributeMissing placeholding expressions using names to find each
67    * first expression with no missing names (or replacing with null should all of them be).
68    * @param expression
69    * @param names
70    * @return
71    */
72   def processCoalesceIfAttributeMissing(expression: Expression, names: Set[String]): Expression =
73     expression match {
74       case funcExpr: UnresolvedFunction if isReplaceCoalesceName(funcExpr) =>
75         processCoalesceIfAttributeMissing(expression.children, names, isCoalesceDisabled(funcExpr))
76       case e: Expression =>
77         e.withNewChildren(e.children.map(processCoalesceIfAttributeMissing(_, names)))
78     }

80   /**
81    * Returns the first expression that has no missing names or null
82    * @param expressions
83    * @param names
84    * @return
85    */
86   protected def processCoalesceIfAttributeMissing(expressions: Seq[Expression], names: Set[String], isDisabled: Boolean): Expression = {
87     val res =
88       expressions.flatMap{ e =>
89         // make sure no nested children are coalesce's
90         val r = processCoalesceIfAttributeMissing(e, names)
91         // get the names from this expression
92         val explookup = fieldsFromExpression(r)
93         // lookup each of them in the names and if any are not true we can't use this expression
94         if (!explookup.attributesUsed.map(names).forall(identity))
95           None
96         else
97           Some(r)
98       }
99 
100     if (res.isEmpty) {
101       if (isDisabled)
102         DisabledRuleExpr // special case as this is probably before the function registry gets called
103       else
104         Literal(null, NullType)
105     } else
106       res.head
107   }

109   /**
110    * Converts both rule expressions and any output expressions removing any coalesceIfAttributeMissing placeholding expressions using names to find each
111    * first expression with no missing names (or replacing with null should all of them be).
112    * @param names
113    * @return
114    */
115   protected[quality] def processCoalesceIfAttributeMissing(rule: Rule, names: Set[String]): Rule =
116     rule match {
117       // rule and output
118       case Rule(id, ExpressionRule(rule: String),
119         iorule @ RunOnPassProcessorImpl(_, _, _, OutputExpression(oruleExpr: String))) if oruleExpr.nonEmpty =>
120         Rule(id, ExpressionRuleExpr(rule, processCoalesceIfAttributeMissing(RuleLogicUtils.expr(rule), names)),
121           iorule.copy(returnIfPassed = OutputExpressionExpr(oruleExpr,
122             processCoalesceIfAttributeMissing(RuleLogicUtils.expr(oruleExpr), names))))
123 
124       // just a rule
125       case orule @ Rule(_, ExpressionRule(rule: String), _) =>
126         orule.copy( expression = ExpressionRuleExpr(rule, processCoalesceIfAttributeMissing(RuleLogicUtils.expr(rule), names)))
127 
128       case _ => rule
129     }
