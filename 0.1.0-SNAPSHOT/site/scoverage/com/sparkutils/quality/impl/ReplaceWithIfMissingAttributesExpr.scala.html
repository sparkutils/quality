15   /**
16    * Processes a given RuleSuite to replace any coalesceIfMissingAttributes.  This may be called before validate / docs but
17    * *must* be called *before* adding the expression to a dataframe.
18    *
19    * @param ruleSuite
20    * @param schema The names to validate against, if empty no attempt to process coalesceIfAttributeMissing will be made
21    * @return
22    */
23   def processIfAttributeMissing(ruleSuite: RuleSuite, schema: StructType = StructType(Seq())) = {
24     val names = LookupIdFunctions.namesFromSchema(schema)
25 
26     val lambdas = ruleSuite.lambdaFunctions.map(lf => ProcessDisableIfMissing.processIfMissingLambdaCoalesce(lf.parsed, names))
27 
28     val wrs = ruleSuite.copy(lambdaFunctions = lambdas)
29 
30     mapRules(wrs) {
31       ProcessDisableIfMissing.processCoalesceIfAttributeMissing(_, names)
32     }
33   }

36   /**
37    * For a given schema evaluates all calls to coalesce in a lambda to swap out missing attributes
38    *
39    * Must be called before any planning - otherwise Spark will already have rejected it.
40    *
41    * @param rule
42    * @return
43    */
44   def processIfMissingLambdaCoalesce(rule: LambdaFunctionParsed, names: Set[String]): LambdaFunction =
45     rule.expr match {
46       case lambdaFunc: SparkLambdaFunction =>
47         lambdaFunc.function match {
48           case func: UnresolvedFunction =>
49             val rexp = processCoalesceIfAttributeMissing(func, names)
50             rule.copy(expr = lambdaFunc.copy(function = rexp))
51 
52           case _ => rule
53         }
54     }

66   /**
67    * Converts expression removing any coalesceIfAttributeMissing placeholding expressions using names to find each
68    * first expression with no missing names (or replacing with null should all of them be).
69    * @param expression
70    * @param names
71    * @return
72    */
73   def processCoalesceIfAttributeMissing(expression: Expression, names: Set[String]): Expression =
74     expression match {
75       case funcExpr: UnresolvedFunction if isReplaceCoalesceName(funcExpr) =>
76         processCoalesceIfAttributeMissing(expression.children, names, isCoalesceDisabled(funcExpr))
77       case e: Expression =>
78         e.withNewChildren(e.children.map(processCoalesceIfAttributeMissing(_, names)))
79     }
80 
81   /**
82    * Returns the first expression that has no missing names or null
83    * @param expressions
84    * @param names
85    * @return
86    */
87   protected def processCoalesceIfAttributeMissing(expressions: Seq[Expression], names: Set[String], isDisabled: Boolean): Expression = {
88     val res =
89       expressions.flatMap{ e =>
90         // make sure no nested children are coalesce's
91         val r = processCoalesceIfAttributeMissing(e, names)
92         // get the names from this expression
93         val explookup = fieldsFromExpression(r)
94         // lookup each of them in the names and if any are not true we can't use this expression
95         if (!explookup.attributesUsed.map(names).forall(identity))
96           None
97         else
98           Some(r)
99       }
100 
101     if (res.isEmpty) {
102       if (isDisabled)
103         DisabledRuleExpr // special case as this is probably before the function registry gets called
104       else
105         Literal(null, NullType)
106     } else
107       res.head
108   }
109 
110   /**
111    * Converts both rule expressions and any output expressions removing any coalesceIfAttributeMissing placeholding expressions using names to find each
112    * first expression with no missing names (or replacing with null should all of them be).
113    * @param names
114    * @return
115    */
116   protected[quality] def processCoalesceIfAttributeMissing(rule: Rule, names: Set[String]): Rule =
117     rule match {
118       // rule and output
119       case Rule(id, ExpressionRule(rule: String),
120         iorule @ RunOnPassProcessorImpl(_, _, _, OutputExpression(oruleExpr: String))) if oruleExpr.nonEmpty =>
121         Rule(id, ExpressionRuleExpr(rule, processCoalesceIfAttributeMissing(RuleLogicUtils.expr(rule), names)),
122           iorule.copy(returnIfPassed = OutputExpressionExpr(oruleExpr,
123             processCoalesceIfAttributeMissing(RuleLogicUtils.expr(oruleExpr), names))))
124 
125       // just a rule
126       case orule @ Rule(_, ExpressionRule(rule: String), _) =>
127         orule.copy( expression = ExpressionRuleExpr(rule, processCoalesceIfAttributeMissing(RuleLogicUtils.expr(rule), names)))
128 
129       case _ => rule
130     }
