42    * Creates a column that runs the RuleSuite.  This also forces registering the lambda functions used by that RuleSuite
43    *
44    * @param ruleSuite The Qualty RuleSuite to evaluate
45    * @param compileEvals Should the rules be compiled out to interim objects - by default true for eval usage, wholeStageCodeGen will evaluate in place
46    * @param resolveWith This experimental parameter can take the DataFrame these rules will be added to and pre-resolve and optimise the sql expressions, see the documentation for details on when to and not to use this. RuleRunner does not currently do wholestagecodegen when resolveWith is used.
47    * @param variablesPerFunc Defaulting to 40, it allows, in combination with variableFuncGroup customisation of handling the 64k jvm method size limitation when performing WholeStageCodeGen.  You _shouldn't_ need it but it's there just in case.
48    * @param variableFuncGroup Defaulting to 20
49    * @param forceRunnerEval Defaulting to false, passing true forces a simplified partially interpreted evaluation (compileEvals must be false to get fully interpreted)
50    * @return A Column representing the Quality DQ expression built from this ruleSuite
51    */
52   def ruleRunnerImpl(ruleSuite: RuleSuite, compileEvals: Boolean = true, resolveWith: Option[DataFrame] = None, variablesPerFunc: Int = 40, variableFuncGroup: Int = 20, forceRunnerEval: Boolean = false): Column = {
53     com.sparkutils.quality.registerLambdaFunctions( ruleSuite.lambdaFunctions )
54     val flattened = flattenExpressions(ruleSuite)
55     val runner = new RuleRunner(RuleLogicUtils.cleanExprs(ruleSuite), PassThrough(flattened), compileEvals, variablesPerFunc, variableFuncGroup, forceRunnerEval)
56     new Column(
57       QualitySparkUtils.resolveWithOverride(resolveWith).map { df =>
58         val resolved = QualitySparkUtils.resolveExpression(df, runner)
59 
60         resolved.asInstanceOf[RuleRunner].copy(child = resolved.children(0) match {
61           // replace the expr
62           case PassThrough(children) => NonPassThrough(children)
63         })
64       } getOrElse runner
65     )
66   }
