11 /**
12  * Replaces any attributes aliased by an expression which is then used in select filters / constraints wth
13  * constituent predicates that can be pushed down.  A simple example is:
14  * {{{
15  *     select as_uuid(lower, higher) as context
16  *     where context = 'string'
17  * }}}
18  * this would be replaced with:
19  * {{{
20  *     select as_uuid(lower, higher) as context
21  *     where (lower = longPairFromUUID('string').lower and higher = longPairFromUUID('string').higher)
22  * }}}
23  * in the case of the string being a constant, constant folding should replace both of these with simple lookups that are easy to push down.
24  *
25  * *NB.* Pushdown predicates based on constant folding only work via the SparkExtension approach, the use of experimental.extraOptimizations does provide this as it happens too late in the plans
26  *
27  * Only In and BinaryComparison's are supported currently
28  */
29 abstract class AsymmetricFilterExpressions extends Rule[LogicalPlan]
30   with PredicateHelperPlus with ConstraintHelper {
31 
32   /**
33    * Should the matchOnExpression be defined in a filter provide the rewrite.  It's possible for joins that both comparedTo AND generating are the
34    * same expression type, this should be explicitly checked for.
35    *
36    * Implementations must return None for any expression they cannot handle, not doing so can lead to MatchErrors.
37    *
38    * @param generating the underlying expression which is generating either an Alias or the direct expression itself
39    * @param comparedTo either an Expression for BinaryComparison where it can be used transparently, irrespective of usage (NamedExpression, Expression etc.), or Seq[Expression] for In's.
40    *                   Where In's are used and the re-write involves structures you should aim to repeat the constituent parts as separate In's in order to get predicate push down
41    * @param filter the Expression used in the filter e.g. EqualTo, In, GreaterThan etc.
42    * @return a rewritten expression tree, logically the opposite of the generating expression or None where generating does not match
43    */
44   def reWriteExpression(generating: Expression, comparedTo: Object, filter: Expression): Option[Expression]
45 
46   // attempt to find / replace any part of the filter tree
47   def transform(expr: Expression, topPlan: LogicalPlan): (Expression, Boolean) = {
48     var found = false
49     (expr.transform {
50       case i@In(expr, list) =>
51         findRootExpression(expr, topPlan).map { lhs =>
52           reWriteExpression(lhs, list, i).map { e => found = true; e }.getOrElse(i)
53         }.getOrElse(i)
54       case e@BinaryComparison(elhs, erhs) =>
55         findRootExpression(elhs, topPlan).flatMap{ a =>
56           val rhs = findRootExpression(erhs, topPlan)
57           rhs.map( b => (a, b) )
58         }.map {
59           case (lhs, rhs) =>
60             // flip the sides as needed
61             reWriteExpression(lhs, rhs, e).orElse(
62               reWriteExpression(rhs, lhs, e)
63             ).map { e => found = true; e }.getOrElse(e)
64         }.getOrElse(e)
65     }, found)
66   }

68   class FilterTransformer(topPlan: LogicalPlan) {
69     def unapply(plan: LogicalPlan): Option[(Expression, LogicalPlan)] =
70       plan match {
71         case Filter(expr, child) =>
72           val (r, found) = transform(expr, topPlan)
73           if (found)
74             Some((r, child))
75           else
76             None
77         case _ => None
78       }
79   }

81   class JoinTransformer(topPlan: LogicalPlan) {
82     def unapply(plan: LogicalPlan): Option[Join] =
83       plan match {
84         case j: Join =>
85           // attempt re-write for join
86           j.condition.flatMap { condition =>
87 
88             val (r, found) = transform(condition, topPlan)
89 
90             if (found)
91               Some(j.copy(condition = Some(r)))
92             else
93               None
94           }
95         case _ => None
96       }
97   }

99   def apply(plan: LogicalPlan): LogicalPlan = {
100     val FilterT = new FilterTransformer(plan)
101     val JoinT = new JoinTransformer(plan)
102     plan transform {
103       case FilterT(resulting, child) =>
104         Filter(resulting, child)
105       case JoinT(join) =>
106         join
107     }

135       // join case (possibly a filter)

139         )

144         )

153

155           And( In(struct(lower, higher), l.map(structg _)),

159         )

161       case (AsUUID(alower, ahigher), AsUUID(blower, bhigher), _: LessThan | _: LessThanOrEqual | _: GreaterThan | _: GreaterThanOrEqual) =>

190       else
191         Literal.TrueLiteral

194   }

197 
198     def orAndFields(useSize: Int): Expression =
199       if (useSize > 1)
200         Or(And(andFields(useSize - 2, leftF, rightF, EqualTo(_: Expression, _: Expression)), create(leftF(useSize-1), rightF(useSize-1))),
201           orAndFields(useSize - 1) )
