14 /**
15  * CODE is based on MutableProjection and generates a transformation between two encoders over a middle operation.
16  * The generated class itself can create new instances directly, unlike Spark projections that need to go through the
17  * codegen source generation cycle, only the compilation is cached.
18  *
19  * If the expression tree contains stateful expressions with codegenfallback the code must be regenerated against a
20  * fresh tree.
21  */

41   def projections(ctx: CodegenContext, expressions: Seq[Expression], mutableRow: String, useSubexprElimination: Boolean = false) = {
42     val validExpr = expressions.zipWithIndex.filter {
43       case (NoOp, _) => false
44       case _ => true
45     }
46     val exprVals = ctx.generateExpressions(validExpr.map(_._1), useSubexprElimination)
47 
48     // 4-tuples: (code for projection, isNull variable name, value variable name, column index)
49     val projectionCodes: Seq[(String, String)] = validExpr.zip(exprVals).map {
50       case ((e, i), ev) =>
51         val value = JavaCode.global(
52           ctx.addMutableState(CodeGenerator.javaType(e.dataType), "value"),
53           e.dataType)
54         val (code, isNull) = if (e.nullable) {
55           val isNull = ctx.addMutableState(CodeGenerator.JAVA_BOOLEAN, "isNull")
56           (s"""
57               |${ev.code}
58               |$isNull = ${ev.isNull};
59               |$value = ${ev.value};
60             """.stripMargin, JavaCode.isNullGlobal(isNull))
61         } else {
62           (s"""
63               |${ev.code}
64               |$value = ${ev.value};
65             """.stripMargin, FalseLiteral)
66         }
67         val update = CodeGenerator.updateColumn(
68           mutableRow,
69           e.dataType,
70           i,
71           ExprCode(isNull, value),
72           e.nullable)
73         (code, update)
74     }
75     projectionCodes
76   }
