103 object LookupIdFunctions {
104 
105   def namesFromSchema(schema: StructType): Set[String] = {
106 
107     def withParent(name: String, parent: String) =
108       if (parent.isEmpty)
109         name
110       else
111         parent + "." + name
112 
113     def accumulate(set: Set[String], schema: StructType, parent: String): Set[String] =
114       schema.foldLeft(set) {
115         (s, field) =>
116           val name = withParent(field.name, parent)
117           field.dataType match {
118             case struct: StructType =>
119               accumulate(s + name, struct, name)
120             case _ => s + name
121           }
122       }
123 
124     accumulate(Set.empty, schema, "")
125   }
126 
127   /**
128    * Use this function to identify which maps / blooms etc. are used by a given rulesuite
129    * collects all rules that are using lookup functions but without constant expressions and the list of lookups that are constants.
130    *
131    */
132   def identifyLookups(ruleSuite: RuleSuite): LookupResults = {
133     val olambdaResults =
134       ruleSuite.lambdaFunctions.flatMap{r =>
135         val exp = RuleLogicUtils.expr(r.rule)
136         LookupIdFunctionImpl.identifyLookups(exp).map((_,r))
137       }.foldLeft(ExpressionLookupResults[Id](Map.empty, Set.empty)) {
138         (acc, res) =>
139           val r = acc.copy( lookupConstants = acc.lookupConstants + (res._2.id -> res._1.constants))
140           if (res._1.hasExpressionLookups)
141             r.copy(lookupExpressions = r.lookupExpressions + res._2.id)
142           else
143             r
144       }
145     val lambdaResults = olambdaResults.copy(lookupConstants = olambdaResults.lookupConstants.filter(_._2.nonEmpty))
146 
147     LookupResults(ruleSuite, ExpressionLookupResults(Map.empty, Set.empty), lambdaResults)
148   }

222    * @return

224   def compareTo[T <: Comparable[T]](left: Any, right: Any): Int =
225     if (left == null && right != null)

251 object Arrays {
252   /**
253    * UnsafeArrayData doesn't allow calling .array, foreach when needed and for others use array
254    *
255    * @param array
256    * @param dataType
257    * @param f
258    * @return
259    */
260   def mapArray[T: ClassTag](array: ArrayData, dataType: DataType, f: Any => T): Array[T] =
261     array match {
262       case _: UnsafeArrayData =>
263         val res = Array.ofDim[T](array.numElements())
264         array.foreach(dataType, (i, v) => res.update(i, f(v)))
265         res
266       case _ => array.array.map(f)
267     }
268 
269   /**
270    * gets an array out of UnsafeArrayData or others
271    * @param array
272    * @param dataType
273    * @return
274    */
275   def toArray(array: ArrayData, dataType: DataType): Array[Any] =
276     array match {
277       case _: UnsafeArrayData =>
278         mapArray(array, dataType, identity)
279       case _ => array.array
280     }
