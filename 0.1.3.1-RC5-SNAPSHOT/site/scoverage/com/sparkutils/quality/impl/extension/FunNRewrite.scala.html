9 /**
10  * Rewrites FunN's when not used as LambdaFunctions (e.g. a parent HigherOrderFunction or HigherOrderFunctionLike).
11  * The application is a simple replacement of the resolved input expression with all occurrences of the
12  * matching LambdaVariable.
13  *
14  * This happens early enough to allow for user functions to be subexpr eliminated (as of Spark 4 not possible
15  * with either CodegenFallback or an expression containing any LambdaVariables).
16  *
17  * IMPORTANT: This only works on 3.2 and above as it introduced transformDownWithPruning
18  */
19 object FunNRewrite extends Rule[LogicalPlan] {
20 
21   override def apply(plan: LogicalPlan): LogicalPlan =
22     QualitySparkUtils.funNRewrite(plan, {
23       case f: FunN if !f.usedAsLambda => // not really needed to guard here, but belts and braces etc.
24         val pairs = f.elementVars.zip(f.arguments).toMap
25         val r =
26           f.function.asInstanceOf[LambdaFunction].function.transform{
27             case e: NamedLambdaVariable if pairs.contains(e) => pairs(e)
28           }
29         r
30     })
31 }
