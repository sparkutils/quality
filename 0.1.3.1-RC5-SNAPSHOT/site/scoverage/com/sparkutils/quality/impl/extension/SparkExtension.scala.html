21 /**
22  * Registers Quality sql functions using the defaults for registerQualityFunctions, these can be overridden without having to subclass DriverPlugin.  Functions are registered via FunctionRegistry.builtIn making them available for view creation, if they should be registered via extensions only then use the quality_force_inject_function = true configuration.
23  *
24  * It also registers plan optimiser rule's such as AsUUIDFilter, which rewrites filters with variables backed by as_uuid.
25  *
26  * Optimiser rules can be disabled via the quality_disable_optimiser_rules system environment variable. "*" disables all rules, otherwise a comma separated list of fqn class names may be used.
27  */
28 class QualitySparkExtension extends ((SparkSessionExtensions) => Unit) with Logging {
29 
30   def parseTypes: String => Option[DataType] = com.sparkutils.quality.impl.RuleRegistrationFunctions.defaultParseTypes _
31   def zero: DataType => Option[Any] = com.sparkutils.quality.impl.RuleRegistrationFunctions.defaultZero _

33   def mapCompare: DataType => Option[(Any, Any) => Int] = (dataType: DataType) => utils.defaultMapCompare(dataType)

52   /**
53    * Adds AsymmetricFilterExpressions for AsUUID
54    * Derived implementations should also call super if they wish to register the default rules (or specify both as extensions).
55    * These are registered after resolution is done.
56    * The first String should be the fqn classname / rule name for the rule
57    * @return
58    */
59   def optimiserRules: Seq[(String, SparkSession => Rule[LogicalPlan])] =
60     Seq((AsUUIDFilter.getClass.getName, _ => AsUUIDFilter), (IDBase64Filter.getClass.getName, _ => IDBase64Filter),
61       (FunNRewrite.getClass.getName, _ => FunNRewrite))
62 
63   override def apply(extensions: SparkSessionExtensions): Unit = {
64     val func =
65       if (com.sparkutils.quality.getConfig(forceInjectFunction, "false").toBoolean) {
66         attemptLogInfo("registering quality functions via injection - they are classed as temporary functions")
67         ShimUtils.registerFunctionViaExtension(extensions) _
68       } else {
69         attemptLogInfo("registering quality functions via builtin function registry - whilst you can use these in global views the extension must always be present")
70         ShimUtils.registerFunctionViaBuiltin _
71       }
72     com.sparkutils.quality.registerQualityFunctions(parseTypes, zero, add, mapCompare, writer,
73       registerFunction = func
74     )
75 
76     if (Testing.testing) {
77       ExtensionTesting.disableRuleResult = ""
78     }
79     val disableConf = com.sparkutils.quality.getConfig(disableRulesConf)
80     if (disableConf != "*") {
81       val disabledRules = disableConf.split(",").map(_.trim).toSet
82       val filteredRules = optimiserRules.filterNot(p => disabledRules.contains(p._1.trim))
83       val str = s"$disableRulesConf = $disabledRules leaving ${filteredRules.map(_._1)} remaining"
84       attemptLogInfo(str)
85       if (Testing.testing) {
86         ExtensionTesting.disableRuleResult = str
87       }
88       filteredRules.map(_._2).foreach(extensions.injectOptimizerRule _)
89     } else {
90       val str = s"All optimiser rules are disabled via $disableRulesConf"
91       attemptLogInfo(str)
92     }
93   }
94 
95 }
