108   /**
109    * Creates a ProcessFactory[I,O], when forceMutable || !compile is true it uses chained MutableProjections,
110    * otherwise, and by default, GenerateDecoderOpEncoderProjection.
111    *
112    * GenerateDecoderOpEncoderProjection is likely far faster on larger rules or higher number of threads for throughput.
113    * Unlike Spark compilation caching, which evaluates codegen for an expression tree every time, the codegen takes place
114    * once, each instance call simply returns a new class instance with fresh state for a thread.  If using forceVarCompilation,
115    * once the number of input fields goes beyond [[maxVarCompilationInputFields]] the code will fallback to INPUT_ROW based processing.
116    *
117    * The chained MutableProjections generate new class code for each instance.  They will also recreate the entire
118    * expression tree if a stateful expression is identified when compile = false
119    *
120    * @param dataFrameFunction
121    * @param compile when false reverts to interpreted mode, when true and forceMutable is true it is recommended to cache the instances
122    * @param forceMutable when true it forces MutableProjection's to be used, compiled or otherwise, the default of false is likely far faster
123    * @param forceVarCompilation defaulting to false it will, when compile is true and forceMutable is false, use variables rather than INPUT_ROW to generate code.
124    *                            Although using true is faster past 10 fields, the compilation approach (as of 0.1.3.1) is experimental
125    *                            and only supported on version OSS 3.2.1 and above
126    * @param toSize specifies the number of fields required to deserialize and create the [[O]]
127    * @tparam I
128    * @tparam O
129    * @return
130    */
131   def processFactory[I: Encoder, O: Encoder](dataFrameFunction: DataFrame => DataFrame, toSize: Int, compile: Boolean = true,
132       forceMutable: Boolean = false, forceVarCompilation: Boolean = false, extraProjection: DataFrame => DataFrame = identity,

136         enableQualityOptimisations = enableQualityOptimisations)
137     else {

140       }
141 
142       val iEnc = implicitly[Encoder[I]]
143       val exprs = QualitySparkUtils.resolveExpressions[I](iEnc, df => {
144         dataFrameFunction(extraProjection(df))
145       })
146 
147       // the code references to the other fields is already present inside of expressions, works for input_row based,
148       // but not wholestage approach, this is performed by all the CodegenSupport execs via attribute lookups
149       val exprsToUse = exprs.drop( exprs.length - toSize)
150 
151       val allOrdinals =
152         exprsToUse.flatMap{
153           e => e.collect {
154             case b: BoundReference => b.ordinal
155           }
156         }.distinct.toSet
157 
158       val projector =
159         if (forceVarCompilation && allOrdinals.size < maxVarCompilationInputFields)
160           GenerateDecoderOpEncoderVarProjection.create[I, O](exprs, toSize, allOrdinals)
161         else
162           GenerateDecoderOpEncoderProjection.generate[I, O](exprs, useSubexprElimination = true, toSize)
163       new ProcessorFactory[I, O] {
164         override def instance: Processor[I, O] = new Processor[I, O] {
165           private val theInstance = projector.newInstance
166           override def apply(i: I): O = theInstance(i)
167           override def setPartition(partition: Int): Unit = theInstance.initialize(partition)
168           override def close(): Unit = {}
169         }
170       }
171     }
172   }
