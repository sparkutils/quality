11   /**
12    * Leverages the foldRunner to replace fields, the input fields are used to create a structure that the rules fold over.
13    * The fields are then dropped from the original Dataframe and added back from the resulting structure.
14    *
15    * NOTE: The field order and types of the original DF will be maintained only when maintainOrder is true.  As it requires access to the schema it may incur extra work.
16    *
17    * @param rules
18    * @param debugMode when true the last results are taken for the replaced fields
19    * @param maintainOrder when true the schema is used to replace fields in the correct location, when false they are simply appended
20    * @param useType In the case you must use select and can't use withColumn you may provide a type directly to stop the NPE
21    * @return
22    */
23   def ifoldAndReplaceFields[P[R] >: DatasetBase[R]](rules: RuleSuite, fields: Either[Seq[String], Seq[(String, Column)]], foldFieldName: String = "foldedFields",
24                            debugMode: Boolean = false, tempFoldDebugName: String = "tempFOLDDEBUG",
25                            maintainOrder: Boolean = true, useType: Option[StructType] = None,
26                            compileEvals: Boolean = false, forceRunnerEval: Boolean = false,
27                            forceTriggerEval: Boolean = false): P[SRow] => P[SRow] = rdf => {
28     val df = rdf.asInstanceOf[DataFrame]
29     import org.apache.spark.sql.functions._
30 
31     val fieldNames = fields.fold(identity, _.map(_._1))
32 
33     val theStruct = fields.fold( fields =>
34       struct(fields.head, fields.tail :_*),
35       pairs =>
36         ShimUtils.column(
37           CreateNamedStruct1(pairs.flatMap(p => Seq(lit(p._1), p._2)).map(ShimUtils.expression(_)))
38         )
39     )
40     val withFolder =
41       /* < 3.2 can't handle select
42       df.select(expr("*"), ruleFolderRunner(rules, theStruct, debugMode = debugMode, useType = useType,
43        compileEvals = compileEvals, forceRunnerEval = forceRunnerEval, forceTriggerEval = forceTriggerEval).as(foldFieldName)) */
44       df.withColumn(foldFieldName, ruleFolderRunner(rules, theStruct, debugMode = debugMode, useType = useType,
45         compileEvals = compileEvals, forceRunnerEval = forceRunnerEval, forceTriggerEval = forceTriggerEval))
46 
47     // create now as the schema will have the folder, which we may want to keep
48     val namesInOrder =
49       if (maintainOrder)
50         withFolder.schema.map(_.name)
51       else
52         Seq()
53     // lift the results
54     val result =
55       if (debugMode)
56         withFolder.drop(fieldNames : _*).selectExpr("*",
57           s"if(size($foldFieldName.result) == 0 or $foldFieldName.result is null, null, element_at($foldFieldName.result, -1)).result as $tempFoldDebugName"
58         ).selectExpr("*", s"$tempFoldDebugName.*").drop(tempFoldDebugName)
59       else
60         withFolder.drop(fieldNames : _*).selectExpr("*", s"$foldFieldName.result.*" )
61 
62     // bring back to top level in the correct order
63     if (maintainOrder)
64       result.select(namesInOrder.head, namesInOrder.tail :_*)
65     else
66       result
67   }.asInstanceOf[P[SRow]]
