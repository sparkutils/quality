16 trait NotATokenParser extends JavaTokenParsers {
17   def breakingTokens: Seq[String]

19   lazy val notAToken: Parser[String] = notAToken()

21   def notAToken(breakingTokens: Seq[String] = breakingTokens, keepWhiteSpace: Boolean = false) =
22     new Parser[String] {
23       def apply(in: Input) = {
24         val source = in.source
25         val offset = in.offset
26         val start =
27           if (keepWhiteSpace)
28             offset
29           else
30             handleWhiteSpace(source, offset)
31         var j = start
32         var break = 0
33         var fromStart = false
34         while (j <= source.length && break == 0) {
35           val breakOption =
36             breakingTokens.find { token =>
37               val len = j - token.length
38               if (len > start) {
39                 val comp = source.subSequence(j - token.length, j)
40 
41                 fromStart = false
42                 token == comp
43               } else
44                 if (j + token.length <= source.length) {
45                   val comp = source.subSequence(j, j + token.length)
46 
47                   fromStart = true
48                   token == comp
49                 } else
50                   false
51 
52             }
53           break = breakOption.fold(0)(_.length)
54 
55           if (break == 0) {
56             j += 1
57           }
58         }
59 
60         if (j >= source.length() && break == 0) {
61           j = source.length() // no breaking token found the while exits due to j consuming the last position
62         }
63 
64         val offsetWouldOOB = ((j - break) - offset - start > 0)
65         val breakWouldOOB = (!(break > 0 && (j - break > start)) && (start == j))
66 
67         if (fromStart && (offsetWouldOOB || breakWouldOOB))
68           Failure("breakingToken found at at start of token parsing", in.drop(start - offset))
69         else
70           if (break > 0 && (j - break > start))
71             Success(source.subSequence(start, j - break).toString, in.drop((j - break) - offset))
72           else
73             if ((offset == start) && (j == start) && break == 0)
74               Failure("no breakingToken found, end of data", in)
75             else
76               Success(source.subSequence(start, j).toString, in.drop(j - offset))
77       }
78     }
79 }

134   /**
135    * Removes any leading tabs/spaces or *'s from the string and any leading characters before the start of the docs
136    * @param str
137    * @return
138    */
139   def cleanDocs(str: String): String = {
140     val pos = str.indexOf("/**")
141     val epos = str.indexOf("*/")
142     if (pos > -1 && epos > -1) {
143       val docstart = str.substring(pos, epos + 2)
144 
145       val lines = docstart.split("\n")
146       val r = "^\\s+\\*".r
147       val endr = "^\\s+\\*/".r
148 
149       // if every one starts with * it's a scaladoc, otherwise it's markdown and leave it alone
150       if (lines.tail.forall(s => r.findPrefixOf(s).isDefined)) {
151         lines.map {
152           case s if endr.findPrefixOf(s).isDefined => s
153           case s => s.replaceAll("^\\s+\\*", "")
154         }.mkString("\n")
155       } else
156         str // single line in the original
157     } else
158       str // not a doc
159   }
160 
161   /**
162    * Parses sql with a leading docs
163    * @param s
164    * @return Some(Docs) when there is a documentation object
165    */
166   def parse( s: String ): Option[Docs] = {
167     val ret = parseAll(phrase(docs), cleanDocs(s))
168     ret match {

172           returnDesc.map(_.str).getOrElse("")
173         ))
174 
175       case NoSuccess(msg, nextInput) =>
176         log.debug(s"DocsParser couldn't pass - $msg")
177         None
178     }
179   }
