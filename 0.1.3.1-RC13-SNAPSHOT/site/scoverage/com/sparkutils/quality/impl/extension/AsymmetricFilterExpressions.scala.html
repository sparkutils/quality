12 /**
13  * Replaces any attributes aliased by an expression which is then used in select filters / constraints wth
14  * constituent predicates that can be pushed down.  A simple example is:
15  * {{{
16  *     select as_uuid(lower, higher) as context
17  *     where context = 'string'
18  * }}}
19  * this would be replaced with:
20  * {{{
21  *     select as_uuid(lower, higher) as context
22  *     where (lower = longPairFromUUID('string').lower and higher = longPairFromUUID('string').higher)
23  * }}}
24  * in the case of the string being a constant, constant folding should replace both of these with simple lookups that are easy to push down.
25  *
26  * *NB.* Pushdown predicates based on constant folding only work via the SparkExtension approach, the use of experimental.extraOptimizations does provide this as it happens too late in the plans
27  *
28  * Only In and BinaryComparison's are supported currently
29  */
30 abstract class AsymmetricFilterExpressions extends Rule[LogicalPlan]
31   with PredicateHelperPlus with ConstraintHelper {
32 
33   /**
34    * Should the matchOnExpression be defined in a filter provide the rewrite.  It's possible for joins that both comparedTo AND generating are the
35    * same expression type, this should be explicitly checked for.
36    *
37    * Implementations must return None for any expression they cannot handle, not doing so can lead to MatchErrors.
38    *
39    * @param generating the underlying expression which is generating either an Alias or the direct expression itself
40    * @param comparedTo either an Expression for BinaryComparison where it can be used transparently, irrespective of usage (NamedExpression, Expression etc.), or Seq[Expression] for In's.
41    *                   Where In's are used and the re-write involves structures you should aim to repeat the constituent parts as separate In's in order to get predicate push down
42    * @param filter the Expression used in the filter e.g. EqualTo, In, GreaterThan etc.
43    * @return a rewritten expression tree, logically the opposite of the generating expression or None where generating does not match
44    */
45   def reWriteExpression(generating: Expression, comparedTo: Object, filter: Expression): Option[Expression]
46 
47   // attempt to find / replace any part of the filter tree
48   def transform(expr: Expression, topPlan: LogicalPlan): (Expression, Boolean) = {
49     var found = false
50     (expr.transform {
51       case i@In(expr, list) =>
52         findRootExpression(expr, topPlan).map { lhs =>
53           reWriteExpression(lhs, list, i).map { e => found = true; e }.getOrElse(i)
54         }.getOrElse(i)
55       case e@BinaryComparison(elhs, erhs) =>
56         findRootExpression(elhs, topPlan).flatMap{ a =>
57           val rhs = findRootExpression(erhs, topPlan)
58           rhs.map( b => (a, b) )
59         }.map {
60           case (lhs, rhs) =>
61             // flip the sides as needed
62             reWriteExpression(lhs, rhs, e).orElse(
63               reWriteExpression(rhs, lhs, e)
64             ).map { e => found = true; e }.getOrElse(e)
65         }.getOrElse(e)
66     }, found)
67   }

69   class FilterTransformer(topPlan: LogicalPlan) {
70     def unapply(plan: LogicalPlan): Option[(Expression, LogicalPlan)] =
71       plan match {
72         case Filter(expr, child) =>
73           val (r, found) = transform(expr, topPlan)
74           if (found)
75             Some((r, child))
76           else
77             None
78         case _ => None
79       }
80   }

82   class JoinTransformer(topPlan: LogicalPlan) {
83     def unapply(plan: LogicalPlan): Option[Join] =
84       plan match {
85         case j: Join =>
86           // attempt re-write for join
87           j.condition.flatMap { condition =>
88 
89             val (r, found) = transform(condition, topPlan)
90 
91             if (found)
92               Some(j.copy(condition = Some(r)))
93             else
94               None
95           }
96         case _ => None
97       }
98   }

100   def apply(plan: LogicalPlan): LogicalPlan = {
101     val FilterT = new FilterTransformer(plan)
102     val JoinT = new JoinTransformer(plan)
103     plan transform {
104       case FilterT(resulting, child) =>
105         Filter(resulting, child)
106       case JoinT(join) =>
107         join
108     }

198
