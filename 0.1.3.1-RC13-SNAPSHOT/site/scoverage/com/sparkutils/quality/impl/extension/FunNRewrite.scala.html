10 /**
11  * Rewrites FunN's when not used as LambdaFunctions (e.g. a parent HigherOrderFunction or HigherOrderFunctionLike).
12  * The application is a simple replacement of the resolved input expression with all occurrences of the
13  * matching LambdaVariable.
14  *
15  * This happens early enough to allow for user functions to be subexpr eliminated (as of Spark 4 not possible
16  * with either CodegenFallback or an expression containing any LambdaVariables).
17  *
18  * The use of USED_AS_LAMBDA and any Spark HoF with a registered LambdaCompilationHandler will keep the expression
19  * as a lambda.  The LambdaCompilationHandler case is a trade-off between overhead of tree copies/compilation in
20  * processors, the underlying hof eval vs compile and sub expression elimination.
21  *
22  * IMPORTANT: This only works on 3.2 and above as it introduced transformDownWithPruning
23  */
24 object FunNRewrite extends Rule[LogicalPlan] {
25 
26   override def apply(plan: LogicalPlan): LogicalPlan =
27     QualitySparkUtils.funNRewrite(plan, {
28       case f: FunN if !f.usedAsLambda &&
29         // if a direct child is a rewrite HoF then we shouldn't disable compilation by ripping it out (#83)
30         !f.children.exists( t => t.collect{
31           case e => compilationHandlers.contains(e.getClass.getName)
32         }.nonEmpty ) =>
33         val pairs = f.elementVars.zip(f.arguments).toMap
34         val r =
35           f.function.asInstanceOf[LambdaFunction].function.transform{
36             case e: NamedLambdaVariable if pairs.contains(e) => pairs(e)
37           }
38         r
39     })
40 }
