41   /**
42    * Are there any stateful expressions in interpreted mode (or fallback) require fresh copies.
43    * Similarly if a hof isn't FunN and doesn't have a rewrite for it we cannot let it through as LambdaVariables are
44    * always stateful.
45    * @param exprs These expressions must be fully resolved, bound and optimised
46    * @param compile when false any stateful expression returns true, by default compilation for stateful CodegenFallback is true
47    * @return
48    */
49   def isCopyNeeded(exprs: Seq[Expression], compile: Boolean = true): Boolean =
50     exprs.exists(_.collect {
51       case e: CodegenFallback if ShimUtils.isStateful(e) =>
52         e
53       case e: HigherOrderFunction if !compilationHandlers.contains(e.getClass.getName) && !e.isInstanceOf[FunN] => e
54     }.nonEmpty ) || ( !compile &&
55       exprs.exists(_.collect {
56         case e: Expression if ShimUtils.isStateful(e)  =>
57           e
58         case e: HigherOrderFunction if !compilationHandlers.contains(e.getClass.getName) && !e.isInstanceOf[FunN] => e
59       }.nonEmpty
60     ))

64    * hit the 65k class namespace limit.

68   /**
69    * Creates a ProcessFactory[I,O], when forceMutable || !compile is true it uses chained MutableProjections,
70    * otherwise, and by default, GenerateDecoderOpEncoderProjection.
71    *
72    * GenerateDecoderOpEncoderProjection is likely far faster on larger rules or higher number of threads for throughput.
73    * Unlike Spark compilation caching, which evaluates codegen for an expression tree every time, the codegen takes place
74    * once, each instance call simply returns a new class instance with fresh state for a thread.  If using forceVarCompilation,
75    * once the number of input fields goes beyond [[maxVarCompilationInputFields]] the code will fallback to INPUT_ROW based processing.
76    *
77    * The chained MutableProjections generate new class code for each instance.  They will also recreate the entire
78    * expression tree if a stateful expression is identified when compile = false
79    *
80    * @param dataFrameFunction
81    * @param compile when false reverts to interpreted mode, when true and forceMutable is true it is recommended to cache the instances
82    * @param forceMutable when true it forces MutableProjection's to be used, compiled or otherwise, the default of false is likely far faster
83    * @param forceVarCompilation defaulting to false it will, when compile is true and forceMutable is false, use variables rather than INPUT_ROW to generate code.
84    *                            Although using true is faster past 10 fields, the compilation approach (as of 0.1.3.1) is experimental
85    *                            and only supported on version OSS 3.2.1 and above
86    * @param toSize specifies the number of fields required to deserialize and create the [[O]]
87    * @tparam I
88    * @tparam O
89    * @return
90    */
91   def processFactory[I: Encoder, O: Encoder](dataFrameFunction: DataFrame => DataFrame, toSize: Int, compile: Boolean = true,
92       forceMutable: Boolean = false, forceVarCompilation: Boolean = false, extraProjection: DataFrame => DataFrame = identity,
93       enableQualityOptimisations: Boolean = true): ProcessorFactory[I, O] = {
94     if (forceMutable || !compile)
95       MutableProjectionProcessor.processFactory[I, O](dataFrameFunction, toSize, compile, extraProjection,
96         enableQualityOptimisations = enableQualityOptimisations)
97     else {
98       if (enableQualityOptimisations) {
99         enableOptimizations(Seq(FunNRewrite, ConstantFolding))
100       }
101 
102       val iEnc = implicitly[Encoder[I]]
103       val exprs = QualitySparkUtils.resolveExpressions[I](iEnc, df => {
104         dataFrameFunction(extraProjection(df))
105       })
106 
107       // the code references to the other fields is already present inside of expressions, works for input_row based,
108       // but not wholestage approach, this is performed by all the CodegenSupport execs via attribute lookups
109       val exprsToUse = exprs.drop( exprs.length - toSize)
110 
111       val allOrdinals =
112         exprsToUse.flatMap{
113           e => e.collect {
114             case b: BoundReference => b.ordinal
115           }
116         }.distinct.toSet
117 
118       val projector =
119         if (forceVarCompilation && allOrdinals.size < maxVarCompilationInputFields)
120           GenerateDecoderOpEncoderVarProjection.create[I, O](exprs, toSize, allOrdinals)
121         else
122           GenerateDecoderOpEncoderProjection.generate[I, O](exprs, useSubexprElimination = true, toSize)
123       new ProcessorFactory[I, O] {
124         override def instance: Processor[I, O] = new Processor[I, O] {
125           private val theInstance = projector.newInstance
126           override def apply(i: I): O = theInstance(i)
127           override def setPartition(partition: Int): Unit = theInstance.initialize(partition)
128           override def close(): Unit = {}
129         }
130       }
131     }
132   }
