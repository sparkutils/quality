39   /**

62   /**
63    * Creates a ProcessFactory[I,O], when forceMutable || !compile is true it uses chained MutableProjections,
64    * otherwise, and by default, GenerateDecoderOpEncoderProjection.
65    *
66    * GenerateDecoderOpEncoderProjection is likely far faster on larger rules or higher number of threads for throughput.
67    * Unlike Spark compilation caching, which evaluates codegen for an expression tree every time, the codegen takes place
68    * once, each instance call simply returns a new class instance with fresh state for a thread.  If using forceVarCompilation,
69    * once the number of input fields goes beyond [[maxVarCompilationInputFields]] the code will fallback to INPUT_ROW based processing.
70    *
71    * The chained MutableProjections generate new class code for each instance.  They will also recreate the entire
72    * expression tree if a stateful expression is identified when compile = false
73    *
74    * @param dataFrameFunction
75    * @param compile when false reverts to interpreted mode, when true and forceMutable is true it is recommended to cache the instances
76    * @param forceMutable when true it forces MutableProjection's to be used, compiled or otherwise, the default of false is likely far faster
77    * @param forceVarCompilation defaulting to false it will, when compile is true and forceMutable is false, use variables rather than INPUT_ROW to generate code.
78    *                            Although using true is faster past 10 fields, the compilation approach (as of 0.1.3.1) is experimental
79    *                            and only supported on version OSS 3.2.1 and above
80    * @param toSize specifies the number of fields required to deserialize and create the [[O]]
81    * @tparam I
82    * @tparam O
83    * @return
84    */
85   def processFactory[I: Encoder, O: Encoder](dataFrameFunction: DataFrame => DataFrame, toSize: Int, compile: Boolean = true,
86       forceMutable: Boolean = false, forceVarCompilation: Boolean = false, extraProjection: DataFrame => DataFrame = identity,
87       enableQualityOptimisations: Boolean = true): ProcessorFactory[I, O] = {
88     if (forceMutable || !compile)
89       MutableProjectionProcessor.processFactory[I, O](dataFrameFunction, toSize, compile, extraProjection,
90         enableQualityOptimisations = enableQualityOptimisations)
91     else {
92       if (enableQualityOptimisations) {
93         enableOptimizations(Seq(FunNRewrite, ConstantFolding))
94       }
95 
96       val iEnc = implicitly[Encoder[I]]
97       val exprs = QualitySparkUtils.resolveExpressions[I](iEnc, df => {
98         dataFrameFunction(extraProjection(df))
99       })
100 
101       // the code references to the other fields is already present inside of expressions, works for input_row based,
102       // but not wholestage approach, this is performed by all the CodegenSupport execs via attribute lookups
103       val exprsToUse = exprs.drop( exprs.length - toSize)
104 
105       val allOrdinals =
106         exprsToUse.flatMap{
107           e => e.collect {
108             case b: BoundReference => b.ordinal
109           }
110         }.distinct.toSet
111 
112       val projector =
113         if (forceVarCompilation && allOrdinals.size < maxVarCompilationInputFields)
114           GenerateDecoderOpEncoderVarProjection.create[I, O](exprs, toSize, allOrdinals)
115         else
116           GenerateDecoderOpEncoderProjection.generate[I, O](exprs, useSubexprElimination = true, toSize)
117       new ProcessorFactory[I, O] {
118         override def instance: Processor[I, O] = new Processor[I, O] {
119           private val theInstance = projector.newInstance
120           override def apply(i: I): O = theInstance(i)
121           override def setPartition(partition: Int): Unit = theInstance.initialize(partition)
122           override def close(): Unit = {}
123         }
124       }
125     }
126   }
