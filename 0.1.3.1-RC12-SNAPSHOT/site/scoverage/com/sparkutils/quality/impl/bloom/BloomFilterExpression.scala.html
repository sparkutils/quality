44   /**
45    * Identifies blooms from either a resolved or unresolved expression
46    *
47    * @param expression a single Expression, can be a complex tree or simple direct probabilityIn or rowid
48    * @return The bloom id's used, for unresolved expression trees this may contain blooms which are not present in the bloom map
49    */
50   def getBlooms(expression: Expression): Seq[String] = {
51     val res = Seq.empty
52 
53     def bloom(expression: Expression, acc: Seq[String]): Seq[String] = {
54 
55       // case one
56       expression match {
57         case UnresolvedFunction4(name, Seq(left, Literal(id: UTF8String, StringType)), _, _) if name.replaceAll("_", "").toLowerCase == "probabilityin" => bloom(left, acc :+ id.toString)
58         case BloomFilterLookupExpression(left, Literal(id: UTF8String, StringType), _) => bloom(left, acc :+ id.toString)
59         case e: Expression => e.children.foldLeft(acc) { (acc, e) => bloom(e, acc) }
60         case _ => acc
61       }
62     }
63 
64     bloom(expression, res)
65   }
66 
67 }
