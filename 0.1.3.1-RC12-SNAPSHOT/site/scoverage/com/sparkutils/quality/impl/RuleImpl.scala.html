64   def expr(rule: String) = {
65     val parser = SparkSession.getActiveSession.map(_.sessionState.sqlParser).getOrElse {
66       newParser()
67     }
68     /*
69     attempt for a simple expression, then try a plan with exactly one output row.
70     "In" will handle ListQuery, "Exists" similarly exists, everything else will likely fail as a parser error
71      */
72     val rawExpr =
73       try {
74         parser.parseExpression(rule)
75       } catch {
76         case ot: Throwable => // e.g. suitable for output expressions
77           try {
78             ScalarSubquery(parser.parsePlan(rule))
79           } catch {
80             case _: Throwable =>
81               // quite possibly a lambda using a subquery so try and force it via ( sub ) trick,
82               // but allow us to replace later after resolution.  DBR > 12 requires () on simple expressions as well
83               val r = rule.split("->")
84               val wrapped =
85                 if (r.size == 2)
86                   s"${r(0)} -> ( ${r(1)} )"
87                 else
88                   s"( ${r(0)} )"
89 
90               try {
91                 parser.parseExpression(wrapped)
92               } catch {
93                 case _: Throwable => throw ot // if this didn't work return the original error
94               }
95           }
96       }
97 
98     val res =
99       rawExpr match {
100         case l: SparkLambdaFunction if hasSubQuery(l) =>
101           // The lambda's will be parsed as UnresolvedAttributes and not the needed lambdas
102           val names = l.arguments.map(a => a.name -> a).toMap
103           l.transformUp {
104             case s: SubqueryExpression =>
105               s.withNewPlan( s.plan.transform{
106               case snippet =>
107                 snippet.transformAllExpressions {
108                   case a: UnresolvedAttribute =>
109                     names.get(a.name).map(lamVar => lamVar).getOrElse(a)
110                 }
111               }
112             )
113           }
114         case _ => rawExpr
115       }
116 
117     res.transformUp {
118       case s: SubqueryExpression => SubQueryWrapper(Seq(s))
119     }
