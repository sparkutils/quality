61   def expr(rule: String) = {
62     val parser = SparkSession.getActiveSession.map(_.sessionState.sqlParser).getOrElse {
63       newParser()
64     }
65     /*
66     attempt for a simple expression, then try a plan with exactly one output row.
67     "In" will handle ListQuery, "Exists" similarly exists, everything else will likely fail as a parser error
68      */
69     val rawExpr =
70       try {
71         parser.parseExpression(rule)
72       } catch {
73         case ot: Throwable => // e.g. suitable for output expressions
74           try {
75             ScalarSubquery(parser.parsePlan(rule))
76           } catch {
77             case _: Throwable =>
78               // quite possibly a lambda using a subquery so try and force it via ( sub ) trick,
79               // but allow us to replace later after resolution.  DBR > 12 requires () on simple expressions as well
80               val r = rule.split("->")
81               val wrapped =
82                 if (r.size == 2)
83                   s"${r(0)} -> ( ${r(1)} )"
84                 else
85                   s"( ${r(0)} )"
86 
87               try {
88                 parser.parseExpression(wrapped)
89               } catch {
90                 case _: Throwable => throw ot // if this didn't work return the original error
91               }
92           }
93       }
94 
95     val res =
96       rawExpr match {
97         case l: SparkLambdaFunction if hasSubQuery(l) =>
98           // The lambda's will be parsed as UnresolvedAttributes and not the needed lambdas
99           val names = l.arguments.map(a => a.name -> a).toMap
100           l.transformUp {
101             case s: SubqueryExpression =>
102               s.withNewPlan( s.plan.transform{
103               case snippet =>
104                 snippet.transformAllExpressions {
105                   case a: UnresolvedAttribute =>
106                     names.get(a.name).map(lamVar => lamVar).getOrElse(a)
107                 }
108               }
109             )
110           }
111         case _ => rawExpr
112       }
113 
114     res.transformUp {
115       case s: SubqueryExpression => SubQueryWrapper(Seq(s))
116     }
117   }
