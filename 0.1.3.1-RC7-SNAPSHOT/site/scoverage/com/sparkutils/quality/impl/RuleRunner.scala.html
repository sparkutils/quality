60     val flattened = flattenExpressions(ruleSuite)
61     val cleaned = RuleLogicUtils.cleanExprs(ruleSuite)
62     val input =
63       // ExpressionProxy and SubExprEvaluationRuntime cannot be used with compileEvals
64       if (compileEvals)
65         PassThroughCompileEvals( flattened )
66       else
67         PassThroughEvalOnly( flattened )
68 
69     val runner =
70       if (forceRunnerEval || resolveWith.isDefined)
71         new RuleRunnerEval(cleaned, input,
72           compileEvals, variablesPerFunc, variableFuncGroup)
73       else
74         new RuleRunner(cleaned, input,
75           compileEvals, variablesPerFunc, variableFuncGroup)
76 
77     column(
78       QualitySparkUtils.resolveWithOverride(resolveWith).map { df =>
79         val resolved = QualitySparkUtils.resolveExpression(df, runner)

81         resolved.withNewChildren(Seq(resolved.children.head match {
82           // replace the expr
83           case PassThroughCompileEvals(children) => NonPassThrough(children)
84           case PassThroughEvalOnly(children) => NonPassThrough(children)
