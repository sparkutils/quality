37   /**
38    * Creates an aggregate by applying filter to rows, calling sum with a starting value (provided by zero) and finally calls result to process the sum and count values for a final result.
39    *
40    * Note, when working with lambda's in the dsl it's often required to use the dataframes col function as the scope is incorrect in the lambda.
41    *
42    * @param sumType the type used to sum across rows
43    * @param filter filter only input rows interesting to count (similar to CountIf, SumIf)
44    * @param sum add to the current sum, takes the current sum as the parameter, sum_with, inc, map_with etc. can be used as implementations
45    * @param result processes the sum result and row count (after filtering) to produce the final result of the aggregate
46    * @param zero the default value for a given sumType
47    * @param add the default addition logic for a given sumType, which combines the sumType across partitions
48    * @return
49    */
50   def agg_expr(sumType: DataType, filter: Column, sum: SumExpression, result: ResultsExpression,

52                add: DataType => Option[(Expression, Expression) => Expression] = (dataType: DataType) => defaultAdd(dataType)): Column =
53     column( AggregateExpressions(sumType, expression(filter), sum.funN(sumType), result.funN(sumType), zero, add, notYetResolved = true) )
