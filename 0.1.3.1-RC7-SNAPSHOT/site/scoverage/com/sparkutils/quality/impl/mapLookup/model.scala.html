54   private def mapFromDF(id: String, df: DataFrame, key: Column, value: Column, broadcastFunction: MapData => Broadcast[MapData] =
55     SparkSession.active.sparkContext.broadcast(_)) = {
56     val translated = df.select(key.as("key"), value.as("value"))
57     val map = translated.toLocalIterator().asScala.map {
58       mapPair =>
59         toCatalyst(mapPair.get(0)) ->
60           toCatalyst(mapPair.get(1))
61     }.toMap
62 
63     val mapData: MapData = ArrayBasedMapData(map)
64     id -> (broadcastFunction(mapData), translated.schema.last.dataType)
65   }
