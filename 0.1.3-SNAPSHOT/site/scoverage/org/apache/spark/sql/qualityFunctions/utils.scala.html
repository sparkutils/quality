37     val left = oleft.asInstanceOf[ArrayData]
38     val right = oright.asInstanceOf[ArrayData]
39     if (left.numElements() > right.numElements())
40       1
41     else if (left.numElements() < right.numElements())
42       -1
43     else {
44       // same
45       var i = 0
46       var res = 0
47       val al = (idx: Int) => left.get(idx, elementType)// Arrays.toArray(left, elementType)
48       val ar = (idx: Int) => right.get(idx, elementType)// Arrays.toArray(right, elementType)
49       while (i < left.numElements() && res == 0) {
50         res = elComp(al(i), ar(i))

52       }

57   def defaultMapCompare(dataType: DataType, extension: DataType => Option[(Any, Any) => Int] = noopCompare): Option[(Any, Any) => Int] =
58     dataType match {
59       case mt: MapType => {
60         val kt = defaultMapCompare(mt.keyType, extension).
61           getOrElse(sys.error(s"Could not find compare function for map key type ${mt.keyType}"))
62 
63         val kr = BoundReference(0, mt.keyType, nullable = true)
64 
65         // actual field is in a key val struct
66         Some(
67           genArrayCompare(keyValueType(mt.keyType, mt.valueType),
68             (left, right) => {
69               val lkv = kr.eval(left.asInstanceOf[InternalRow])
70               val rkv = kr.eval(right.asInstanceOf[InternalRow])
71 
72               val r = kt(lkv, rkv)
73               if (r != 0)
74                 r
75               else {
76                 /* shouldn't happen unless MapData is created directly and it's not
77                    actually a map checking on key uniqueness, here for completeness */
78                 val vt = defaultMapCompare(mt.valueType, extension).
79                   getOrElse(sys.error(s"Could not find compare function for map value type ${mt.valueType}"))
80                 val vr = BoundReference(1, mt.valueType, nullable = true)
81 
82                 val lvv = vr.eval(left.asInstanceOf[InternalRow])
83                 val rvv = vr.eval(right.asInstanceOf[InternalRow])
84 
85                 vt(lvv, rvv)
86               }
87             })
88         )
89       }
90       case dt: AtomicType => Some(compareToOrdering(QualitySparkUtils.sparkOrdering(dt)))
91       case arrayType: ArrayType =>
92         val oelcomp = defaultMapCompare(arrayType.elementType, extension)
93         oelcomp.map{
94           elcomp =>
95             genArrayCompare(arrayType.elementType, elcomp)
96         }
97       case structType: StructType =>
98         val allFields = structType.fields.zipWithIndex.map{
99           case (f, index) =>
100             defaultMapCompare( f.dataType, extension).map(cf => (BoundReference(index, f.dataType, nullable = true), cf)).
101               getOrElse(sys.error(s"Could not find compare function for ${f.dataType} in struct $structType"))
102         }
103         Some(
104           // fail early for any non 0
105           (left, right) => {
106             var i = 0
107             var res = 0
108             while (i < allFields.size && res == 0) {
109               val leftV = allFields(i)._1.eval(left.asInstanceOf[InternalRow])
110               val rightV = allFields(i)._1.eval(right.asInstanceOf[InternalRow])
111               res = allFields(i)._2(leftV, rightV)
112               i += 1
113             }
114             res
115           }
116         )
117       case _ => extension(dataType).orElse(
118         sys.error(s"Could not find compare function for ${dataType}")
119       )
120     }
