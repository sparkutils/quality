51   /**
52    * Attempts to load all the views present in the config.  If a view is already registered in that name it will be replaced.
53    * @param viewConfigs
54    * @return the names of views which have been replaced
55    */
56   def loadViews(viewConfigs: Seq[ViewConfig]): ViewLoadResults = {
57 
58     // assume some will not load, attempt count is used to stop cycles, 2x should be enough
59     var attemptCount = 0
60     var done = false
61     var leftToProcess = viewConfigs.map(t => t.name -> t).toMap
62 
63     val replaced = mutable.Set.empty[String]
64     // if we are missing a relation and the name is declared try to load that first
65     var mapOf = leftToProcess
66     var processed = mutable.Set.empty[String]
67 
68     def processView(viewPair: ViewConfig): Unit =
69       if (attemptCount < (viewConfigs.size * 2)) {
70         try {
71           val (name, config) = (viewPair.name, viewPair)
72           if (Validation.defaultViewLookup(name)) {
73             replaced += name
74           }
75 
76           config.source.fold(identity, SparkSession.active.sql(_))
77            .createOrReplaceTempView(name)
78 
79           // it worked, remove it
80           leftToProcess = leftToProcess - name
81           processed = processed + name
82         } catch {
83           case ae: AnalysisException =>
84             val res = ViewLoader.tableOrViewNotFound(ae)
85             val sql = viewPair.source.right.getOrElse("")
86 
87             res.fold(a => throw ViewLoaderAnalysisException(a, s"AnalysisException for view ${viewPair.name}: $sql", viewPair.name, sql),
88               views => {
89                 val missingNames =
90                   views.flatMap { name =>
91                     val lookupName =
92                       if (mapOf.contains(name))
93                         name
94                       else
95                         s"`$name`"
96 
97                     if (mapOf.contains(lookupName)) { // quoted must also be in the view name if it's got minus' etc., on Spark < 3.2 this will be incorrectly unquoted
98                       attemptCount += 1
99                       processView(mapOf(lookupName))
100                       None
101                     } else // not one we can actually do anything about
102                       Some(name)
103                   }
104                 if (!missingNames.isEmpty) {
105                   throw MissingViewAnalysisException(ae, s"Missing relations for view ${viewPair.name}: $missingNames used in sql $sql", viewPair.name, sql, missingNames)
106                 }
107               }
108             )
109         }
110       }
111 
112     while((attemptCount < (viewConfigs.size * 2)) && !done){
113       attemptCount += 1
114       leftToProcess.headOption.fold {
115         done = true;
116       } { p =>
117         processView(p._2)
118       }
119     }
120     ViewLoadResults(replaced.toSet, !done, leftToProcess.keySet)
121   }
122

124   def tableOrViewNotFound(ae: AnalysisException): Either[AnalysisException, Set[String]] =
125     ae.plan.fold[Either[AnalysisException, Set[String]]]{
126       // spark 2.4 just has exception: Table or view not found: names
127       if (ae.message.contains("Table or view not found"))
128         Right(Set(ae.message.split(":")(1).trim))
129       else
130         Left(ae)
131     } {
132       plan =>
133         val c =
134           plan.collect {
135             case ur: UnresolvedRelation =>
136               ur.tableName
137           }
138 
139         if (c.isEmpty)
140           Left(ae) // not what we expected
141         else
142           Right(c.toSet)
143     }
