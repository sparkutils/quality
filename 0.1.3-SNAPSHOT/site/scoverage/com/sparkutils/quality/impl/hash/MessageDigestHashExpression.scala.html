29   override def digest: Array[Long] = {
30     val bytes = messageDigest.digest
31     val res = Array.ofDim[Long]((bytes.length + 7) / 8) // next whole long size up
32 
33     // to make sure the array is large enough pad as needed, so a 32bit digest will always take 64
34     val use =
35       if (bytes.length / 8 != res.length) {
36         val nbytes = Array.ofDim[Byte](res.length * 8)
37         Array.copy(bytes, 0, nbytes, 0, bytes.length)
38         nbytes
39       } else
40         bytes
41 
42     for{ i <- 0 until res.length} {
43       res(i) = BytePackingUtils.unencodeLong(i * 8, use)
44     }
45

47   }
