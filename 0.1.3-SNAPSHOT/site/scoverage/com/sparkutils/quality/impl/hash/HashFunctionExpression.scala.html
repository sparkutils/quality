26   override def digest: Array[Long] = {
27     val bytes = hasher.hash.asBytes
28     val res = Array.ofDim[Long]((bytes.length + 7) / 8) // next whole long size up
29 
30     // to make sure the array is large enough pad as needed, so a 32bit digest will always take 64
31     val use =
32       if (bytes.length / 8 != res.length) {
33         val nbytes = Array.ofDim[Byte](res.length * 8)
34         Array.copy(bytes, 0, nbytes, 0, bytes.length)
35         nbytes
36       } else
37         bytes
38 
39     for{ i <- 0 until res.length} {
40       res(i) = BytePackingUtils.unencodeLong(i * 8, use)
41     }
42

44   }
