22   def breakingTokens: Set[String]

24   lazy val notAToken =
25     new Parser[String] {
26       def apply(in: Input) = {
27         val source = in.source
28         val offset = in.offset
29         val start = handleWhiteSpace(source, offset)
30         var j = start
31         var break = 0
32         var fromStart = false
33         while (j < source.length && break == 0) {
34           val breakOption =
35             breakingTokens.find { token =>
36               val len = j - token.length
37               if (len > start) {
38                 val comp = source.subSequence(j - token.length, j)
39 
40                 fromStart = false
41                 token == comp
42               } else
43                 if (j + token.length < source.length) {
44                   val comp = source.subSequence(j, j + token.length)
45 
46                   fromStart = true
47                   token == comp
48                 } else
49                   false
50 
51             }
52           break = breakOption.fold(0)(_.length)
53 
54           if (break == 0) {
55             j += 1
56           }
57         }
58 
59         if (break > 0 && fromStart)
60           Failure("breakingToken found at at start of token parsing", in.drop(start - offset))
61         else
62           if (break > 0)
63             Success(source.subSequence(start, j - break).toString, in.drop((j - break) - offset))
64           else
65             Success(source.subSequence(start, j).toString, in.drop(j - offset))
66       }
67     }
68 }
