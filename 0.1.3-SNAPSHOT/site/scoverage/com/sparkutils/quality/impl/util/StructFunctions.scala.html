192 object UpdateFields {
193   private def nameParts(fieldName: String): Seq[String] = {
194     require(fieldName != null, "fieldName cannot be null")
195 
196     if (fieldName.isEmpty) {
197       fieldName :: Nil
198     } else {
199       UnresolvedAttribute.parseAttributeName(fieldName)
200       //CatalystSqlParser.parseMultipartIdentifier(fieldName)
201     }
202   }
203 
204   /**
205    * Adds/replaces field of `StructType` into `col` expression by name.
206    */
207   def apply(col: Expression, fieldName: String, expr: Expression): UpdateFields =
208     updateFieldsHelper(col, nameParts(fieldName), name => WithField(name, expr))
209 
210   /**
211    * Drops fields of `StructType` in `col` expression by name.
212    */
213   def apply(col: Expression, fieldName: String): UpdateFields =
214     updateFieldsHelper(col, nameParts(fieldName), name => DropField(name))
215 
216   private def updateFieldsHelper(
217                                   structExpr: Expression,
218                                   namePartsRemaining: Seq[String],
219                                   valueFunc: String => StructFieldsOperation) : UpdateFields = {
220     val fieldName = namePartsRemaining.head
221     if (namePartsRemaining.length == 1) {
222       UpdateFields(Seq(structExpr, valueFunc(fieldName)))
223     } else {
224       val newStruct = if (structExpr.resolved) {
225         val resolver = SQLConf.get.resolver
226         ExtractValue(structExpr, Literal(fieldName), resolver)
227       } else {
228         UnresolvedExtractValue(structExpr, Literal(fieldName))
229       }
230 
231       val newValue = updateFieldsHelper(
232         structExpr = newStruct,
233         namePartsRemaining = namePartsRemaining.tail,
234         valueFunc = valueFunc)
235       UpdateFields(Seq(structExpr, WithField(fieldName, newValue) ))
236     }
237   }
