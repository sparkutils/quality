14  * Parser which adds token based string reading
15  */
16 trait NotATokenParser extends JavaTokenParsers {
17   def breakingTokens: Set[String]
18 
19   lazy val notAToken =
20     new Parser[String] {
21       def apply(in: Input) = {
22         val source = in.source
23         val offset = in.offset
24         val start = handleWhiteSpace(source, offset)
25         var j = start
26         var break = 0
27         var fromStart = false
28         while (j < source.length && break == 0) {
29           val breakOption =
30             breakingTokens.find { token =>
31               val len = j - token.length
32               if (len > start) {
33                 val comp = source.subSequence(j - token.length, j)
34 
35                 fromStart = false
36                 token == comp
37               } else
38                 if (j + token.length < source.length) {
39                   val comp = source.subSequence(j, j + token.length)
40 
41                   fromStart = true
42                   token == comp
43                 } else
44                   false
45 
46             }
47           break = breakOption.fold(0)(_.length)
48 
49           if (break == 0) {
50             j += 1
51           }
52         }
53 
54         if (break > 0 && fromStart)
55           Failure("breakingToken found at at start of token parsing", in.drop(start - offset))
56         else
57           if (break > 0)
58             Success(source.subSequence(start, j - break).toString, in.drop((j - break) - offset))
59           else
60             Success(source.subSequence(start, j).toString, in.drop(j - offset))
61       }
62     }
63 }

116   /**
117    * Removes any leading tabs/spaces or *'s from the string and any leading characters before the start of the docs
118    * @param str
119    * @return
120    */
121   def cleanDocs(str: String): String = {
122     val pos = str.indexOf("/**")
123     val epos = str.indexOf("*/")
124     if (pos > -1 && epos > -1) {
125       val docstart = str.substring(pos, epos + 2)
126 
127       val lines = docstart.split("\n")
128       val r = "^\\s+\\*".r
129       val endr = "^\\s+\\*/".r
130 
131       // if every one starts with * it's a scaladoc, otherwise it's markdown and leave it alone
132       if (lines.tail.forall(s => r.findPrefixOf(s).isDefined)) {
133         lines.map {
134           case s if endr.findPrefixOf(s).isDefined => s
135           case s => s.replaceAll("^\\s+\\*", "")
136         }.mkString("\n")
137       } else
138         str // single line in the original
139     } else
140       str // not a doc
141   }
142 
143   /**
144    * Parses sql with a leading docs
145    * @param s
146    * @return Some(Docs) when there is a documentation object
147    */
148   def parse( s: String ): Option[Docs] = {
149     val ret = parseAll(phrase(docs), cleanDocs(s) + " ") // trailing space acts a termination char
150     ret match {

154           result._1._2.map(_.str).getOrElse("")
155         ))
156 
157       case NoSuccess(msg, nextInput) =>
158         log.debug(s"DocsParser couldn't pass - $msg")
159         None
160     }
