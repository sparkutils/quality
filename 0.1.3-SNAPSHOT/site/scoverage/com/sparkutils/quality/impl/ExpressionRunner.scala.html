22   /**
23    * Runs the ruleSuite expressions saving results as a tuple of (ruleResult: yaml, resultType: String)
24    *
25    * @param ruleSuite
26    * @param name
27    * @return
28    */
29   def apply(ruleSuite: RuleSuite, name: String = "expressionResults", renderOptions: Map[String, String] = Map.empty, ddlType: String = ""): Column = {
30     com.sparkutils.quality.registerLambdaFunctions( ruleSuite.lambdaFunctions )
31     val expressions = flattenExpressions(ruleSuite)
32     val collectExpressions =
33       if (ddlType.isEmpty)
34         expressions.map( i => YamlEncoderExpr(i, renderOptions))
35       else
36         expressions
37 
38     val ddl_type =
39       if (ddlType.isEmpty)
40         expressionResultTypeYaml
41       else
42         DataType.fromDDL(ddlType)
43 
44     new Column(ExpressionRunner(RuleLogicUtils.cleanExprs(ruleSuite), collectExpressions, ddl_type)).as(name)
45   }
46 
47   protected[quality] def expressionsResultToRow[R](ruleSuiteResult: GeneralExpressionsResult[R]): InternalRow =
48     InternalRow(
49       packId(ruleSuiteResult.id),
50       ArrayBasedMapData(
51         ruleSuiteResult.ruleSetResults, packId, (a: Any) => {
52           val v = a.asInstanceOf[Map[VersionedId, GeneralExpressionResult]]
53           ArrayBasedMapData(
54             v, packId, (a: Any) => a match {
55               case r: GeneralExpressionResult =>
56                 InternalRow(UTF8String.fromString( r.ruleResult ), UTF8String.fromString( r.resultDDL) )
57               case s: String =>  UTF8String.fromString( s )
58               case r: R => r
59             }
60           )
61         }
62       )
63     )
64 }
