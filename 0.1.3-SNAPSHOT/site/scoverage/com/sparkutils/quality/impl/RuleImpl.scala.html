53   /**
54    * Same as functions.expr without the wrapping Column
55    * @param rule
56    * @return
57    */
58   def expr(rule: String) = {
59     val parser = SparkSession.getActiveSession.map(_.sessionState.sqlParser).getOrElse {
60       QualitySparkUtils.newParser()
61     }
62     /*
63     attempt for a simple expression, then try a plan with exactly one output row.
64     "In" will handle ListQuery, "Exists" similarly exists, everything else will likely fail as a parser error
65      */
66     val rawExpr =
67       try {
68         parser.parseExpression(rule)
69       } catch {
70         case ot: Throwable => // e.g. suitable for output expressions
71           try {
72             ScalarSubquery(parser.parsePlan(rule))
73           } catch {
74             case _: Throwable =>
75               // quite possibly a lambda using a subquery so try and force it via ( sub ) trick,
76               // but allow us to replace later after resolution.  DBR > 12 requires () on simple expressions as well
77               val r = rule.split("->")
78               val wrapped =
79                 if (r.size == 2)
80                   s"${r(0)} -> ( ${r(1)} )"
81                 else
82                   s"( ${r(0)} )"
83 
84               try {
85                 parser.parseExpression(wrapped)
86               } catch {
87                 case _: Throwable => throw ot // if this didn't work return the original error
88               }
89           }
90       }
91 
92     val res =
93       rawExpr match {
94         case l: SparkLambdaFunction if hasSubQuery(l) =>
95           // The lambda's will be parsed as UnresolvedAttributes and not the needed lambdas
96           val names = l.arguments.map(a => a.name -> a).toMap
97           l.transform {
98             case s: SubqueryExpression => s.withNewPlan( s.plan.transform{
99               case snippet =>
100                 snippet.transformAllExpressions {
101                   case a: UnresolvedAttribute =>
102                     names.get(a.name).map(lamVar => lamVar).getOrElse(a)
103                 }
104             })
105           }
106         case _ => rawExpr
107       }
108 
109     res
110   }

116   def anyToRuleResult(any: Any): RuleResult =
117     any match {
118       case b: Boolean => if (b) Passed else Failed
119       case 0 | 0.0 => Failed
120       case 1 | 1.0 => Passed
121       case -1 | -1.0 | "softfail" | "maybe" | "SoftFail" | "softFail" | "Softfail" | "Maybe" => SoftFailed
122       case -2  | -2.0 | "disabledrule" | "disabled" | "DisabledRule" | "disabledRule" | "Disabledrule" => DisabledRule
123       case d: Double => Probability(d) // only spark 2 unless configured to behave like spark 2
124       case d: Decimal => Probability(d.toDouble)
125       case "Passed" | "true" | "True" | "passed" | "Pass"| "pass" | "Yes" | "yes" | "1" | "1.0" => Passed
126       case "Failed" | "false" | "False" | "failed" | "Fail"| "fail" | "No" | "no" | "0" | "0.0" => Failed
127       case _ => Failed // anything else is a fail
128     }
