50       ruleSetId.as("ruleSetId").cast(IntegerType),

60     val versionedRules = df.sparkSession.sql(
61       """
62        select ruleExpr, ruleId, ruleVersion, ruleEngineSalience, ruleEngineId, ruleEngineVersion,
63        ruleSetId, ruleSetVersion, -- don't need to bump versions as they can coexist
64         suiteversions.ruleSuiteId, suiteversions.ruleSuiteVersion -- force the versions to be bumped to latest ruleSuite Versions
65         from
66          (select distinct ruleSuiteId, ruleSuiteVersion from rules) suiteversions join
67          rules l0 on l0.ruleSuiteId = suiteversions.ruleSuiteId and l0.ruleSuiteVersion <= suiteversions.ruleSuiteVersion
68          where
69           not exists (
70             select 0 from rules l1
71             where l1.ruleSuiteId = l0.ruleSuiteId
72              and l1.ruleId = l0.ruleId
73              and l1.ruleSetId = l0.ruleSetId
74              and l1.ruleSuiteVersion <= suiteversions.ruleSuiteVersion
75              and l1.ruleVersion > l0.ruleVersion
76           )
77           and l0.ruleExpr != "DELETED"
78        """)

135   protected[quality] def lambdaOutputSQL(tableName: String, extra: String ="") =
136       s"""
137        select $extra ruleExpr, functionId, functionVersion,
138         versions.ruleSuiteId, versions.ruleSuiteVersion -- force the versions to be bumped to latest ruleSuiteVersions
139         from
140          (select distinct ruleSuiteId, ruleSuiteVersion from $tableName) versions join
141          $tableName l0 on l0.ruleSuiteId = versions.ruleSuiteId and l0.ruleSuiteVersion <= versions.ruleSuiteVersion
142          where
143           not exists (
144             select 0 from $tableName l1
145             where l1.ruleSuiteId = l0.ruleSuiteId and
146              l1.functionId = l0.functionId and
147              l1.ruleSuiteVersion <= versions.ruleSuiteVersion
148              and l1.functionVersion > l0.functionVersion
149           )
150           and l0.ruleExpr != "DELETED"
151        """

189   protected[quality] case class SameOrNextVersionLower[T](map: Map[Id, Seq[T]]) extends Function1[Id, Option[Seq[T]]] {
190     implicit val ordering = new Ordering[Id] {
191       def compare(x: Id, y: Id): Int = {
192         if (x == y)
193           0
194         else
195           if (x.id != y.id)
196             x.id - y.id
197           else
198             x.version - y.version
199       }
200     }
201 
202     val sorted = TreeMap[Id, Seq[T]]() ++ map
203 
204     override def apply(id: Id): Option[Seq[T]] =
205       sorted.to(id).lastOption.map(_._2)
206   }
