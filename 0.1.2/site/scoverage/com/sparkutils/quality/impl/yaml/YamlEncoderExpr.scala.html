140   private def createMapNode(mt: MapType, map: MapData, keyType: Any => Node, valueType: Any => Node)(implicit renderOptions: Map[String, String]) =
141     if (map == null)
142       createNullNode
143     else {
144       val smap = MapUtils.toScalaMap(map, mt.keyType, mt.valueType)
145       val tuples =

155   private def createStructNode(st: StructType)(implicit renderOptions: Map[String, String]) = {
156     val converters = st.fields.map(f =>
157       makeStructFieldConverter.applyOrElse(f.dataType, makeStructFieldConverterExt))
158 
159     (row: InternalRow) => {
160       if (row == null)

163         val tuples = st.fields.zipWithIndex.map {

165             new NodeTuple(

169         }

171       }

230   override def eval(inputRow: InternalRow): Any = {
231     val input = child.eval(inputRow)
232 
233     val writer = new StringWriter()
234     val options = new DumperOptions()
235     import org.yaml.snakeyaml.DumperOptions
236     options.setDefaultFlowStyle(DumperOptions.FlowStyle.FLOW)
237 
238     val resolver = new Resolver
239     val node = valueConverter(input)
240 
241     // 1.24 and 1.33 aren't compatible interface for serialize so Yaml.serialize(node, output) isn't an option
242     val serializer = new Serializer(new Emitter(writer, options), resolver, options, null)
243     try {
244       serializer.open()
245       serializer.serialize(node)
246       serializer.close()
247     } catch {
248       case e: IOException =>
249         throw new YAMLException(e)
250     }
251 
252     val str = writer.getBuffer.toString
253     writer.close()
254     UTF8String.fromString(str)
255   }
