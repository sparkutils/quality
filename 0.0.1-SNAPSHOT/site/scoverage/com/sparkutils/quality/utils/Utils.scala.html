60 trait LookupIdFunctions {
61 
62   def namesFromSchema(schema: StructType): Set[String] = {
63 
64     def withParent(name: String, parent: String) =
65       if (parent.isEmpty)
66         name
67       else
68         parent + "." + name
69 
70     def accumulate(set: Set[String], schema: StructType, parent: String): Set[String] =
71       schema.foldLeft(set) {
72         (s, field) =>
73           val name = withParent(field.name, parent)
74           field.dataType match {
75             case struct: StructType =>
76               accumulate(s + name, struct, name)
77             case _ => s + name
78           }
79       }
80 
81     accumulate(Set.empty, schema, "")
82   }
83 
84   /**
85    * Use this function to identify which maps / blooms etc. are used by a given rulesuite
86    * collects all rules that are using lookup functions but without constant expressions and the list of lookups that are constants.
87    *
88    */
89   def identifyLookups(ruleSuite: RuleSuite): LookupResults = {
90     val olambdaResults =
91       ruleSuite.lambdaFunctions.flatMap{r =>
92         val exp = RuleLogicUtils.expr(r.rule)
93         LookupIdFunctionImpl.identifyLookups(exp).map((_,r))
94       }.foldLeft(ExpressionLookupResults[Id](Map.empty, Set.empty)) {
95         (acc, res) =>
96           val r = acc.copy( lookupConstants = acc.lookupConstants + (res._2.id -> res._1.constants))
97           if (res._1.hasExpressionLookups)
98             r.copy(lookupExpressions = r.lookupExpressions + res._2.id)
99           else
100             r
101       }
102     val lambdaResults = olambdaResults.copy(lookupConstants = olambdaResults.lookupConstants.filter(_._2.nonEmpty))
103 
104     LookupResults(ruleSuite, ExpressionLookupResults(Map.empty, Set.empty), lambdaResults)
105   }
106 }
