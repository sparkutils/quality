79     dataType match {
80       case _: MapType => Some((left, right) => MapMerge(Seq(left, right), (dataType: DataType) => defaultAdd(dataType, extension)))
81       case _: IntegerType | LongType | DoubleType =>
82         Some((left, right) => add(left, right, null))
83       case a: DecimalType =>
84         Some((left, right) => add(left, right, a))
85       case _ => extension(dataType)
86     }

124       if (paramNumbers.isEmpty && minimum == -1 )

128           if ((paramNumbers.nonEmpty && !paramNumbers.contains(exps.size)) || (minimum > exps.size)) {
129             val sizeerr =
130               if (paramNumbers.nonEmpty)

134             throw QualityException(s"Wrong number of arguments provided to Quality function $name. $sizeerr")
135           }
136           argsf(exps)
137         }
138     registerFunction(name, create)
139     if (!mustKeepNames(name)) {
140       registerFunction(name.replaceAll("_",""), create)
141     }
