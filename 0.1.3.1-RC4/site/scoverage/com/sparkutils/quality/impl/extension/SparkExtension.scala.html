21 /**
22  * Registers Quality sql functions using the defaults for registerQualityFunctions, these can be overridden without having to subclass DriverPlugin.  Functions are registered via FunctionRegistry.builtIn making them available for view creation, if they should be registered via extensions only then use the quality_force_inject_function = true configuration.
23  *
24  * It also registers plan optimiser rule's such as AsUUIDFilter, which rewrites filters with variables backed by as_uuid.
25  *
26  * Optimiser rules can be disabled via the quality_disable_optimiser_rules system environment variable. "*" disables all rules, otherwise a comma separated list of fqn class names may be used.
27  */
28 class QualitySparkExtension extends ((SparkSessionExtensions) => Unit) with Logging {

30   def parseTypes: String => Option[DataType] = com.sparkutils.quality.impl.RuleRegistrationFunctions.defaultParseTypes _

52   /**

54    * Derived implementations should also call super if they wish to register the default rules (or specify both as extensions).
55    * These are registered after resolution is done.
56    * The first String should be the fqn classname / rule name for the rule
57    * @return
58    */
59   def optimiserRules: Seq[(String, SparkSession => Rule[LogicalPlan])] =
60     Seq((AsUUIDFilter.getClass.getName, _ => AsUUIDFilter), (IDBase64Filter.getClass.getName, _ => IDBase64Filter))
61 
62   override def apply(extensions: SparkSessionExtensions): Unit = {
63     val func =
64       if (com.sparkutils.quality.getConfig(forceInjectFunction, "false").toBoolean) {
65         attemptLogInfo("registering quality functions via injection - they are classed as temporary functions")
66         ShimUtils.registerFunctionViaExtension(extensions) _
67       } else {
68         attemptLogInfo("registering quality functions via builtin function registry - whilst you can use these in global views the extension must always be present")
69         ShimUtils.registerFunctionViaBuiltin _
70       }
71     com.sparkutils.quality.registerQualityFunctions(parseTypes, zero, add, mapCompare, writer,
72       registerFunction = func
73     )
74 
75     if (Testing.testing) {
76       ExtensionTesting.disableRuleResult = ""
77     }
78     val disableConf = com.sparkutils.quality.getConfig(disableRulesConf)
79     if (disableConf != "*") {
80       val disabledRules = disableConf.split(",").map(_.trim).toSet
81       val filteredRules = optimiserRules.filterNot(p => disabledRules.contains(p._1.trim))
82       val str = s"$disableRulesConf = $disabledRules leaving ${filteredRules.map(_._1)} remaining"
83       attemptLogInfo(str)
84       if (Testing.testing) {
85         ExtensionTesting.disableRuleResult = str
86       }
87       filteredRules.map(_._2).foreach(extensions.injectOptimizerRule _)
88     } else {
89       val str = s"All optimiser rules are disabled via $disableRulesConf"
90       attemptLogInfo(str)
91     }
92   }

94 }
