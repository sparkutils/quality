43   /**
44    * Identifies blooms from either a resolved or unresolved expression
45    *
46    * @param expression a single Expression, can be a complex tree or simple direct probabilityIn or rowid
47    * @return The bloom id's used, for unresolved expression trees this may contain blooms which are not present in the bloom map
48    */
49   def getBlooms(expression: Expression): Seq[String] = {
50     val res = Seq.empty
51 
52     def bloom(expression: Expression, acc: Seq[String]): Seq[String] = {
53 
54       // case one
55       expression match {
56         case UnresolvedFunction4(name, Seq(left, Literal(id: UTF8String, StringType)), _, _) if name.replaceAll("_", "").toLowerCase == "probabilityin" => bloom(left, acc :+ id.toString)
57         case BloomFilterLookupExpression(left, Literal(id: UTF8String, StringType), _) => bloom(left, acc :+ id.toString)
58         case e: Expression => e.children.foldLeft(acc) { (acc, e) => bloom(e, acc) }
59         case _ => acc
60       }
61     }
62 
63     bloom(expression, res)
64   }
65 
66 }
