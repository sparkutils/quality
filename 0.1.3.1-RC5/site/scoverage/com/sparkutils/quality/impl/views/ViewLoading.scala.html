49   /**
50    * Attempts to load all the views present in the config.  If a view is already registered in that name it will be replaced.
51    * @param viewConfigs
52    * @return the names of views which have been replaced
53    */
54   def loadViews(viewConfigs: Seq[ViewConfig]): ViewLoadResults = {
55 
56     // assume some will not load, attempt count is used to stop cycles, 2x should be enough
57     var attemptCount = 0
58     var done = false
59     var leftToProcess = viewConfigs.map(t => t.name -> t).toMap
60 
61     val replaced = mutable.Set.empty[String]
62     // if we are missing a relation and the name is declared try to load that first
63     var mapOf = leftToProcess
64     var processed = mutable.Set.empty[String]
65 
66     def processView(viewPair: ViewConfig): Unit =
67       if (attemptCount < (viewConfigs.size * 2)) {
68         try {
69           val (name, config) = (viewPair.name, viewPair)
70           if (Validation.defaultViewLookup(name)) {
71             replaced += name
72           }
73 
74           config.source.fold(identity, SparkSession.active.sql(_))
75            .createOrReplaceTempView(name)
76 
77           // it worked, remove it
78           leftToProcess = leftToProcess - name
79           processed = processed + name
80         } catch {
81           // 3.5 introduced two different exceptions, ParserException and ExtendedAnalysisException
82           // AnalysisException no longer has the plan
83           case e: Exception =>
84             val res = ShimUtils.tableOrViewNotFound(e).getOrElse(throw e)
85             val sql = viewPair.source.right.getOrElse("")
86 
87             res.fold(a => throw ViewLoaderAnalysisException(a, s"AnalysisException for view ${viewPair.name}: $sql", viewPair.name, sql),
88               views => {
89                 val missingNames =
90                   views.flatMap { name =>
91                     val lookupName =
92                       if (mapOf.contains(name))
93                         name
94                       else
95                         s"`$name`"
96 
97                     if (mapOf.contains(lookupName)) { // quoted must also be in the view name if it's got minus' etc., on Spark < 3.2 this will be incorrectly unquoted
98                       attemptCount += 1
99                       processView(mapOf(lookupName))
100                       None
101                     } else // not one we can actually do anything about
102                       Some(name)
103                   }
104                 if (!missingNames.isEmpty) {
105                   throw MissingViewAnalysisException(e, s"Missing relations for view ${viewPair.name}: $missingNames used in sql $sql", viewPair.name, sql, missingNames)
106                 }
107               }
108             )
109         }
110       }
111 
112     while((attemptCount < (viewConfigs.size * 2)) && !done){
113       attemptCount += 1
114       leftToProcess.headOption.fold {
115         done = true;
116       } { p =>
117         processView(p._2)
118       }
119     }
120     ViewLoadResults(replaced.toSet, !done, leftToProcess.keySet)
121   }
