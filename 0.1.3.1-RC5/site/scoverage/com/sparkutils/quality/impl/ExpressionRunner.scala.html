22   /**
23    * Runs the ruleSuite expressions saving results as a tuple of (ruleResult: yaml, resultType: String)
24    *
25    */
26   def apply(ruleSuite: RuleSuite, name: String = "expressionResults", renderOptions: Map[String, String] = Map.empty, ddlType: String = "",
27             variablesPerFunc: Int = 40, variableFuncGroup: Int = 20, forceRunnerEval: Boolean = false, compileEvals: Boolean = true): Column = {
28     com.sparkutils.quality.registerLambdaFunctions( ruleSuite.lambdaFunctions )
29     val expressions = flattenExpressions(ruleSuite)
30     val collectExpressions =
31       if (ddlType.isEmpty)
32         expressions.map( i => YamlEncoderExpr(i, renderOptions))
33       else
34         expressions
35     val exprs =
36       // ExpressionProxy and SubExprEvaluationRuntime cannot be used with compileEvals
37       if (compileEvals)
38         PassThroughCompileEvals(collectExpressions)
39       else
40         PassThroughEvalOnly(collectExpressions)
41 
42     val ddl_type =
43       if (ddlType.isEmpty)
44         expressionResultTypeYaml
45       else
46         DataType.fromDDL(ddlType)
47 
48     val cleaned = RuleLogicUtils.cleanExprs(ruleSuite)
49 
50     ShimUtils.column(
51       if (forceRunnerEval)
52         new ExpressionRunnerEval(cleaned, exprs,
53           ddl_type, variablesPerFunc = variablesPerFunc, variableFuncGroup = variableFuncGroup,
54           compileEvals = compileEvals)
55       else
56         new ExpressionRunner(cleaned, exprs,
57           ddl_type, variablesPerFunc = variablesPerFunc, variableFuncGroup = variableFuncGroup,
58           compileEvals = compileEvals)
59     ).as(name)
60   }
61 }
