24    *
25    * @param ruleSuite
26    * @param name

29   def apply(ruleSuite: RuleSuite, name: String = "expressionResults", renderOptions: Map[String, String] = Map.empty): Column = {
30     com.sparkutils.quality.registerLambdaFunctions( ruleSuite.lambdaFunctions )
31     val expressions = flattenExpressions(ruleSuite)
32     val collectExpressions = expressions.map( i => YamlEncoderExpr(i, renderOptions))
33     new Column(ExpressionRunner(ruleSuite, collectExpressions)).as(name)
34   }
