29   def applyFunN(ff: FunForward, transformFunction: (LambdaFunction, NamedExpression, Expression) => Expression =
30     (fun, nvl, exp) =>
31       fun.function.transform {
32         case unvl: NamedLambdaVariable if
33           nvl.exprId == unvl.exprId =>
34           exp
35       } ): FunN = {
36       val placeHolders = ff.params.collect{ case r: RefExpression => r.index }.toSet
37       val nonPlaceHolders = ff.function.arguments.zipWithIndex.foldLeft(ff.function.function.asInstanceOf[LambdaFunction]){ // always a LambdaFunction given you can't partially apply a 0 arg expression
38         case (fun, (exp, index) ) =>
39           // if it's an nvl leave alone
40           if (placeHolders(index))
41             fun
42           else {
43             val nvl = fun.arguments(index)
44             fun.copy(function = transformFunction(fun, nvl, exp))
45           }
46       }
47       ff.function.copy(function = nonPlaceHolders.copy( arguments =
48         nonPlaceHolders.arguments.zipWithIndex.filter{case (_, index) => placeHolders(index)}.map(_._1) ) )
