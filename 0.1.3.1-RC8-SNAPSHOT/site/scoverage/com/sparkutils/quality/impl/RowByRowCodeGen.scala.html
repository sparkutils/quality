14 /**
15  * CODE is based on MutableProjection and generates a transformation between two encoders over a middle operation.
16  * The generated class itself can create new instances directly, unlike Spark projections that need to go through the
17  * codegen source generation cycle, only the compilation is cached.
18  *
19  * If the expression tree contains stateful expressions with codegenfallback the code must be regenerated against a
20  * fresh tree.
21  */

37   def projections(ctx: CodegenContext, expressions: Seq[Expression], mutableRow: String, useSubexprElimination: Boolean = false) = {
38     val validExpr = expressions.zipWithIndex.filter {
39       case (NoOp, _) => false
40       case _ => true
41     }
42     val exprVals = ctx.generateExpressions(validExpr.map(_._1), useSubexprElimination)
43 
44     // 4-tuples: (code for projection, isNull variable name, value variable name, column index)
45     val projectionCodes: Seq[(String, String)] = validExpr.zip(exprVals).map {
46       case ((e, i), ev) =>
47         val value = JavaCode.global(
48           ctx.addMutableState(CodeGenerator.javaType(e.dataType), "value"),
49           e.dataType)
50         val (code, isNull) = if (e.nullable) {
51           val isNull = ctx.addMutableState(CodeGenerator.JAVA_BOOLEAN, "isNull")
52           (s"""
53               |${ev.code}
54               |$isNull = ${ev.isNull};
55               |$value = ${ev.value};
56             """.stripMargin, JavaCode.isNullGlobal(isNull))
57         } else {
58           (s"""
59               |${ev.code}
60               |$value = ${ev.value};
61             """.stripMargin, FalseLiteral)
62         }
63         val update = CodeGenerator.updateColumn(
64           mutableRow,
65           e.dataType,
66           i,
67           ExprCode(isNull, value),
68           e.nullable)
69         (code, update)
70     }
71     projectionCodes
72   }

91       case s: PlanExpression[_] => true

98     val (resTypeIsStruct, resType) =

103         (false, null)

105     val ctx = newCodeGenContext()
106 
107 
108     ctx.INPUT_ROW = "enc"
109     val encProjectionCodes = projections(ctx, exprFrom, "encRow")

112     val encUpdates = ctx.splitExpressionsWithCurrentInputs(encProjectionCodes.map(_._2))
113     val encSubExprs = ctx.subexprFunctionsCode

116 
117     val projectionCodes = projections(ctx, expressions, "mutableRow", useSubexprElimination)
118 
119     // Evaluate all the subexpressions.
120     val evalSubexpr = ctx.subexprFunctionsCode.replace(encSubExprs, "")
121 
122     val allProjections = ctx.splitExpressionsWithCurrentInputs(projectionCodes.map(_._1))
123     val allUpdates = ctx.splitExpressionsWithCurrentInputs(projectionCodes.map(_._2))

127

132 
133     val codeBody = s"""
134       public java.lang.Object generate(Object[] references) {
135         return new SpecificMutableProjection(references);
136       }
137 
138       class SpecificMutableProjection extends ${classOf[DecoderOpEncoderProjection[I,O]].getName}<Object, Object> {
139 
140         private Object[] references;
141         private InternalRow inRow;

147 
148         public SpecificMutableProjection(Object[] references) {
149           this.references = references;
150           mutableRow = new $genericMutableRowType(${expressions.size});

152           encRow = new $genericMutableRowType(${exprFrom.length});
153           decRow = new $genericMutableRowType(${toSize});
154           ${
155             if (toSize == 1 && resTypeIsStruct)

159           }
160           ${ctx.initMutableStates()}
161         }
162 
163         public ${classOf[DecoderOpEncoderProjection[I,O]].getName} newInstance() {

165         }
166 
167         public void initialize(int partitionIndex) {

169         }
170 
171         //public ${implicitly[Encoder[O]].clsTag.runtimeClass.getName} apply(${implicitly[Encoder[I]].clsTag.runtimeClass.getName} _i) {
172         public java.lang.Object apply(java.lang.Object _i) {
173           inRow.update(0, _i);
174           InternalRow enc = (InternalRow) inRow;
175           // enc subexprs
176           $encSubExprs
177           // enc projections

179           // enc updates
180           $encUpdates

183           // eval sub exprs

193           ${
194             if (toSize == 1 && resTypeIsStruct)
195               s"interim = mutableRow.getStruct(${expressions.length - 1}, ${resType.length});"
196             else
197               (for(i <- 0 until toSize) yield

207           // dec updates

218     val code = CodeFormatter.stripOverlappingComments(

222     val (clazz, _) = CodeGenerator.compile(code)

224     if (isCopyNeeded(expressions))

234 
235         override def initialize(partitionIndex: Int): Unit = initial.initialize(partitionIndex)
236       }
237     else
238       // with no stateful CodegenFallback's the generated newInstance function is correct
239       initial
240   }
