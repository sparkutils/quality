42                        debugMode: Boolean = false, resolveWith: Option[DataFrame] = None, variablesPerFunc: Int = 40,
43                        variableFuncGroup: Int = 20, forceRunnerEval: Boolean = false, forceTriggerEval: Boolean = false): Column = {

45     val realType =
46       if (debugMode)
47       // wrap it in an array with the priority result
48       ArrayType(StructType(Seq(StructField("salience", IntegerType), StructField("result", resultDataType))))
49         else
50         resultDataType

52     val (expressions, indexes) = flattenExpressions(ruleSuite)
53 
54     val cleaned = RuleLogicUtils.cleanExprs(ruleSuite)
55     val exprs =

63     val runner =
