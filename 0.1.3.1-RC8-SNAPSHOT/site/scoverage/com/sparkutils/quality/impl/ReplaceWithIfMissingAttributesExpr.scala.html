17   /**
18    * Processes a given RuleSuite to replace any coalesceIfMissingAttributes.  This may be called before validate / docs but
19    * *must* be called *before* adding the expression to a dataframe.
20    *
21    * @param ruleSuite
22    * @param schema The names to validate against, if empty no attempt to process coalesceIfAttributeMissing will be made
23    * @return
24    */
25   def processIfAttributeMissing(ruleSuite: RuleSuite, schema: StructType = StructType(Seq())) = {
26     val names = LookupIdFunctions.namesFromSchema(schema)
27 
28     val lambdas = ruleSuite.lambdaFunctions.map(lf => ProcessDisableIfMissing.processIfMissingLambdaCoalesce(lf.parsed, names))
29 
30     val wrs = ruleSuite.copy(lambdaFunctions = lambdas)
31 
32     mapRules(wrs) {
33       ProcessDisableIfMissing.processCoalesceIfAttributeMissing(_, names)
34     }
35   }

38   /**
39    * For a given schema evaluates all calls to coalesce in a lambda to swap out missing attributes
40    *
41    * Must be called before any planning - otherwise Spark will already have rejected it.
42    *
43    * @param rule
44    * @return
45    */
46   def processIfMissingLambdaCoalesce(rule: LambdaFunctionParsed, names: Set[String]): LambdaFunction =
47     rule.expr match {
48       case lambdaFunc: SparkLambdaFunction =>
49         lambdaFunc.function match {
50           case func: UnresolvedFunction =>
51             val rexp = processCoalesceIfAttributeMissing(func, names)
52             rule.copy(expr = lambdaFunc.copy(function = rexp))
53 
54           case _ => rule
55         }
56     }

68   /**
69    * Converts expression removing any coalesceIfAttributeMissing placeholding expressions using names to find each
70    * first expression with no missing names (or replacing with null should all of them be).
71    * @param expression
72    * @param names
73    * @return
74    */
75   def processCoalesceIfAttributeMissing(expression: Expression, names: Set[String]): Expression =
76     expression match {
77       case funcExpr: UnresolvedFunction if isReplaceCoalesceName(funcExpr) =>
78         processCoalesceIfAttributeMissing(expression.children, names, isCoalesceDisabled(funcExpr))
79       case e: Expression =>
80         e.withNewChildren(e.children.map(processCoalesceIfAttributeMissing(_, names)))
81     }
82 
83   /**
84    * Returns the first expression that has no missing names or null
85    * @param expressions
86    * @param names
87    * @return
88    */
89   protected def processCoalesceIfAttributeMissing(expressions: Seq[Expression], names: Set[String], isDisabled: Boolean): Expression = {
90     val res =
91       expressions.flatMap{ e =>
92         // make sure no nested children are coalesce's
93         val r = processCoalesceIfAttributeMissing(e, names)
94         // get the names from this expression
95         val explookup = fieldsFromExpression(r)
96         // lookup each of them in the names and if any are not true we can't use this expression
97         if (!explookup.attributesUsed.map(names).forall(identity))
98           None
99         else
100           Some(r)
101       }
102 
103     if (res.isEmpty) {
104       if (isDisabled) DisabledRuleExpr // special case as this is probably before the function registry gets called
105       else
106         Literal(null, NullType)
107     } else
108       res.head
109   }
110 
111   /**
112    * Converts both rule expressions and any output expressions removing any coalesceIfAttributeMissing placeholding expressions using names to find each
113    * first expression with no missing names (or replacing with null should all of them be).
114    * @param names
115    * @return
116    */
117   protected[quality] def processCoalesceIfAttributeMissing(rule: Rule, names: Set[String]): Rule =
118     rule match {
119       // rule and output
120       case Rule(id, ExpressionRule(rule: String),
121         iorule @ RunOnPassProcessorImpl(_, _, _, OutputExpression(oruleExpr: String))) if oruleExpr.nonEmpty =>
122         Rule(id, ExpressionRuleExpr(rule, processCoalesceIfAttributeMissing(RuleLogicUtils.expr(rule), names)),
123           iorule.copy(returnIfPassed = OutputExpressionExpr(oruleExpr,
124             processCoalesceIfAttributeMissing(RuleLogicUtils.expr(oruleExpr), names))))
125 
126       // just a rule
127       case orule @ Rule(_, ExpressionRule(rule: String), _) =>
128         orule.copy( expression = ExpressionRuleExpr(rule, processCoalesceIfAttributeMissing(RuleLogicUtils.expr(rule), names)))
129 
130       case _ => rule
131     }
