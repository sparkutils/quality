47   /**
48    * Creates a column that runs the RuleSuite.  This also forces registering the lambda functions used by that RuleSuite
49    *
50    * @param ruleSuite The Qualty RuleSuite to evaluate
51    * @param compileEvals Should the rules be compiled out to interim objects - by default false, allowing optimisations
52    * @param resolveWith This experimental parameter can take the DataFrame these rules will be added to and pre-resolve and optimise the sql expressions, see the documentation for details on when to and not to use this. RuleRunner does not currently do wholestagecodegen when resolveWith is used.
53    * @param variablesPerFunc Defaulting to 40, it allows, in combination with variableFuncGroup customisation of handling the 64k jvm method size limitation when performing WholeStageCodeGen.  You _shouldn't_ need it but it's there just in case.
54    * @param variableFuncGroup Defaulting to 20
55    * @param forceRunnerEval Defaulting to false, passing true forces a simplified partially interpreted evaluation (compileEvals must be false to get fully interpreted)
56    * @return A Column representing the Quality DQ expression built from this ruleSuite
57    */
58   def ruleRunnerImpl(ruleSuite: RuleSuite, compileEvals: Boolean = false, resolveWith: Option[DataFrame] = None,
59                      variablesPerFunc: Int = 40, variableFuncGroup: Int = 20, forceRunnerEval: Boolean = false): Column = {
60     com.sparkutils.quality.registerLambdaFunctions( ruleSuite.lambdaFunctions )
61     val flattened = flattenExpressions(ruleSuite)
62     val cleaned = RuleLogicUtils.cleanExprs(ruleSuite)
63     val input =
64       // ExpressionProxy and SubExprEvaluationRuntime cannot be used with compileEvals
65       if (compileEvals)
66         PassThroughCompileEvals( flattened )
67       else
68         PassThroughEvalOnly( flattened )
69 
70     val runner =
71       if (forceRunnerEval || resolveWith.isDefined)
72         new RuleRunnerEval(cleaned, input,
73           compileEvals, variablesPerFunc, variableFuncGroup)
74       else
75         new RuleRunner(cleaned, input,
76           compileEvals, variablesPerFunc, variableFuncGroup)
77 
78     column(
79       QualitySparkUtils.resolveWithOverride(resolveWith).map { df =>
80         val resolved = QualitySparkUtils.resolveExpression(df, runner)
81 
82         resolved.withNewChildren(Seq(resolved.children.head match {
83           // replace the expr
84           case PassThroughCompileEvals(children) => NonPassThrough(children)
85           case PassThroughEvalOnly(children) => NonPassThrough(children)
86         }))
87       } getOrElse runner
88     )
89   }

284 /**
285  * Children will be rewritten by the plan, either by resolveWith, it's then re-incorporated into ruleSuite
286  *
287  * The variablePerFunc (40) and variableFuncGroup (20) parameters are chosen due to 64k method limitation,
288  * is null for all of a 1600 column row blasts the 64k jvm limit.  You may customise these values if a given rulesuite
289  * requires it.
290  *
291  * @param variablesPerFunc How many variables are in a function
292  * @param variableFuncGroup How many functions are then grouped into a new function

306   lazy val realChildren =
307     child match {
308       case r @ NonPassThrough(_) => r.rules
309       case PassThroughCompileEvals(children) => children
